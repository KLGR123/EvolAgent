from datetime import datetime

# First, let's inspect any existing research files to understand current status
print("Inspecting existing research files in workspace...")
print("=" * 60)

import os
workspace_files = [f for f in os.listdir('workspace/') if f.endswith('.json')]
print(f"Found {len(workspace_files)} JSON files in workspace:")
for file in workspace_files:
    print(f"- {file}")

# Let's inspect the most recent comprehensive file if it exists
if workspace_files:
    latest_file = max(workspace_files, key=lambda x: os.path.getmtime(f'workspace/{x}'))
    print(f"\nInspecting latest file: {latest_file}")
    
    try:
        with open(f'workspace/{latest_file}', 'r', encoding='utf-8') as f:
            existing_data = json.load(f)
        
        print("File structure:")
        for key in existing_data.keys():
            value = existing_data[key]
            if isinstance(value, list):
                print(f"- {key}: list with {len(value)} items")
            elif isinstance(value, dict):
                print(f"- {key}: dict with {len(value)} keys")
            else:
                print(f"- {key}: {type(value).__name__}")
    except Exception as e:
        print(f"Error reading file: {e}")
        existing_data = {}
else:
    existing_data = {}

print("\n" + "=" * 90)

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.5',
    'Accept-Encoding': 'gzip, deflate',
    'Connection': 'keep-alive',
    'Upgrade-Insecure-Requests': '1',
}

# Define comprehensive list of Chinese stone carving heritage sites
heritage_sites = [

    {

    }

# Define search keywords
heritage_keywords = [
    'unesco world heritage', 'world heritage site', 'unesco',
    'intangible cultural heritage', 'cultural heritage',
    'heritage list', 'world heritage list', 'heritage status',
    'world heritage committee', 'heritage designation'
]

stone_carving_keywords = [
    'stone carving', 'rock carving', 'stone sculpture',

# Store comprehensive findings
heritage_findings = []

print("\n🎯 SYSTEMATIC RESEARCH OF CHINA'S STONE CARVING HERITAGE SITES:")
print("=" * 70)

for i, site in enumerate(heritage_sites, 1):
    print(f"\n🏛️ Site {i}/{len(heritage_sites)}: {site['name']}")
    print(f"Region: {site['region']}")
    print(f"Type: {site['type']}")
    print(f"Expected Status: {site['expected_status']}")
    print(f"URL: {site['url']}")
    print("-" * 70)
    
    try:
        print(f"Accessing {site['name']}...")
        response = requests.get(site['url'], headers=headers, timeout=30)
        
        if response.status_code == 200:
            print(f"✅ Successfully accessed (Status: {response.status_code})")
            
            # Parse HTML content
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Get page title
            title = soup.find('title')
            page_title = title.get_text().strip() if title else 'No title'
            print(f"Page title: {page_title}")
            
            # Remove script and style elements
            for element in soup(["script", "style"]):
                element.decompose()
            
            # Get clean text
            text = soup.get_text()
            lines = (line.strip() for line in text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
            clean_text = ' '.join(chunk for chunk in chunks if chunk)
            
            print(f"Content length: {len(clean_text)} characters")
            
            # Analyze content for heritage and stone carving information
            text_lower = clean_text.lower()
            
            # Find heritage indicators
            found_heritage_keywords = [kw for kw in heritage_keywords if kw in text_lower]
            found_stone_keywords = [kw for kw in stone_carving_keywords if kw in text_lower]
            
            print(f"🏆 Heritage keywords found: {found_heritage_keywords}")
            print(f"🗿 Stone carving keywords found: {found_stone_keywords}")
            
            # Extract relevant sentences with FIXED variable naming
            if found_heritage_keywords and found_stone_keywords:
                print("\n🎯 HIGHLY RELEVANT HERITAGE SITE CONFIRMED!")
                
                # Split text into sentences
                sentences = re.split(r'[.!?]+', clean_text)
                unesco_sentences = []
                heritage_sentences = []
                stone_carving_sentences = []
                
                for sentence in sentences:
                    sentence_clean = sentence.strip()
                    if len(sentence_clean) > 20:  # Avoid very short fragments
                        sentence_lower = sentence_clean.lower()  # CONSISTENT variable naming
                        
                        # UNESCO-specific sentences
                        if 'unesco' in sentence_lower and ('world heritage' in sentence_lower or 'heritage site' in sentence_lower):
                            unesco_sentences.append(sentence_clean)
                        
                        # Heritage-related sentences
                        elif any(kw in sentence_lower for kw in heritage_keywords):
                            heritage_sentences.append(sentence_clean)
                        
                        # Stone carving technique sentences
                        elif any(kw in sentence_lower for kw in stone_carving_keywords):
                            stone_carving_sentences.append(sentence_clean)
                
                print(f"\n📝 UNESCO SENTENCES EXTRACTED ({len(unesco_sentences)} found):")
                for j, sentence in enumerate(unesco_sentences[:3], 1):
                    print(f"{j}. {sentence}")
                
                print(f"\n🏛️ HERITAGE SENTENCES EXTRACTED ({len(heritage_sentences)} found):")
                for j, sentence in enumerate(heritage_sentences[:2], 1):
                    print(f"{j}. {sentence}")
                
                print(f"\n🗿 STONE CARVING SENTENCES EXTRACTED ({len(stone_carving_sentences)} found):")
                for j, sentence in enumerate(stone_carving_sentences[:2], 1):
                    print(f"{j}. {sentence}")
                
                # Determine heritage status
                unesco_confirmed = 'unesco world heritage' in text_lower or 'world heritage site' in text_lower
                heritage_status = 'UNESCO World Heritage Site' if unesco_confirmed else 'Cultural Heritage Site'
                
                print(f"\n✅ HERITAGE STATUS CONFIRMED: {heritage_status}")
                
                # Store comprehensive finding
                site_finding = {
                    'name': site['name'],
                    'region': site['region'],
                    'type': site['type'],
                    'heritage_status': heritage_status,
                    'unesco_confirmed': unesco_confirmed,
                    'heritage_keywords_found': found_heritage_keywords,
                    'stone_carving_keywords_found': found_stone_keywords,
                    'unesco_sentences': unesco_sentences[:3],
                    'heritage_sentences': heritage_sentences[:2],
                    'stone_carving_sentences': stone_carving_sentences[:2],
                    'page_title': page_title,
                    'source_url': site['url'],
                    'content_length': len(clean_text),
                    'relevance_score': len(found_heritage_keywords) + len(found_stone_keywords),
                    'research_timestamp': datetime.now().isoformat()
                }
                
                heritage_findings.append(site_finding)
                
            elif found_heritage_keywords or found_stone_keywords:
                print("\nℹ️ Partially relevant content found")
                print(f"   Heritage keywords: {found_heritage_keywords}")
                print(f"   Stone carving keywords: {found_stone_keywords}")
            else:
                print("\nℹ️ Limited relevance to stone carving heritage")
                
        else:
            print(f"❌ Failed to access (Status: {response.status_code})")
            
    except Exception as e:
        print(f"❌ Error researching {site['name']}: {str(e)}")
    
    print("\n" + "=" * 90)
    time.sleep(2)  # Respectful delay between requests
