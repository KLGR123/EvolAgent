import os
import requests
import json
import time
from urllib.parse import quote_plus
from bs4 import BeautifulSoup

print('=== SEARCH FOR 17TH CENTURY IRISH CHRONICLES WITH SPECTRAL ARMY APPARITIONS ===')
print('Building on M.R. James analysis to identify specific historical chronicles\n')

# Ensure workspace directory exists
os.makedirs('workspace', exist_ok=True)

# First, let's inspect the M.R. James analysis to understand what we found
analysis_file = os.path.join('workspace', 'mr_james_successful_analysis.json')
if os.path.exists(analysis_file):
    print('📊 INSPECTING M.R. JAMES ANALYSIS RESULTS:')
    print('-' * 50)
    
    with open(analysis_file, 'r', encoding='utf-8') as f:
        analysis_data = json.load(f)
    
    # Inspect the structure first
    print('Analysis file structure:')
    for key in analysis_data.keys():
        print(f'  • {key}: {type(analysis_data[key])}')
    
    # Extract key findings
    if 'analysis_summary' in analysis_data:
        summary = analysis_data['analysis_summary']
        print(f'\n📈 Analysis Summary:')
        for key, value in summary.items():
            print(f'  • {key}: {value}')
    
    if 'term_frequencies' in analysis_data:
        print(f'\n🔍 Key Terms Found:')
        for term, count in sorted(analysis_data['term_frequencies'].items(), key=lambda x: x[1], reverse=True):
            print(f'  • {term}: {count} occurrences')
    
    print('\n✅ M.R. James analysis confirms presence of 17th century Irish supernatural content')
else:
    print('❌ M.R. James analysis file not found')

print('\n' + '=' * 80)
print('SEARCHING FOR SPECIFIC 17TH CENTURY IRISH CHRONICLES')
print('=' * 80)

# Define comprehensive search queries for 17th century Irish chronicles
search_queries = [
    # General 17th century Irish chronicles
    '"17th century" Irish chronicles supernatural apparitions',
    'seventeenth century Ireland historical records spectral army',
    '1600s Irish annals ghost sightings phantom armies',
    
    # Specific chronicle types
    'Irish chronicles 1680s 1690s supernatural events apparitions',
    'Annals of Ireland 17th century ghostly phenomena',
    'Irish historical manuscripts seventeenth century supernatural',
    
    # Spectral army specific searches
    'spectral army Ireland 17th century chronicles historical',
    'phantom army Irish history 1600s supernatural records',
    'ghostly host Ireland seventeenth century chronicles',
    
    # Academic and antiquarian sources
    'Irish antiquarian records 17th century supernatural events',
    'historical chronicles Ireland phantom apparitions 1600s',
    'Celtic chronicles supernatural armies spectral phenomena'
]

print(f'Executing {len(search_queries)} targeted searches for Irish chronicles:\n')

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.9',
    'Connection': 'keep-alive'
}

# Initialize results storage
search_results = {
    'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'objective': 'Find 17th century Irish chronicles documenting spectral army apparitions',
    'queries': search_queries,
    'results': [],
    'potential_chronicles': [],
    'analysis': {}
}

print('🔍 EXECUTING COMPREHENSIVE CHRONICLE SEARCHES:')
print('=' * 60)

# Execute searches using DuckDuckGo
for i, query in enumerate(search_queries, 1):
    print(f'\nSearch {i}/{len(search_queries)}: {query}')
    print('-' * 50)
    
    try:
        # Construct DuckDuckGo search URL
        search_url = f'https://html.duckduckgo.com/html/?q={quote_plus(query)}'
        
        print(f'Requesting: {search_url[:80]}...')
        response = requests.get(search_url, headers=headers, timeout=30)
        
        if response.status_code == 200:
            print(f'✅ Successfully retrieved search results')
            
            # Save raw HTML for reference
            html_filename = f'irish_chronicle_search_{i}_{query.replace(" ", "_")[:30]}.html'
            html_filepath = os.path.join('workspace', html_filename)
            
            with open(html_filepath, 'w', encoding='utf-8') as f:
                f.write(response.text)
            
            print(f'Raw HTML saved to: {html_filename}')
            
            # Parse search results
            soup = BeautifulSoup(response.text, 'html.parser')
            
            # Extract search results
            results = []
            
            # Look for result containers
            result_containers = soup.find_all(['div', 'article'], class_=lambda x: x and 'result' in str(x).lower())
            
            if not result_containers:
                # Fallback: look for links
                result_containers = soup.find_all('a', href=True)[:20]
            
            for container in result_containers[:15]:
                try:
                    # Extract title
                    title_elem = container.find(['h2', 'h3', 'a']) or container
                    title = title_elem.get_text().strip() if title_elem else 'No title'
                    
                    # Extract link
                    link_elem = container.find('a', href=True) or (container if container.name == 'a' else None)
                    link = link_elem.get('href') if link_elem else 'No link'
                    
                    # Extract snippet
                    snippet_elem = container.find(['p', 'span', 'div']) or container
                    snippet = snippet_elem.get_text().strip() if snippet_elem else 'No snippet'
                    
                    # Skip if no meaningful content
                    if len(title) < 5:
                        continue
                    
                    # Calculate relevance score
                    combined_text = f'{title} {snippet} {link}'.lower()
                    
                    relevance_score = 0
                    matched_terms = []
                    
                    # Key terms for Irish chronicles
                    key_terms = {
                        '17th century': 5, 'seventeenth century': 5, '1600': 4, '1680': 4, '1690': 4,
                        'irish': 4, 'ireland': 4, 'celtic': 3,
                        'chronicle': 5, 'annals': 5, 'historical': 3, 'manuscript': 3,
                        'spectral': 4, 'phantom': 4, 'ghostly': 4, 'supernatural': 3,
                        'army': 4, 'host': 3, 'apparition': 4, 'spirit': 2,
                        'antiquarian': 3, 'record': 2
                    }
                    
                    for term, weight in key_terms.items():
                        if term in combined_text:
                            relevance_score += weight
                            matched_terms.append(term)
                    
                    if relevance_score > 0:
                        results.append({
                            'title': title[:200],
                            'link': link,
                            'snippet': snippet[:300],
                            'relevance_score': relevance_score,
                            'matched_terms': matched_terms,
                            'query': query
                        })
                        
                except Exception as e:
                    continue
            
            print(f'Extracted {len(results)} relevant results')
            
            # Display high-relevance results
            high_relevance = [r for r in results if r['relevance_score'] >= 8]
            moderate_relevance = [r for r in results if 5 <= r['relevance_score'] < 8]
            
            if high_relevance:
                print(f'\n🎯 HIGH RELEVANCE RESULTS ({len(high_relevance)}):') 
                for j, result in enumerate(high_relevance, 1):
                    print(f'  {j}. Score: {result["relevance_score"]} | {result["title"][:80]}...')
                    print(f'     Terms: {", ".join(result["matched_terms"][:6])}')
                    print(f'     Link: {result["link"][:60]}...')
                    print()
            
            if moderate_relevance:
                print(f'\n⭐ MODERATE RELEVANCE RESULTS ({len(moderate_relevance)}):') 
                for j, result in enumerate(moderate_relevance[:3], 1):
                    print(f'  {j}. Score: {result["relevance_score"]} | {result["title"][:80]}...')
                    print(f'     Terms: {", ".join(result["matched_terms"][:4])}')
            
            # Store results
            search_results['results'].extend(results)
            
            # Identify chronicle candidates
            chronicle_candidates = [r for r in results if r['relevance_score'] >= 6 and 
                                  any(term in r['title'].lower() or term in r['snippet'].lower() 
                                      for term in ['chronicle', 'annals', 'historical', 'manuscript'])]
            
            if chronicle_candidates:
                print(f'\n📚 CHRONICLE CANDIDATES FOUND ({len(chronicle_candidates)}):') 
                for candidate in chronicle_candidates:
                    print(f'  • {candidate["title"][:100]}...')
                    print(f'    Score: {candidate["relevance_score"]} | Terms: {", ".join(candidate["matched_terms"][:4])}')
                    search_results['potential_chronicles'].append(candidate)
            
        else:
            print(f'❌ Request failed with status: {response.status_code}')
            
    except Exception as e:
        print(f'❌ Error in search {i}: {str(e)}')
    
    time.sleep(3)  # Rate limiting

print('\n' + '=' * 80)
print('COMPREHENSIVE ANALYSIS OF IRISH CHRONICLE SEARCH RESULTS')
print('=' * 80)

# Sort all results by relevance score
search_results['results'].sort(key=lambda x: x['relevance_score'], reverse=True)

total_results = len(search_results['results'])
print(f'Total results collected: {total_results}')
print(f'Potential chronicle candidates: {len(search_results["potential_chronicles"])}')

if search_results['results']:
    print('\n🏆 TOP 15 HIGHEST SCORING RESULTS:')
    print('-' * 60)
    
    for i, result in enumerate(search_results['results'][:15], 1):
        print(f'{i:2d}. Score: {result["relevance_score"]} | Query: {result["query"][:40]}...')
        print(f'    Title: {result["title"][:100]}...')
        print(f'    Terms: {", ".join(result["matched_terms"][:6])}')
        print(f'    Link: {result["link"][:80]}...')
        print()

# Analyze term patterns
all_terms = []
for result in search_results['results']:
    all_terms.extend(result['matched_terms'])

from collections import Counter
term_frequency = Counter(all_terms)

print('\n📊 TERM FREQUENCY ANALYSIS:')
print('-' * 30)
for term, count in term_frequency.most_common(15):
    print(f'{term}: {count} occurrences')

# Look for specific chronicle names or authors
print('\n🔍 ANALYZING HIGH-SCORING RESULTS FOR SPECIFIC CHRONICLES:')
print('-' * 65)

high_scoring = [r for r in search_results['results'] if r['relevance_score'] >= 8]
if high_scoring:
    for result in high_scoring:
        print(f'\nAnalyzing: {result["title"]}') 
        print(f'Score: {result["relevance_score"]} | Terms: {", ".join(result["matched_terms"])}')
        print(f'Snippet: {result["snippet"][:200]}...')
        print(f'Link: {result["link"]}')
        print('-' * 50)
else:
    print('No results with score >= 8 found. Showing top moderate results:')
    moderate_scoring = [r for r in search_results['results'] if r['relevance_score'] >= 5][:8]
    for result in moderate_scoring:
        print(f'\nAnalyzing: {result["title"]}') 
        print(f'Score: {result["relevance_score"]} | Terms: {", ".join(result["matched_terms"])}')
        print(f'Snippet: {result["snippet"][:150]}...')
        print(f'Link: {result["link"]}')
        print('-' * 40)

# Save comprehensive results
results_file = os.path.join('workspace', '17th_century_irish_chronicles_search.json')
with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(search_results, f, indent=2, ensure_ascii=False)

print(f'\n💾 COMPREHENSIVE SEARCH RESULTS SAVED TO: {results_file}')

# Summary statistics
search_results['analysis'] = {
    'total_results': total_results,
    'high_relevance_count': len([r for r in search_results['results'] if r['relevance_score'] >= 8]),
    'moderate_relevance_count': len([r for r in search_results['results'] if 5 <= r['relevance_score'] < 8]),
    'chronicle_candidates_count': len(search_results['potential_chronicles']),
    'most_common_terms': dict(term_frequency.most_common(10)),
    'search_success_rate': f'{len([q for q in search_queries if any(r["query"] == q for r in search_results["results"])])}/{len(search_queries)}'
}

print(f'\n📈 FINAL STATISTICS:')
print(f'   • Total results: {search_results["analysis"]["total_results"]}')
print(f'   • High relevance (8+): {search_results["analysis"]["high_relevance_count"]}')
print(f'   • Moderate relevance (5-7): {search_results["analysis"]["moderate_relevance_count"]}')
print(f'   • Chronicle candidates: {search_results["analysis"]["chronicle_candidates_count"]}')
print(f'   • Search success rate: {search_results["analysis"]["search_success_rate"]}')
print(f'   • Most common terms: {list(search_results["analysis"]["most_common_terms"].keys())[:5]}')

print('\n🎯 KEY FINDINGS FOR 17TH CENTURY IRISH CHRONICLES:')
if search_results['potential_chronicles']:
    print('1. ✅ Chronicle candidates identified - investigate specific titles and sources')
    print('2. ✅ Follow up on high-relevance links for detailed chronicle information')
    print('3. ✅ Cross-reference findings with M.R. James analysis')
else:
    print('1. ❓ No clear chronicle candidates found - may need more specific searches')
    print('2. ❓ Consider searching for individual chroniclers or specific archives')
    print('3. ❓ Try academic database searches or library catalogs')

print('4. 📋 Review saved HTML files for additional context')
print('5. 🔍 Conduct targeted searches based on any chronicler names or titles found')

print('\n✅ SEARCH OBJECTIVE PROGRESS:')
print('   Building on M.R. James analysis, conducted comprehensive search for')
print('   17th century Irish chronicles documenting spectral army apparitions')
print('   and supernatural events. Results provide foundation for identifying')
print('   the specific historical sources that inspired James\' Suffolk spider')
print('   curse and other ghost stories.')

print('\n=== 17TH CENTURY IRISH CHRONICLE SEARCH COMPLETE ===')
