# Now calculate total flight time for each astronaut
print(f"\n=== CALCULATING TOTAL FLIGHT TIME FOR EACH ASTRONAUT ===")
print("Processing each Group 2 astronaut who flew in space...\n")

for astronaut in astronauts_who_flew:
    astronaut_name = astronaut['name']
    missions = astronaut['notable_missions']
    
    print(f"=== {astronaut_name.upper()} ===")
    print(f"Full Name: {astronaut['full_name']}")
    print(f"Background: {astronaut['background']}")
    print(f"Missions to analyze: {', '.join(missions)}")
    
    total_minutes = 0
    mission_details = []
    
    for mission in missions:
        if mission in mission_durations:
            duration = mission_durations[mission]['duration_minutes']
            date = mission_durations[mission]['date']
            crew = mission_durations[mission]['crew']
            
            total_minutes += duration
            
            # Convert minutes to days, hours, minutes for display
            days = duration // (24 * 60)
            hours = (duration % (24 * 60)) // 60
            mins = duration % 60
            
            mission_details.append({
                'mission': mission,
                'duration_minutes': duration,
                'duration_display': f"{days}d {hours}h {mins}m",
                'date': date,
                'crew_size': len(crew)
            })
            
            print(f"  ‚úì {mission}: {duration:,} minutes ({days}d {hours}h {mins}m) - {date}")
        else:
            print(f"  ‚ö†Ô∏è {mission}: Duration data not found in database")
    
    # Calculate total time display
    total_days = total_minutes // (24 * 60)
    total_hours = (total_minutes % (24 * 60)) // 60
    total_mins = total_minutes % 60
    
    print(f"\n  üìä TOTAL FLIGHT TIME:")
    print(f"      Minutes: {total_minutes:,}")
    print(f"      Display: {total_days}d {total_hours}h {total_mins}m")
    print(f"      Missions: {len(mission_details)}")
    
    # Add to flight records
    astronaut_record = {
        'name': astronaut_name,
        'full_name': astronaut['full_name'],
        'background': astronaut['background'],
        'total_flight_time_minutes': total_minutes,
        'total_flight_time_display': f"{total_days}d {total_hours}h {total_mins}m",
        'number_of_missions': len(mission_details),
        'missions': mission_details
    }
