                'gregory chappell': text_lower.count('gregory chappell'),
                'g chappell': text_lower.count('g chappell'),
                'chappell': text_lower.count('chappell')

            if file_mentions > 0:
                print('âœ… Greg Chappell content found')

                all_content.append({
                    'filename': filename,
                    'text': page_text,
                    'text_lower': text_lower,
                    'mentions': file_mentions

                death_patterns = [

                    r'greg chappell.*?obituary',
                    r'greg chappell.*?passed away',
                    r'death.*?greg chappell',
                    r'died.*?greg chappell',
                    r'obituary.*?greg chappell'
                ]
                
                for pattern in death_patterns:
                    matches = re.findall(pattern, text_lower, re.DOTALL)
                    if matches:
                        for match in matches:
                            death_evidence.append({
                                'file': filename,
                                'pattern': pattern,
                                'match': match[:200],
                                'type': 'death_pattern'
                            })
                
                # Direct pattern matching for alive-related information
                alive_patterns = [
                    r'greg chappell.*?still alive',
                    r'greg chappell.*?living',
                    r'greg chappell.*?current',
                    r'greg chappell.*?recent',
                    r'greg chappell.*?today',
                    r'greg chappell.*?now'
                ]
                
                for pattern in alive_patterns:
                    matches = re.findall(pattern, text_lower, re.DOTALL)
                    if matches:
                        for match in matches:
                            alive_evidence.append({
                                'file': filename,
                                'pattern': pattern,
                                'match': match[:200],
                                'type': 'alive_pattern'

                print(f'Death patterns found: {len([e for e in death_evidence if e["file"] == filename])}')

        'analysis_method': 'direct_text_pattern_matching',

        'death_evidence': death_evidence,
