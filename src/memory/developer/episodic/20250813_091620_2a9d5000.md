headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',

}

def safe_request(url, timeout=20):
    """Make a safe HTTP request with error handling"""
    try:
        response = requests.get(url, headers=headers, timeout=timeout, verify=False)
        response.raise_for_status()
        return response
    except Exception as e:
        print(f"  ‚úó Request failed: {str(e)[:100]}...")
        return None

def extract_text_content(response):
    """Extract clean text from HTML response"""
    if not response:
        return ""
    
    soup = BeautifulSoup(response.content, 'html.parser')
    
    # Remove script and style elements
    for script in soup(["script", "style"]):
        script.decompose()
    
    # Get text content
    text = soup.get_text()
    lines = (line.strip() for line in text.splitlines())
    chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
    clean_text = ' '.join(chunk for chunk in chunks if chunk)
    
    return clean_text

        # Search for writer names in context with Dustin Nguyen

            r'(?:written by|story by|script by|writer[:\s]+)\s*([A-Z][a-z]+\s+[A-Z][a-z]+)',
