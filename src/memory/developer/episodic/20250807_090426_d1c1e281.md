import os
import json
from bs4 import BeautifulSoup

print('=== EXAMINING SAVED SEARCH RESULTS FOR CONTENT ANALYSIS ===')
print('Previous search attempt saved HTML files but extracted no content')
print('Investigating the actual structure and content of saved files\n')

# First, let's check what files were actually saved
workspace_dir = 'workspace'
if not os.path.exists(workspace_dir):
    print(f'❌ Workspace directory {workspace_dir} does not exist')
    print('Creating workspace directory...')
    os.makedirs(workspace_dir, exist_ok=True)

# List all HTML files in workspace
html_files = [f for f in os.listdir(workspace_dir) if f.endswith('.html')]
print(f'Found {len(html_files)} HTML files in workspace:')
for i, filename in enumerate(html_files, 1):
    print(f'  {i}. {filename}')

if not html_files:
    print('\n❌ No HTML files found in workspace. Previous search may have failed.')
    print('Will need to conduct new searches with alternative methods.')
else:
    # Examine the first HTML file to understand the structure
    first_file = html_files[0]
    filepath = os.path.join(workspace_dir, first_file)
    
    print(f'\n=== ANALYZING FIRST HTML FILE: {first_file} ===')
    print('=' * 70)
    
    try:
        with open(filepath, 'r', encoding='utf-8') as f:
            html_content = f.read()
        
        print(f'File size: {len(html_content):,} characters')
        
        # Check if file contains actual content or is blocked/empty
        if len(html_content) < 1000:
            print('⚠️ File is very small - likely blocked or empty')
            print('First 500 characters:')
            print(html_content[:500])
        else:
            # Parse with BeautifulSoup
            soup = BeautifulSoup(html_content, 'html.parser')
            
            # Check for common Google search result indicators
            title = soup.find('title')
            print(f'Page title: {title.get_text() if title else "No title found"}')
            
            # Look for search result containers
            search_containers = [
                soup.find_all('div', class_=lambda x: x and 'g' in str(x)),
                soup.find_all('div', class_=lambda x: x and 'result' in str(x)),
                soup.find_all('h3'),
                soup.find_all('a', href=True)
            ]
            
            print('\n🔍 ANALYZING SEARCH RESULT STRUCTURE:')
            for i, containers in enumerate(search_containers):
                container_type = ['div.g-class', 'div.result-class', 'h3 tags', 'links'][i]
                print(f'  • {container_type}: {len(containers)} found')
            
            # Extract all text and check for our key terms
            page_text = soup.get_text().lower()
            print(f'\nTotal extracted text length: {len(page_text):,} characters')
            
            # Check for key terms that should appear in search results
            key_terms_check = {
                'glanvill': 'glanvill' in page_text,
                'saducismus': 'saducismus' in page_text,
                'triumphatus': 'triumphatus' in page_text,
                'spectral': 'spectral' in page_text,
                'supernatural': 'supernatural' in page_text,
                'ireland': 'ireland' in page_text,
                'suffolk': 'suffolk' in page_text,
                '1681': '1681' in page_text,
                'google': 'google' in page_text,
                'search': 'search' in page_text
            }
            
            print('\n📊 KEY TERMS DETECTION:')
            for term, found in key_terms_check.items():
                status = '✅' if found else '❌'
                print(f'  {status} {term}: {found}')
            
            # If we found Google/search terms but not our target terms, likely blocked
            if key_terms_check['google'] or key_terms_check['search']:
                if not any([key_terms_check['glanvill'], key_terms_check['saducismus'], key_terms_check['spectral']]):
                    print('\n⚠️ DIAGNOSIS: Google search page detected but no target content found')
                    print('This suggests anti-scraping measures or JavaScript-rendered results')
                else:
                    print('\n✅ DIAGNOSIS: Valid search results with target content detected')
            else:
                print('\n❌ DIAGNOSIS: No recognizable Google search structure found')
            
            # Show a sample of the extracted text
            if len(page_text) > 200:
                print('\n📄 SAMPLE EXTRACTED TEXT (first 500 characters):')
                print('-' * 50)
                print(page_text[:500])
                print('-' * 50)
            
            # Look for specific patterns that might indicate blocked content
            blocking_indicators = [
                'captcha',
                'unusual traffic',
                'automated queries',
                'terms of service',
                'access denied',
                'blocked',
                'robot'
            ]
            
            found_blocking = []
            for indicator in blocking_indicators:
                if indicator in page_text:
                    found_blocking.append(indicator)
            
            if found_blocking:
                print(f'\n🚫 BLOCKING INDICATORS DETECTED: {found_blocking}')
                print('Google is likely blocking our automated requests')
            
    except Exception as e:
        print(f'❌ Error reading HTML file: {str(e)}')

# Check if we have the JSON results file from previous search
json_files = [f for f in os.listdir(workspace_dir) if f.endswith('.json')]
print(f'\n=== CHECKING FOR PREVIOUS SEARCH RESULTS JSON ===')
print(f'Found {len(json_files)} JSON files:')
for filename in json_files:
    print(f'  • {filename}')

if json_files:
    # Examine the JSON structure first
    json_file = json_files[0]
    json_path = os.path.join(workspace_dir, json_file)
    
    print(f'\n🔍 EXAMINING JSON STRUCTURE: {json_file}')
    try:
        with open(json_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        print('JSON file structure:')
        if isinstance(data, dict):
            for key in data.keys():
                value = data[key]
                if isinstance(value, list):
                    print(f'  • {key}: list with {len(value)} items')
                elif isinstance(value, dict):
                    print(f'  • {key}: dict with {len(value)} keys')
                else:
                    print(f'  • {key}: {type(value).__name__} - {str(value)[:100]}...')
        
        # Check the findings specifically
        if 'findings' in data and isinstance(data['findings'], list):
            print(f'\n📋 FINDINGS ANALYSIS:')
            print(f'  • Total findings: {len(data["findings"])}')
            
            if data['findings']:
                sample_finding = data['findings'][0]
                print('  • Sample finding structure:')
                for key, value in sample_finding.items():
                    if isinstance(value, list):
                        print(f'    - {key}: list with {len(value)} items')
                    else:
                        print(f'    - {key}: {type(value).__name__}')
                
                # Check relevance scores
                scores = [f.get('relevance_score', 0) for f in data['findings']]
                max_score = max(scores) if scores else 0
                print(f'  • Maximum relevance score: {max_score}')
                print(f'  • Non-zero scores: {len([s for s in scores if s > 0])}')
            
    except Exception as e:
        print(f'❌ Error reading JSON file: {str(e)}')

print('\n' + '=' * 80)
print('DIAGNOSIS AND NEXT STEPS')

    print('❌ NO FILES FOUND - Need to start fresh with alternative search methods')
