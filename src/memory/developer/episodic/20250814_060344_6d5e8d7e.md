Identification of OpenCV Version and Contributors for Initial Mask-RCNN Model Support Implementation

**Description**: Research OpenCV version history and release notes to identify when Mask-RCNN model support was first added to the library. Focus on finding the specific OpenCV version that introduced Mask-RCNN functionality, then examine the contributors, commit history, and development team for that particular release. Extract the names of all contributors who worked on implementing Mask-RCNN support in OpenCV.

**Use Cases**:
- Academic research on the evolution of deep learning features in OpenCV, specifically to cite the exact version and contributors when publishing papers involving Mask-RCNN-based experiments
- Software compliance auditing for enterprise computer vision applications, ensuring that only OpenCV versions with officially supported Mask-RCNN functionality are deployed in regulated environments
- Automated generation of contributor acknowledgments in open-source documentation or grant proposals, by extracting the names of developers who implemented Mask-RCNN support in OpenCV
- Technical due diligence for startups evaluating third-party computer vision libraries, verifying the maturity and maintenance history of Mask-RCNN support in OpenCV before integration
- Building internal knowledge bases or wikis for R&D teams, providing detailed timelines and contributor lists for major feature additions like Mask-RCNN in OpenCV
- Supporting bug triage and regression analysis by correlating Mask-RCNN-related issues and commits with specific OpenCV releases, helping QA teams pinpoint when changes were introduced

```
import requests
import json
from datetime import datetime
import os
import re
from bs4 import BeautifulSoup
import time

print("=== OPENCV MASK-RCNN VERSION HISTORY RESEARCH ===\n")
print("Objective: Find when Mask-RCNN support was first added to OpenCV")
print("Focus: Identify specific version, contributors, and development team\n")

# Create workspace directory
workspace_dir = 'workspace'
os.makedirs(workspace_dir, exist_ok=True)
print(f"Using workspace directory: {workspace_dir}\n")

# GitHub API configuration for OpenCV repository
base_url = "https://api.github.com"
repo = "opencv/opencv"

headers = {
    'Accept': 'application/vnd.github.v3+json',
    'User-Agent': 'Python-OpenCV-Research'
}

print(f"Repository: {repo}\n")

# Step 1: Search for Mask-RCNN related commits and issues
print("=== STEP 1: SEARCHING FOR MASK-RCNN RELATED CONTENT ===\n")

search_strategies = [
    {
        'name': 'Mask-RCNN commits',
        'type': 'commits',
        'query': f'repo:{repo} Mask-RCNN',
        'description': 'Search for commits mentioning Mask-RCNN'
    },
    {
        'name': 'Mask R-CNN commits (space variant)',
        'type': 'commits', 
        'query': f'repo:{repo} "Mask R-CNN"',
        'description': 'Search for commits with space in name'
    },
    {
        'name': 'MaskRCNN commits (no hyphen)',
        'type': 'commits',
        'query': f'repo:{repo} MaskRCNN',
        'description': 'Search for commits with no hyphen/space'
    },
    {
        'name': 'Mask-RCNN issues',
        'type': 'issues',
        'query': f'repo:{repo} Mask-RCNN',
        'description': 'Search for issues mentioning Mask-RCNN'
    }
]

all_results = {}

for strategy in search_strategies:
    print(f"--- {strategy['name'].upper()} ---")
    print(f"Query: {strategy['query']}")
    
    if strategy['type'] == 'commits':
        search_url = f"{base_url}/search/commits"
        headers_with_preview = headers.copy()
        headers_with_preview['Accept'] = 'application/vnd.github.cloak-preview+json'
    else:
        search_url = f"{base_url}/search/issues"
        headers_with_preview = headers
    
    params = {
        'q': strategy['query'],
        'sort': 'committer-date' if strategy['type'] == 'commits' else 'created',
        'order': 'asc',  # Oldest first to find when it was introduced
        'per_page': 30
    }
    
    try:
        print("Making API request...")
        response = requests.get(search_url, headers=headers_with_preview, params=params)
        
        if response.status_code == 200:
            results = response.json()
            total_count = results['total_count']
            items = results['items']
            
            print(f"Status: SUCCESS")
            print(f"Total found: {total_count}")
            print(f"Retrieved: {len(items)}")
            
            all_results[strategy['name']] = {
                'query': strategy['query'],
                'type': strategy['type'],
                'total_count': total_count,
                'items': items,
                'status': 'success'
            }
            
            # Show sample results
            if items:
                print("\nSample results:")
                for i, item in enumerate(items[:3], 1):
                    if strategy['type'] == 'commits':
                        commit_msg = item['commit']['message'][:100]
                        commit_date = item['commit']['committer']['date']
                        author = item['commit']['author']['name']
                        print(f"  {i}. {commit_msg}...")
                        print(f"     Date: {commit_date}")
                        print(f"     Author: {author}")
                        print(f"     SHA: {item['sha']}")
                    else:
                        print(f"  {i}. Issue #{item['number']}: {item['title'][:80]}...")
                        print(f"     Created: {item['created_at']}")
                        print(f"     State: {item['state']}")
                    print()
            else:
                print("No results found.")
                
        else:
            print(f"Status: ERROR ({response.status_code})")
            print(f"Error: {response.text[:200]}...")
            all_results[strategy['name']] = {
                'status': 'error',
                'status_code': response.status_code
            }
            
    except Exception as e:
        print(f"Status: EXCEPTION - {str(e)}")
        all_results[strategy['name']] = {
            'status': 'exception',
            'error': str(e)
        }
    
    print()
    time.sleep(2)  # Be respectful to API

print("=== STEP 2: ANALYZING SEARCH RESULTS ===\n")

# Collect all commits and issues for analysis
all_commits = []
all_issues = []

for strategy_name, results in all_results.items():
    if results.get('status') == 'success' and results.get('items'):
        if results['type'] == 'commits':
            all_commits.extend(results['items'])
        else:
            all_issues.extend(results['items'])

print(f"Total commits found: {len(all_commits)}")
print(f"Total issues found: {len(all_issues)}")

# Sort commits by date to find the earliest
if all_commits:
    # Remove duplicates based on SHA
    unique_commits = {}
    for commit in all_commits:
        sha = commit['sha']
        if sha not in unique_commits:
            unique_commits[sha] = commit
    
    sorted_commits = sorted(unique_commits.values(), 
                          key=lambda x: x['commit']['committer']['date'])
    
    print(f"\nUnique commits: {len(sorted_commits)}")
    print("\n=== EARLIEST MASK-RCNN COMMITS ===\n")
    
    for i, commit in enumerate(sorted_commits[:10], 1):
        commit_msg = commit['commit']['message']
        commit_date = commit['commit']['committer']['date']
        author = commit['commit']['author']['name']
        committer = commit['commit']['committer']['name']
        
        print(f"{i}. {commit_msg[:100]}...")
        print(f"   Date: {commit_date}")
        print(f"   Author: {author}")
        print(f"   Committer: {committer}")
        print(f"   SHA: {commit['sha']}")
        print(f"   URL: {commit['html_url']}")
        print()

# Analyze issues as well
if all_issues:
    unique_issues = {}
    for issue in all_issues:
        number = issue['number']
        if number not in unique_issues:
            unique_issues[number] = issue
    
    sorted_issues = sorted(unique_issues.values(),
                         key=lambda x: x['created_at'])
    
    print(f"\n=== EARLIEST MASK-RCNN ISSUES ===\n")
    
    for i, issue in enumerate(sorted_issues[:5], 1):
        print(f"{i}. Issue #{issue['number']}: {issue['title']}")
        print(f"   Created: {issue['created_at']}")
        print(f"   State: {issue['state']}")
        print(f"   Labels: {[label['name'] for label in issue['labels']]}")
        print(f"   URL: {issue['html_url']}")
        print()

print("=== STEP 3: GETTING OPENCV RELEASE INFORMATION ===\n")

# Get OpenCV releases to correlate with commit dates
print("Fetching OpenCV releases...")
releases_url = f"{base_url}/repos/{repo}/releases"
params = {'per_page': 100}  # Get many releases

try:
    releases_response = requests.get(releases_url, headers=headers, params=params)
    
    if releases_response.status_code == 200:
        releases = releases_response.json()
        print(f"Found {len(releases)} releases")
        
        # Sort releases by published date
        sorted_releases = sorted(releases, key=lambda x: x['published_at'])
        
        print("\nRecent releases (for context):")
        for i, release in enumerate(sorted_releases[-10:], 1):
            print(f"  {i}. {release['tag_name']} - {release['name']}")
            print(f"     Published: {release['published_at']}")
            print(f"     Prerelease: {release['prerelease']}")
            print()
            
    else:
        print(f"Error fetching releases: {releases_response.status_code}")
        releases = []
        
except Exception as e:
    print(f"Exception fetching releases: {str(e)}")
    releases = []

# Save comprehensive results
research_data = {
    'research_timestamp': datetime.now().isoformat(),
    'repository': repo,
    'search_strategies': search_strategies,
    'search_results': all_results,
    'analysis': {
        'total_unique_commits': len(unique_commits) if all_commits else 0,
        'total_unique_issues': len(unique_issues) if all_issues else 0,
        'earliest_commits': sorted_commits[:10] if all_commits else [],
        'earliest_issues': sorted_issues[:5] if all_issues else []
    },
    'releases_found': len(releases) if 'releases' in locals() else 0,
    'next_steps': [
        'Analyze specific commits for Mask-RCNN implementation details',
        'Check commit history around identified dates',
        'Look for pull requests related to Mask-RCNN',
        'Examine release notes for version that introduced Mask-RCNN'
    ]
}

output_file = f'{workspace_dir}/opencv_maskrcnn_research.json'
with open(output_file, 'w') as f:
    json.dump(research_data, f, indent=2)

print(f"\nComprehensive research data saved to: {output_file}")

print("\n=== INITIAL RESEARCH SUMMARY ===\n")
print(f"Repository analyzed: {repo}")
print(f"Search strategies tested: {len(search_strategies)}")
print(f"Unique commits found: {len(unique_commits) if all_commits else 0}")
print(f"Unique issues found: {len(unique_issues) if all_issues else 0}")
print(f"OpenCV releases retrieved: {len(releases) if 'releases' in locals() else 0}")

if all_commits:
    earliest_commit = sorted_commits[0]
    print(f"\nEarliest Mask-RCNN commit:")
    print(f"  Date: {earliest_commit['commit']['committer']['date']}")
    print(f"  Author: {earliest_commit['commit']['author']['name']}")
    print(f"  Message: {earliest_commit['commit']['message'][:100]}...")
    print(f"  SHA: {earliest_commit['sha']}")

print("\nNext: Analyze specific commits and correlate with OpenCV versions")
```
