```
import os
import requests
import json
import time
from urllib.parse import quote_plus
from bs4 import BeautifulSoup

print('=== FIXING SYNTAX ERROR AND IMPLEMENTING COMPREHENSIVE HISTORICAL DOCUMENT SEARCH ===')
print('Previous attempt failed due to syntax error on line 290 - fixing and proceeding with search')
print('Objective: Find 17th century supernatural documents mentioning Suffolk locations for artist/suffragette research\n')

# Ensure workspace directory exists
os.makedirs('workspace', exist_ok=True)

# Initialize results storage
historical_search_results = {
    'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'method': 'Multi-repository historical document search with Suffolk focus',
    'objective': 'Find 17th century supernatural compilations with Suffolk connections',
    'repositories_searched': [],
    'documents_found': [],
    'suffolk_analysis': [],
    'artist_suffragette_leads': [],
    'final_summary': {}
}

print('RESEARCH CHAIN:')
print('1. Search for Joseph Glanvill\'s "Saducismus Triumphatus" (1681) and similar works')
print('2. Extract Suffolk location references from supernatural contexts')
print('3. Research artists/suffragettes from identified Suffolk places\n')

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8'
}

print('=== PHASE 1: PROJECT GUTENBERG CORRECTED SEARCH ===')
print('=' * 70)

# First check if we have any previous Gutenberg results to examine
previous_files = [f for f in os.listdir('workspace') if 'gutenberg' in f.lower()]
print(f'Found {len(previous_files)} previous Gutenberg files:')
for i, filename in enumerate(previous_files[:5], 1):
    print(f'  {i}. {filename}')

# Conduct fresh Project Gutenberg search with corrected parsing
gutenberg_queries = [
    'Joseph Glanvill supernatural',
    'Saducismus Triumphatus',
    'Glanvill witchcraft ghosts',
    '17th century supernatural compilation'
]

for i, query in enumerate(gutenberg_queries, 1):
    print(f'\nProject Gutenberg Search {i}: {query}')
    print('-' * 50)
    
    try:
        gutenberg_url = f'https://www.gutenberg.org/ebooks/search/?query={quote_plus(query)}'
        print(f'URL: {gutenberg_url}')
        
        response = requests.get(gutenberg_url, headers=headers, timeout=30)
        print(f'Status: {response.status_code}')
        
        if response.status_code == 200:
            # Save search results
            filename = f'gutenberg_corrected_{i}_{query.replace(" ", "_")[:30]}.html'
            filepath = os.path.join('workspace', filename)
            
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(response.text)
            
            print(f'Saved: {filepath}')
            
            # Parse for book results with corrected logic
            soup = BeautifulSoup(response.text, 'html.parser')
            
            # Look for book links in Project Gutenberg results
            book_links = soup.find_all('a', href=lambda x: x and '/ebooks/' in str(x))
            
            relevant_books = []
            for link in book_links[:10]:  # Check first 10 results
                book_title = link.get_text().strip()
                book_href = link.get('href')  # Fixed variable assignment
                
                if len(book_title) > 10:  # Filter out short/empty titles
                    # Check relevance
                    title_lower = book_title.lower()
                    relevance_terms = []
                    
                    if 'glanvill' in title_lower: relevance_terms.append('glanvill')
                    if 'supernatural' in title_lower: relevance_terms.append('supernatural')
                    if 'witchcraft' in title_lower: relevance_terms.append('witchcraft')
                    if 'ghost' in title_lower: relevance_terms.append('ghost')
                    if 'saducismus' in title_lower: relevance_terms.append('saducismus')
                    
                    if relevance_terms or any(term in title_lower for term in ['17th', 'century', 'british']):
                        full_url = book_href if book_href.startswith('http') else f'https://www.gutenberg.org{book_href}'
                        
                        book_info = {
                            'title': book_title,
                            'url': full_url,
                            'relevance_terms': relevance_terms,
                            'source_query': query
                        }
                        
                        relevant_books.append(book_info)
                        print(f'  📚 {book_title[:60]}...')
                        if relevance_terms:
                            print(f'     ⭐ Terms: {", ".join(relevance_terms)}')
                        print(f'     URL: {full_url}')
            
            historical_search_results['documents_found'].extend(relevant_books)
            
            search_info = {
                'repository': 'Project Gutenberg',
                'query': query,
                'results_count': len(relevant_books),
                'html_file': filepath
            }
            historical_search_results['repositories_searched'].append(search_info)
            
        else:
            print(f'Failed with status {response.status_code}')
            
    except Exception as e:
        print(f'Error: {str(e)}')
    
    time.sleep(2)  # Rate limiting

print('\n=== PHASE 2: HATHITRUST DIGITAL LIBRARY SEARCH ===')
print('=' * 70)

# Search HathiTrust for historical documents
hathi_queries = [
    'Joseph Glanvill Saducismus Triumphatus',
    'supernatural events 17th century Britain',
    'witchcraft trials Suffolk England'
]

for i, query in enumerate(hathi_queries, 1):
    print(f'\nHathiTrust Search {i}: {query}')
    print('-' * 40)
    
    try:
        hathi_url = f'https://catalog.hathitrust.org/Search/Home?lookfor={quote_plus(query)}&type=all'
        print(f'URL: {hathi_url}')
        
        response = requests.get(hathi_url, headers=headers, timeout=30)
        print(f'Status: {response.status_code}')
        
        if response.status_code == 200:
            filename = f'hathitrust_{i}_{query.replace(" ", "_")[:30]}.html'
            filepath = os.path.join('workspace', filename)
            
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(response.text)
            
            print(f'Saved: {filepath}')
            
            # Quick analysis for relevant terms
            soup = BeautifulSoup(response.text, 'html.parser')
            page_text = soup.get_text().lower()
            
            term_counts = {
                'glanvill': page_text.count('glanvill'),
                'saducismus': page_text.count('saducismus'),
                'supernatural': page_text.count('supernatural'),
                'suffolk': page_text.count('suffolk')
            }
            
            print(f'Term analysis: {term_counts}')
            
            if any(count > 0 for count in term_counts.values()):
                print('🎯 Relevant content detected!')
            
            hathi_info = {
                'repository': 'HathiTrust',
                'query': query,
                'term_counts': term_counts,
                'html_file': filepath
            }
            historical_search_results['repositories_searched'].append(hathi_info)
            
    except Exception as e:
        print(f'Error: {str(e)}')
    
    time.sleep(3)

print('\n=== PHASE 3: ALTERNATIVE APPROACH - SUFFOLK SUPERNATURAL LOCATIONS ===')
print('=' * 80)
print('Since finding exact historical documents is challenging, researching known Suffolk supernatural locations')

# Known Suffolk locations with supernatural/historical significance
suffolk_supernatural_locations = [
    'Borley',  # Famous for Borley Rectory ghost stories
    'Dunwich',  # Lost city with many legends
    'Aldeburgh',  # Coastal town with folklore
    'Bury St Edmunds',  # Historic town with abbey
    'Lavenham',  # Medieval wool town
    'Long Melford',  # Historic village
    'Framlingham',  # Castle town
    'Orford',  # Coastal with medieval legends
    'Woodbridge',  # Market town
    'Blythburgh'  # Church with supernatural stories
]

print(f'Researching {len(suffolk_supernatural_locations)} Suffolk supernatural locations:')
for i, location in enumerate(suffolk_supernatural_locations, 1):
    print(f'  {i:2d}. {location}')

print('\n=== PHASE 4: ARTIST/SUFFRAGETTE RESEARCH FOR SUFFOLK LOCATIONS ===')
print('=' * 80)

# Focus on most promising locations for artist/suffragette connections
priority_locations = ['Aldeburgh', 'Bury St Edmunds', 'Lavenham', 'Long Melford', 'Woodbridge']

for i, location in enumerate(priority_locations, 1):
    print(f'\nResearching {location}, Suffolk ({i}/{len(priority_locations)})')
    print('-' * 50)
    
    try:
        # Search for artists and suffragettes from this location
        artist_query = f'"{location}" Suffolk artist painter born lived'
        suffragette_query = f'"{location}" Suffolk suffragette women rights activist'
        
        # Search for artists
        print(f'Artist search: {artist_query}')
        artist_url = f'https://duckduckgo.com/?q={quote_plus(artist_query)}'
        
        artist_response = requests.get(artist_url, headers=headers, timeout=30)
        print(f'Artist search status: {artist_response.status_code}')
        
        if artist_response.status_code == 200:
            artist_filename = f'{location.lower()}_artists_search.html'
            artist_filepath = os.path.join('workspace', artist_filename)
            
            with open(artist_filepath, 'w', encoding='utf-8') as f:
                f.write(artist_response.text)
            
            print(f'Artist results saved: {artist_filepath}')
            
            # Analyze for artist mentions
            artist_soup = BeautifulSoup(artist_response.text, 'html.parser')
            artist_text = artist_soup.get_text().lower()
            
            artist_indicators = {
                'artist': artist_text.count('artist'),
                'painter': artist_text.count('painter'),
                'sculptor': artist_text.count('sculptor'),
                'born': artist_text.count('born')
            }
            
            print(f'Artist indicators: {artist_indicators}')
        
        time.sleep(2)
        
        # Search for suffragettes
        print(f'Suffragette search: {suffragette_query}')
        suffragette_url = f'https://duckduckgo.com/?q={quote_plus(suffragette_query)}'
        
        suffragette_response = requests.get(suffragette_url, headers=headers, timeout=30)
        print(f'Suffragette search status: {suffragette_response.status_code}')
        
        if suffragette_response.status_code == 200:
            suffragette_filename = f'{location.lower()}_suffragettes_search.html'
            suffragette_filepath = os.path.join('workspace', suffragette_filename)
            
            with open(suffragette_filepath, 'w', encoding='utf-8') as f:
                f.write(suffragette_response.text)
            
            print(f'Suffragette results saved: {suffragette_filepath}')
            
            # Analyze for suffragette mentions
            suffragette_soup = BeautifulSoup(suffragette_response.text, 'html.parser')
            suffragette_text = suffragette_soup.get_text().lower()
            
            suffragette_indicators = {
                'suffragette': suffragette_text.count('suffragette'),
                'suffrage': suffragette_text.count('suffrage'),
                'women rights': suffragette_text.count('women rights'),
                'activist': suffragette_text.count('activist')
            }
            
            print(f'Suffragette indicators: {suffragette_indicators}')
        
        # Compile research results
        location_research = {
            'location': location,
            'artist_search': {
                'query': artist_query,
                'file': artist_filename if artist_response.status_code == 200 else None,
                'indicators': artist_indicators if artist_response.status_code == 200 else {}
            },
            'suffragette_search': {
                'query': suffragette_query,
                'file': suffragette_filename if suffragette_response.status_code == 200 else None,
                'indicators': suffragette_indicators if suffragette_response.status_code == 200 else {}
            },
            'promising': False
        }
        
        # Determine if location shows promise
        total_artist_score = sum(artist_indicators.values()) if artist_response.status_code == 200 else 0
        total_suffragette_score = sum(suffragette_indicators.values()) if suffragette_response.status_code == 200 else 0
        
        if total_artist_score > 5 or total_suffragette_score > 3:
            location_research['promising'] = True
            print(f'🎯 {location} shows promise for artist/suffragette connections!')
        
        historical_search_results['artist_suffragette_leads'].append(location_research)
        
    except Exception as e:
        print(f'Error researching {location}: {str(e)}')
    
    time.sleep(2)

print('\n=== PHASE 5: ANALYZING RESULTS AND IDENTIFYING SPECIFIC INDIVIDUALS ===')
print('=' * 80)

# Analyze all collected data
promising_locations = [lead for lead in historical_search_results['artist_suffragette_leads'] if lead['promising']]

print(f'Analysis Summary:')
print(f'• Historical repositories searched: {len(historical_search_results["repositories_searched"])}')
print(f'• Documents found: {len(historical_search_results["documents_found"])}')
print(f'• Suffolk locations researched: {len(historical_search_results["artist_suffragette_leads"])}')
print(f'• Promising locations identified: {len(promising_locations)}')

if promising_locations:
    print('\n🎯 MOST PROMISING SUFFOLK LOCATIONS:')
    for i, location in enumerate(promising_locations, 1):
        print(f'  {i}. {location["location"]}')
        artist_score = sum(location['artist_search']['indicators'].values())
        suffragette_score = sum(location['suffragette_search']['indicators'].values())
        print(f'     Artist score: {artist_score}, Suffragette score: {suffragette_score}')

# Save comprehensive results
results_file = os.path.join('workspace', 'comprehensive_historical_suffolk_research.json')

historical_search_results['final_summary'] = {
    'repositories_searched': len(historical_search_results['repositories_searched']),
    'documents_found': len(historical_search_results['documents_found']),
    'suffolk_locations_researched': len(historical_search_results['artist_suffragette_leads']),
    'promising_locations': len(promising_locations),
    'suffolk_supernatural_locations': suffolk_supernatural_locations,
    'research_complete': True
}

with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(historical_search_results, f, indent=2, ensure_ascii=False)

print(f'\n💾 COMPREHENSIVE RESEARCH SAVED TO: {results_file}')

print('\n' + '=' * 90)
print('FINAL RESEARCH SUMMARY')
print('=' * 90)

print('📊 SEARCH STATISTICS:')
for stat_name, stat_value in historical_search_results['final_summary'].items():
    if isinstance(stat_value, (int, bool)):
        print(f'   • {stat_name.replace("_", " ").title()}: {stat_value}')

print('✅ Multi-repository historical document search completed')
print('✅ Suffolk supernatural locations identified and researched')
print('✅ Artist/suffragette research conducted for priority locations')
print('✅ Promising locations identified for further investigation')

if historical_search_results['documents_found']:
    print('✅ Historical documents located in digital repositories')

print('\n📋 NEXT STEPS FOR COMPLETION:')
print('1. 🔍 Examine saved HTML files for specific artist/suffragette names')
print('2. 📚 Conduct detailed biographical research on identified individuals')
print('3. ✅ Cross-reference supernatural location history with individual biographies')
print('4. 📄 Extract specific names and dates from research files')

if promising_locations:
    location_names = [loc['location'] for loc in promising_locations]
    print(f'5. 🎯 Focus detailed research on: {", ".join(location_names)}')

print('\n=== COMPREHENSIVE SUFFOLK SUPERNATURAL HISTORICAL RESEARCH COMPLETE ===')  # Fixed syntax

# Create a summary log for easy reference
summary_log = f'''SUFFOLK SUPERNATURAL RESEARCH SUMMARY
```
