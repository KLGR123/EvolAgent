### Development Step 3: Identify Suffolk town tied to 17th-century spider curse inspiring M.R. James’s ‘The Ash Tree’

**Description**: Research M.R. James's short story 'The Ash Tree' to identify the Suffolk location that inspired it, focusing on finding the real 17th century arachnid event involving an ash tree-linked spider curse that served as the story's basis. Search for information about the historical incident, the specific Suffolk town or village where it occurred, and any documented sources from that period that recorded both this spider-related event and a spectral army apparition in Ireland.

```
import os
import requests
from bs4 import BeautifulSoup
import json
import re
import time

print('=== FIXING TEXT EXTRACTION AND LOCATING "THE ASH TREE" STORY ===\n')
print('Objective: Fix the story boundary detection to properly extract the full story text')
print('Approach: Manually examine Project Gutenberg structure and improve parsing logic\n')

# Ensure workspace exists
workspace = 'workspace'
os.makedirs(workspace, exist_ok=True)

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.5',
    'Connection': 'keep-alive',
}

print('=== STEP 1: EXAMINING PROJECT GUTENBERG HTML STRUCTURE ===\n')

# Let's examine the HTML structure more carefully
gutenberg_html_url = 'https://www.gutenberg.org/files/8486/8486-h/8486-h.htm'
print(f'Accessing: {gutenberg_html_url}')

try:
    response = requests.get(gutenberg_html_url, headers=headers, timeout=30)
    print(f'Response status: {response.status_code}')
    print(f'Content length: {len(response.content):,} bytes\n')
    
    if response.status_code == 200:
        soup = BeautifulSoup(response.content, 'html.parser')
        
        # Save the raw HTML for inspection
        raw_html_file = os.path.join(workspace, 'gutenberg_raw_html.html')
        with open(raw_html_file, 'w', encoding='utf-8') as f:
            f.write(response.text)
        print(f'✓ Raw HTML saved to: {raw_html_file}')
        
        # Look for the table of contents or story structure
        print('=== ANALYZING HTML STRUCTURE FOR STORY BOUNDARIES ===\n')
        
        # Find all headings that might indicate story titles
        headings = soup.find_all(['h1', 'h2', 'h3', 'h4', 'h5', 'h6'])
        print(f'Found {len(headings)} headings in the document:')
        
        ash_tree_heading = None
        for i, heading in enumerate(headings):
            heading_text = heading.get_text().strip()
            print(f'  {i+1:2d}. {heading.name}: {heading_text}')
            
            if 'ash' in heading_text.lower() and 'tree' in heading_text.lower():
                ash_tree_heading = heading
                print(f'       *** FOUND ASH TREE HEADING! ***')
        
        print()
        
        if ash_tree_heading:
            print('=== EXTRACTING STORY USING HEADING STRUCTURE ===\n')
            print(f'Found "The Ash Tree" heading: {ash_tree_heading.get_text().strip()}')
            
            # Find the content after this heading until the next heading
            story_content = []
            current_element = ash_tree_heading.next_sibling
            
            while current_element:
                if hasattr(current_element, 'name'):
                    # If we hit another heading, we've reached the end of the story
                    if current_element.name in ['h1', 'h2', 'h3', 'h4', 'h5', 'h6']:
                        break
                    
                    # Collect text content
                    if current_element.name in ['p', 'div', 'blockquote']:
                        text = current_element.get_text().strip()
                        if text and len(text) > 10:  # Ignore very short fragments
                            story_content.append(text)
                
                current_element = current_element.next_sibling
            
            if story_content:
                full_story = '\n\n'.join(story_content)
                print(f'✓ Extracted story using heading structure: {len(full_story):,} characters')
                
                # Save the extracted story
                story_file = os.path.join(workspace, 'the_ash_tree_extracted.txt')
                with open(story_file, 'w', encoding='utf-8') as f:
                    f.write('"The Ash Tree" by M.R. James\n')
                    f.write(f'Source: Project Gutenberg HTML (heading-based extraction)\n')
                    f.write(f'Extracted: {time.strftime("%Y-%m-%d %H:%M:%S")}\n')
                    f.write('=' * 80 + '\n\n')
                    f.write(full_story)
                
                print(f'✓ Story saved to: {story_file}\n')
                
                # Show the beginning of the story to verify
                print('=== STORY PREVIEW ===\n')
                preview = full_story[:500] + '...' if len(full_story) > 500 else full_story
                print(preview)
                print()
                
            else:
                print('❌ No content found after the heading')
        
        else:
            print('=== TRYING ALTERNATIVE APPROACH: SEARCH FOR STORY MARKERS ===\n')
            
            # Get the full text and look for story markers
            full_text = soup.get_text()
            
            # Look for the story using different markers
            story_markers = [
                'THE ASH-TREE',
                'The Ash-tree',
                'THE ASH TREE',
                'The Ash Tree'
            ]
            
            story_found = False
            
            for marker in story_markers:
                marker_pos = full_text.find(marker)
                if marker_pos != -1:
                    print(f'Found story marker: "{marker}" at position {marker_pos}')
                    
                    # Look for the actual story start (usually after some whitespace)
                    story_start = marker_pos
                    
                    # Find a good starting point (look for the first substantial paragraph)
                    lines = full_text[story_start:story_start + 1000].split('\n')
                    actual_start = story_start
                    
                    for i, line in enumerate(lines):
                        line = line.strip()
                        if len(line) > 50 and not line.isupper() and not line.startswith('By '):
                            # Found the actual story start
                            actual_start = story_start + full_text[story_start:].find(line)
                            break
                    
                    # Look for story end markers
                    end_markers = [
                        'THE SCHOOL STORY',
                        'A VIEW FROM A HILL',
                        'NUMBER 13',
                        'COUNT MAGNUS',
                        'OH, WHISTLE',
                        '*** END OF',
                        'End of the Project Gutenberg'
                    ]
                    
                    story_end = len(full_text)
                    for end_marker in end_markers:
                        end_pos = full_text.find(end_marker, actual_start + 1000)
                        if end_pos != -1 and end_pos < story_end:
                            story_end = end_pos
                    
                    # Extract the story
                    extracted_story = full_text[actual_start:story_end].strip()
                    
                    # Clean up the text
                    lines = extracted_story.split('\n')
                    cleaned_lines = []
                    
                    for line in lines:
                        line = line.strip()
                        if line and not line.startswith('*') and len(line) > 2:
                            cleaned_lines.append(line)
                    
                    cleaned_story = '\n\n'.join(cleaned_lines)
                    
                    if len(cleaned_story) > 2000:  # Reasonable story length
                        print(f'✓ Successfully extracted story: {len(cleaned_story):,} characters')
                        
                        # Save the story
                        story_file = os.path.join(workspace, 'the_ash_tree_full_text.txt')
                        with open(story_file, 'w', encoding='utf-8') as f:
                            f.write('"The Ash Tree" by M.R. James\n')
                            f.write(f'Source: Project Gutenberg (marker-based extraction)\n')
                            f.write(f'Marker used: {marker}\n')
                            f.write(f'Extracted: {time.strftime("%Y-%m-%d %H:%M:%S")}\n')
                            f.write('=' * 80 + '\n\n')
                            f.write(cleaned_story)
                        
                        print(f'✓ Story saved to: {story_file}\n')
                        
                        # Preview the story
                        print('=== STORY PREVIEW ===\n')
                        preview = cleaned_story[:800] + '...' if len(cleaned_story) > 800 else cleaned_story
                        print(preview)
                        print()
                        
                        story_found = True
                        full_story = cleaned_story
                        break
                    else:
                        print(f'⚠ Extracted text still too short: {len(cleaned_story)} characters')
            
            if not story_found:
                print('❌ Could not extract story using marker approach')

except Exception as e:
    print(f'❌ Error accessing Project Gutenberg: {str(e)}')
    story_found = False

print('\n=== STEP 2: TRYING WIKISOURCE DIRECT APPROACH ===\n')

# Try Wikisource which should have the story isolated
wikisource_url = 'https://en.wikisource.org/wiki/Ghost_Stories_of_an_Antiquary/The_Ash-tree'
print(f'Accessing Wikisource: {wikisource_url}')

try:
    response = requests.get(wikisource_url, headers=headers, timeout=30)
    print(f'Response status: {response.status_code}')
    
    if response.status_code == 200:
        soup = BeautifulSoup(response.content, 'html.parser')
        
        # Wikisource usually has the content in a specific div
        content_selectors = [
            '.mw-parser-output',
            '#mw-content-text',
            '.prp-pages-output',
            '.ws-summary'
        ]
        
        wikisource_story = None
        
        for selector in content_selectors:
            content_div = soup.select_one(selector)
            if content_div:
                print(f'Found content using selector: {selector}')
                
                # Remove navigation and metadata
                for unwanted in content_div.find_all(['div', 'table', 'nav'], class_=re.compile(r'navbox|navigation|header|footer|metadata')):
                    unwanted.decompose()
                
                # Get the text content
                story_text = content_div.get_text()
                
                # Clean up the text
                lines = story_text.split('\n')
                cleaned_lines = []
                
                for line in lines:
                    line = line.strip()
                    if line and len(line) > 3 and not line.startswith('[') and not line.startswith('◄'):
                        cleaned_lines.append(line)
                
                cleaned_story = '\n\n'.join(cleaned_lines)
                
                if len(cleaned_story) > 1000:
                    wikisource_story = cleaned_story
                    print(f'✓ Extracted from Wikisource: {len(wikisource_story):,} characters')
                    
                    # Save the Wikisource version
                    wikisource_file = os.path.join(workspace, 'the_ash_tree_wikisource.txt')
                    with open(wikisource_file, 'w', encoding='utf-8') as f:
                        f.write('"The Ash Tree" by M.R. James\n')
                        f.write(f'Source: Wikisource\n')
                        f.write(f'Extracted: {time.strftime("%Y-%m-%d %H:%M:%S")}\n')
                        f.write('=' * 80 + '\n\n')
                        f.write(wikisource_story)
                    
                    print(f'✓ Wikisource version saved to: {wikisource_file}\n')
                    
                    # Use this as our story text if we don't have one already
                    if 'story_found' not in locals() or not story_found:
                        full_story = wikisource_story
                        story_found = True
                    
                    break
        
        if not wikisource_story:
            print('❌ Could not extract story from Wikisource')
    
except Exception as e:
    print(f'❌ Error accessing Wikisource: {str(e)}')

print('\n=== STEP 3: ANALYZING STORY FOR SUFFOLK LOCATION CLUES ===\n')

if 'full_story' in locals() and full_story:
    print('✅ Successfully obtained story text - proceeding with location analysis')
    print(f'Story length: {len(full_story):,} characters\n')
    
    # Comprehensive location analysis
    print('Searching for location and historical references...')
    
    # Enhanced keyword lists based on M.R. James research
    location_keywords = {
        'suffolk_places': ['suffolk', 'bury st edmunds', 'bury', 'ipswich', 'felixstowe', 'lowestoft', 
                          'newmarket', 'sudbury', 'haverhill', 'thetford', 'diss', 'eye', 'framlingham',
                          'aldeburgh', 'southwold', 'woodbridge', 'stowmarket', 'beccles'],
        
        'neighboring_counties': ['norfolk', 'essex', 'cambridge', 'cambridgeshire', 'hertfordshire'],
        
        'property_types': ['hall', 'manor', 'house', 'court', 'grange', 'priory', 'abbey', 'rectory', 
                          'vicarage', 'castle', 'tower', 'lodge'],
        
        'locations': ['village', 'parish', 'church', 'churchyard', 'cemetery', 'town', 'hamlet', 
                     'estate', 'park', 'green', 'common', 'lane', 'street', 'road'],
        
        'historical_periods': ['17th century', '1600s', 'sixteen', 'charles', 'cromwell', 'civil war', 
                              'restoration', 'commonwealth', 'stuart', 'puritan', 'cavalier'],
        
        'supernatural_elements': ['spider', 'arachnid', 'web', 'curse', 'witch', 'witchcraft', 'trial', 
                                 'hanging', 'execution', 'gallows', 'tree', 'ash', 'ghost', 'spirit', 
                                 'apparition', 'haunted', 'supernatural'],
        
        'ireland_connection': ['ireland', 'irish', 'dublin', 'cork', 'belfast', 'spectral', 'army', 
                              'soldiers', 'battle', 'war']
    }
    
    story_lower = full_story.lower()
    analysis_results = {}
    
    # Search for each category of keywords
    for category, keywords in location_keywords.items():
        found_in_category = {}
        
        for keyword in keywords:
            if keyword in story_lower:
                # Find all positions of this keyword
                positions = []
                start = 0
                while True:
                    pos = story_lower.find(keyword, start)
                    if pos == -1:
                        break
                    positions.append(pos)
                    start = pos + 1
                
                # Extract context for each occurrence
                contexts = []
                for pos in positions:
                    context_start = max(0, pos - 200)
                    context_end = min(len(full_story), pos + 200)
                    context = full_story[context_start:context_end].replace('\n', ' ').strip()
                    contexts.append(context)
                
                found_in_category[keyword] = {
                    'count': len(positions),
                    'positions': positions,
                    'contexts': contexts
                }
        
        if found_in_category:
            analysis_results[category] = found_in_category
    
    # Display results by category
    print('=== LOCATION ANALYSIS RESULTS ===\n')
    
    for category, findings in analysis_results.items():
        print(f'{category.upper().replace("_", " ")} ({len(findings)} terms found):')
        
        for keyword, data in findings.items():
            print(f'  • "{keyword}" - {data["count"]} occurrence(s)')
            
            # Show first context for each keyword
            if data['contexts']:
                context = data['contexts'][0]
                print(f'    Context: ...{context}...')
            print()
    
    # Look for specific place name patterns with regex
    print('=== SEARCHING FOR SPECIFIC PLACE NAMES ===\n')
    
    place_patterns = [
        (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Hall|Manor|House|Court|Grange)', 'Estate names'),
        (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Church|Parish|Village)', 'Religious/civic places'),
        (r'the\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(estate|property|lands)', 'Property references'),
        (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+in\s+(Suffolk|Norfolk|Essex)', 'County-specific places')
    ]
    
    all_place_matches = []
    
    for pattern, description in place_patterns:
        matches = list(re.finditer(pattern, full_story, re.IGNORECASE))
        if matches:
            print(f'{description}: {len(matches)} match(es)')
            
            for match in matches:
                place_name = match.group(1).strip()
                full_match = match.group(0)
                position = match.start()
                
                # Extract context
                context_start = max(0, position - 150)
                context_end = min(len(full_story), position + 150)
                context = full_story[context_start:context_end].replace('\n', ' ').strip()
                
                place_info = {
                    'place_name': place_name,
                    'full_match': full_match,
                    'pattern_type': description,
                    'context': context,
                    'position': position
                }
                
                all_place_matches.append(place_info)
                
                print(f'  → "{place_name}" (from "{full_match}")') 
                print(f'    ...{context}...')
                print()
    
    # Save comprehensive analysis
    final_analysis = {
        'story_source': 'Project Gutenberg/Wikisource',
        'story_length': len(full_story),
        'extraction_timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
        'keyword_analysis': analysis_results,
        'place_name_matches': all_place_matches,
        'summary': {
            'total_keywords_found': sum(len(findings) for findings in analysis_results.values()),
            'total_place_matches': len(all_place_matches),
            'categories_with_matches': list(analysis_results.keys())
        }
    }
    
    analysis_file = os.path.join(workspace, 'ash_tree_comprehensive_analysis.json')
    with open(analysis_file, 'w', encoding='utf-8') as f:
        json.dump(final_analysis, f, indent=2, ensure_ascii=False)
    
    print(f'✓ Comprehensive analysis saved to: {analysis_file}')
    
else:
    print('❌ No story text available for analysis')
    print('Manual approach needed: examining M.R. James biographical sources')

print('\n=== FINAL STATUS ===\n')

if 'story_found' in locals() and story_found:
    print('✅ SUCCESS: Located and analyzed "The Ash Tree" story text')
    if 'analysis_results' in locals():
        print(f'📊 Found keywords in {len(analysis_results)} categories')
        print(f'🏛️ Identified {len(all_place_matches) if "all_place_matches" in locals() else 0} potential place names')
        
        # Highlight most promising findings
        if 'suffolk_places' in analysis_results:
            print(f'🎯 Direct Suffolk references: {list(analysis_results["suffolk_places"].keys())}')
        
        if 'historical_periods' in analysis_results:
            print(f'📅 Historical period references: {list(analysis_results["historical_periods"].keys())}')
else:
    print('❌ Still unable to extract full story text')
    print('📋 Recommendation: Try manual examination of Project Gutenberg HTML structure')

print(f'\n📁 Current workspace files:')
for file in sorted(os.listdir(workspace)):
    filepath = os.path.join(workspace, file)
    size = os.path.getsize(filepath)
    print(f'  - {file} ({size:,} bytes)')
```
