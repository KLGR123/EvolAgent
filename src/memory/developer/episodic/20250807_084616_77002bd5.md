    print(f'Accessing: {source["name"]}')

    try:
        response = requests.get(source['url'], headers=headers, timeout=30)
        print(f'Response status: {response.status_code}')
        
        if response.status_code == 200:
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Remove scripts, styles, navigation
            for element in soup(['script', 'style', 'nav', 'header', 'footer', 'aside']):
                element.decompose()
            
            page_text = soup.get_text()
            print(f'Content length: {len(page_text):,} characters')
            
            # Search for Suffolk-related content
            suffolk_keywords = ['suffolk', 'bury st edmunds', 'aldeburgh', 'framlingham', 'eye', 'clare', 'diss', 'antiquarian', 'archaeological', 'manuscript', 'parish']
            
            found_keywords = {}
            page_lower = page_text.lower()
            
            for keyword in suffolk_keywords:
                if keyword in page_lower:
                    positions = []
                    start = 0
                    while True:
                        pos = page_lower.find(keyword, start)
                        if pos == -1:
                            break
                        positions.append(pos)
                        start = pos + 1
                    
                    if positions:
                        # Get context for first occurrence
                        pos = positions[0]
                        context_start = max(0, pos - 200)
                        context_end = min(len(page_text), pos + 200)
                        context = page_text[context_start:context_end].replace('\n', ' ').strip()
                        
                        found_keywords[keyword] = {
                            'count': len(positions),
                            'context': context
                        }
            
            if found_keywords:
                print(f'✓ Found {len(found_keywords)} Suffolk-related keywords')
                for keyword, data in found_keywords.items():
                    print(f'  - "{keyword}": {data["count"]} occurrence(s)')
                    print(f'    Context: ...{data["context"]}...')
                    print()
                
                biographical_findings.append({
                    'source': source['name'],
                    'url': source['url'],
                    'keywords_found': found_keywords
                })
            else:
                print('  - No Suffolk-related keywords found')
        else:
            print(f'  - Failed to access (Status: {response.status_code})')
    
    except Exception as e:
        print(f'  - Error: {str(e)}')
