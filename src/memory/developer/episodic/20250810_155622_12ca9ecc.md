```
import os
import json
import re
from ddgs import DDGS

print('=== SURREY LODGE ORGANIZATION & FOUNDER IDENTIFICATION ===\n')
print('Objective: Extract the specific organization that developed Surrey Lodge and identify its founder\'s birth year')
print('Strategy: Fix scoping issues and systematically extract key information, then search for founder details\n')

print('=== STEP 1: LOADING AND ANALYZING RESEARCH DATA ===\n')

# Load the main JSON research file
json_file = 'workspace/surrey_lodge_research_results.json'
with open(json_file, 'r', encoding='utf-8') as f:
    research_data = json.load(f)

print(f'Research data loaded successfully')
print(f'Timestamp: {research_data.get("timestamp", "Unknown")}')
print(f'Total findings: {research_data.get("total_findings", "Unknown")}')
print(f'Success rate: {research_data.get("success_rate", "Unknown")}%')

# Extract the most promising finding from the research
if 'top_findings' in research_data and len(research_data['top_findings']) > 0:
    primary_finding = research_data['top_findings'][0]
    
    print('\n--- ANALYZING PRIMARY FINDING ---')
    print(f'Title: {primary_finding.get("title", "N/A")}')
    print(f'Body: {primary_finding.get("body", "N/A")}')
    print(f'URL: {primary_finding.get("url", "N/A")}')
    print(f'Relevance Score: {primary_finding.get("relevance_score", "N/A")}')
    print(f'Evidence: {primary_finding.get("evidence", "N/A")}')
    
    # The primary finding shows: "Founder of Organization That Developed Surrey Lodge South London Victorian Era Temperance Hall"
    # This suggests we need to search for more specific information about this organization and founder

# Load the text summary report
txt_file = 'workspace/surrey_lodge_summary_report.txt'
with open(txt_file, 'r', encoding='utf-8') as f:
    summary_content = f.read()

print('\n=== STEP 2: ANALYZING TEXT SUMMARY FOR SPECIFIC DETAILS ===\n')

lines = summary_content.splitlines()
organization_lines = []
founder_lines = []
birth_lines = []

# Fixed variable scoping - define variables properly in loop scope
for line in lines:
    line_clean = line.strip()
    line_lower = line_clean.lower()  # Define line_lower within the loop scope
    
    # Look for organization mentions
    org_keywords = ['organization', 'company', 'society', 'association', 'developed']
    if any(term in line_lower for term in org_keywords):
        organization_lines.append(line_clean)
    
    # Look for founder mentions
    founder_keywords = ['founder', 'founded', 'established', 'created']
    if any(term in line_lower for term in founder_keywords):
        founder_lines.append(line_clean)
    
    # Look for birth year mentions (years starting with 18 or 19)
    birth_keywords = ['birth', 'born']
    if any(term in line_lower for term in birth_keywords) or re.search(r'\b(18|19)\d{2}\b', line_clean):
        birth_lines.append(line_clean)

print('ORGANIZATION-RELATED INFORMATION:')
if organization_lines:
    for i, info in enumerate(organization_lines, 1):
        print(f'  {i}. {info}')
else:
    print('  No explicit organization information found in summary')

print('\nFOUNDER-RELATED INFORMATION:')
if founder_lines:
    for i, info in enumerate(founder_lines, 1):
        print(f'  {i}. {info}')
else:
    print('  No explicit founder information found in summary')

print('\nBIRTH/DATE-RELATED INFORMATION:')
if birth_lines:
    for i, info in enumerate(birth_lines, 1):
        print(f'  {i}. {info}')
else:
    print('  No explicit birth year information found in summary')

print('\n=== STEP 3: ANALYZING FOURTH FINDING FOR ORGANIZATION CLUES ===\n')

# The fourth finding mentions "Surrey Lodge No 266 United Ancient Order of Druids UAOD" - this could be key
if len(research_data['top_findings']) >= 4:
    fourth_finding = research_data['top_findings'][3]
    print('FOURTH FINDING ANALYSIS:')
    print(f'Title: {fourth_finding.get("title", "N/A")}')
    print(f'Body: {fourth_finding.get("body", "N/A")}')
    
    # Extract potential organization name
    body_text = fourth_finding.get('body', '')
    if 'United Ancient Order of Druids' in body_text:
        print('\n🎯 POTENTIAL ORGANIZATION IDENTIFIED:')
        print('United Ancient Order of Druids (UAOD)')
        print('Surrey Lodge No 266 appears to be associated with this organization')
        
        organization_name = 'United Ancient Order of Druids'
        print(f'\nOrganization to research: {organization_name}')

print('\n=== STEP 4: SEARCHING FOR UNITED ANCIENT ORDER OF DRUIDS FOUNDER ===\n')

# Search for information about the United Ancient Order of Druids and its founder
searcher = DDGS(timeout=10)
query = 'United Ancient Order of Druids founder history established'
print(f'Searching for: {query}')

try:
    search_results = searcher.text(query, max_results=5, backend=['google', 'duckduckgo'], region='en-us')
    
    if search_results:
        print(f'\nFound {len(search_results)} results about UAOD:')
        
        for i, result in enumerate(search_results, 1):
            print(f'\nResult {i}:')
            print(f'  Title: {result.get("title", "N/A")}')
            print(f'  Body: {result.get("body", "N/A")[:200]}...')
            print(f'  URL: {result.get("href", "N/A")}')
            
            # Look for founder information in the results
            body_text = result.get('body', '').lower()
            title_text = result.get('title', '').lower()
            
            # Search for founder names and dates
            founder_patterns = [
                r'founded by ([A-Z][a-z]+ [A-Z][a-z]+)',
                r'established by ([A-Z][a-z]+ [A-Z][a-z]+)',
                r'founder ([A-Z][a-z]+ [A-Z][a-z]+)',
                r'([A-Z][a-z]+ [A-Z][a-z]+) founded',
                r'([A-Z][a-z]+ [A-Z][a-z]+) established'
            ]
            
            combined_text = title_text + ' ' + body_text
            
            for pattern in founder_patterns:
                matches = re.findall(pattern, combined_text, re.IGNORECASE)
                if matches:
                    print(f'    🎯 POTENTIAL FOUNDER(S): {matches}')
            
            # Look for birth years
            birth_years = re.findall(r'\b(17|18|19)\d{2}\b', combined_text)
            if birth_years:
                print(f'    📅 YEARS MENTIONED: {set(birth_years)}')
    
    else:
        print('No search results found for UAOD founder')
        
except Exception as e:
    print(f'Search error: {str(e)}')

print('\n=== STEP 5: TARGETED SEARCH FOR UAOD FOUNDER BIRTH YEAR ===\n')

# More specific search for UAOD founder and birth year
specific_queries = [
    'United Ancient Order of Druids founder birth year',
    'UAOD Ancient Order Druids established founder born',
    '"United Ancient Order of Druids" history founder birth'
]

founder_candidates = []
birth_year_candidates = []

for query in specific_queries:
    print(f'Searching: {query}')
    try:
        results = searcher.text(query, max_results=3, backend=['google', 'duckduckgo'], region='en-us')
        
        for result in results:
            body_text = result.get('body', '')
            title_text = result.get('title', '')
            combined_text = title_text + ' ' + body_text
            
            # Extract potential founder names
            founder_matches = re.findall(r'([A-Z][a-z]+ [A-Z][a-z]+)(?:\s+(?:founded|established|created))', combined_text, re.IGNORECASE)
            founder_candidates.extend(founder_matches)
            
            # Extract birth years
            birth_matches = re.findall(r'\b(17|18|19)\d{2}\b', combined_text)
            birth_year_candidates.extend(birth_matches)
            
            # Look for specific birth year patterns
            birth_patterns = [
                r'born\s+(\d{4})',
                r'birth\s+(\d{4})',
                r'\((\d{4})[-–]',
                r'b\.\s*(\d{4})'
            ]
            
            for pattern in birth_patterns:
                birth_matches = re.findall(pattern, combined_text, re.IGNORECASE)
                birth_year_candidates.extend(birth_matches)
    
    except Exception as e:
        print(f'  Search error: {str(e)}')

print('\n=== STEP 6: COMPILING FINDINGS ===\n')

# Remove duplicates and analyze findings
unique_founders = list(set(founder_candidates))
unique_birth_years = list(set(birth_year_candidates))

print('COMPILATION OF RESEARCH FINDINGS:')
print('\n🏛️ ORGANIZATION IDENTIFIED:')
print('  • United Ancient Order of Druids (UAOD)')
print('  • Surrey Lodge No 266 was associated with this organization')
print('  • This appears to be the organization that developed Surrey Lodge')

if unique_founders:
    print(f'\n👤 POTENTIAL FOUNDER(S) IDENTIFIED ({len(unique_founders)}):')
    for founder in unique_founders:
        print(f'  • {founder}')
else:
    print('\n❌ No specific founder names extracted from search results')

if unique_birth_years:
    print(f'\n📅 POTENTIAL BIRTH YEARS FOUND ({len(unique_birth_years)}):')
    for year in sorted(unique_birth_years):
        print(f'  • {year}')
else:
    print('\n❌ No specific birth years extracted from search results')

# Save comprehensive results
final_results = {
    'timestamp': '2024-12-19',
    'objective': 'Identify Surrey Lodge development organization and founder birth year',
    'organization_identified': 'United Ancient Order of Druids (UAOD)',
    'surrey_lodge_connection': 'Surrey Lodge No 266 United Ancient Order of Druids',
    'evidence_source': 'Surrey Lodge research findings - fourth result',
    'founder_candidates': unique_founders,
    'birth_year_candidates': unique_birth_years,
    'research_status': 'Organization identified, founder search conducted',
    'next_steps': [
        'Verify UAOD as the correct organization that developed Surrey Lodge',
        'Conduct more targeted biographical research on UAOD founders',
        'Cross-reference with historical records for birth year confirmation'
    ]
}

with open('workspace/surrey_lodge_final_identification.json', 'w', encoding='utf-8') as f:
    json.dump(final_results, f, indent=2)

print('\n💾 Final identification results saved to: workspace/surrey_lodge_final_identification.json')

print('\n=== CONCLUSION ===\n')
print('🎯 ORGANIZATION: United Ancient Order of Druids (UAOD)')
print('🏠 CONNECTION: Surrey Lodge No 266 was associated with UAOD')
print('📍 LOCATION: South London, Victorian era')
print('🎭 TRANSFORMATION: Became temperance hall with first UK rotating stage (1935)')

if unique_founders and unique_birth_years:
    print(f'👤 MOST LIKELY FOUNDER: {unique_founders[0] if unique_founders else "To be determined"}')
    print(f'📅 POTENTIAL BIRTH YEAR: {min(unique_birth_years) if unique_birth_years else "To be determined"}')
else:
    print('❓ FOUNDER AND BIRTH YEAR: Require additional targeted biographical research')
```
