**Description**: Search for information about the comic book 'Little Monsters' to identify its illustrator/artist. Focus on finding the artist who illustrated this comic and any information about their collaborations with writers. Look for official publication details, creator credits, and any collaborative relationships mentioned in connection with this work.

**Use Cases**:

- Conducting academic research on comic book collaboration networks by scraping historical HTML sources for creator attributions and mapping writer-artist partnerships
- Enhancing e-commerce product listings by mining third-party vendor HTML for illustrator names to enrich item metadata and improve search relevance
- Enforcing quality control in a web-publishing workflow by automatically checking that online comic pages include correct credit phrases before publication
- Generating bibliographic records for an academic repository by parsing conference and online journal HTML for ‚Äústory by‚Äù and ‚Äúart by‚Äù tags to compile publication author lists
- Supporting legal due diligence in M&A transactions by extracting creator credits from archived HTML content to assess intellectual property ownership and licensing terms

```
import os
import json

print("=== LITTLE MONSTERS COMIC ARTIST IDENTIFICATION ===")
print("Final safe approach - no variable scoping issues")
print("=" * 70)

# Check workspace directory
workspace_dir = 'workspace'
if not os.path.exists(workspace_dir):
    print("No workspace directory found")
    os.makedirs(workspace_dir, exist_ok=True)

print(f"\nChecking workspace files...")
all_files = os.listdir(workspace_dir)
html_files = [f for f in all_files if f.endswith('.html')]
print(f"HTML files available: {len(html_files)}")

# Target the specific files mentioned in HISTORY
target_files = ['comicvine_search.html', 'mycomicshop_search.html']
final_candidates = []

for file_num in range(len(target_files)):
    current_file = target_files[file_num]
    file_path = os.path.join(workspace_dir, current_file)
    
    if not os.path.exists(file_path):
        print(f"\n{current_file} - NOT FOUND")
        continue
    
    print(f"\n{'='*50}")
    print(f"PROCESSING: {current_file}")
    print(f"{'='*50}")
    
    try:
        # Read file content
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            content = f.read()
        
        print(f"‚úì Loaded {len(content):,} characters")
        
        # Search for Little Monsters
        content_lower = content.lower()
        lm_count = content_lower.count('little monsters')
        print(f"'Little Monsters' mentions: {lm_count}")
        
        if lm_count > 0:
            print(f"\nüîç Searching for comic creator patterns...")
            
            # Look for specific creator-related phrases
            creator_phrases = [
                'created by',
                'written by', 
                'art by',
                'artist:',
                'writer:',
                'illustrated by',
                'drawn by',
                'pencils by',
                'story by'
            ]
            
            found_creator_phrases = []
            for phrase in creator_phrases:
                if phrase in content_lower:
                    found_creator_phrases.append(phrase)
            
            if found_creator_phrases:
                print(f"  Found creator phrases: {found_creator_phrases}")
            else:
                print(f"  No standard creator phrases found")
            
            # Check for Gold Key Comics (known publisher of Little Monsters)
            if 'gold key' in content_lower:
                print(f"  ‚úì Found Gold Key Comics reference")
            
            # Extract contexts around Little Monsters mentions
            print(f"\nüìù Analyzing contexts...")
            
            # Find positions manually to avoid variable scoping
            positions = []
            search_pos = 0
            while len(positions) < 10:  # Limit to 10 positions
                found_pos = content_lower.find('little monsters', search_pos)
                if found_pos == -1:
                    break
                positions.append(found_pos)
                search_pos = found_pos + 1
            
            print(f"  Found {len(positions)} occurrences to analyze")
            
            # Process each position
            for pos_idx in range(min(3, len(positions))):  # Only first 3
                pos = positions[pos_idx]
                
                # Extract context
                start = max(0, pos - 200)
                end = min(len(content), pos + 200)
                context = content[start:end]
                
                print(f"\n  Context {pos_idx + 1}:")
                
                # Clean context for display
                lines = context.split('\n')
                clean_lines = []
                for line in lines:
                    stripped = line.strip()
                    if len(stripped) > 5 and len(stripped) < 150:
                        clean_lines.append(stripped)
                
                # Show first few meaningful lines
                for line_idx in range(min(3, len(clean_lines))):
                    line = clean_lines[line_idx]
                    if 'little monsters' in line.lower():
                        print(f"    ‚Üí {line}")
                    else:
                        print(f"      {line[:80]}..." if len(line) > 80 else f"      {line}")
                
                # Look for potential artist names in this context
                words = context.split()
                
                # Manual name detection to avoid scoping issues
                for word_idx in range(len(words) - 1):  # -1 to check next word
                    word1 = words[word_idx]
                    word2 = words[word_idx + 1]
                    
                    # Clean words
                    clean1 = word1.strip('.,;:()[]{}"\'-<>')
                    clean2 = word2.strip('.,;:()[]{}"\'-<>')
                    
                    # Check if both look like name parts
                    if (len(clean1) > 2 and len(clean1) < 20 and
                        len(clean2) > 2 and len(clean2) < 20 and
                        clean1[0].isupper() and clean1.isalpha() and
                        clean2[0].isupper() and clean2.isalpha()):
                        
                        potential_name = f"{clean1} {clean2}"
                        
                        # Filter out obvious non-names
                        bad_names = [
                            'Little Monsters', 'Gold Key', 'Comic Book', 'Comic Vine',
                            'Search Results', 'Web Site', 'Home Page', 'Click Here',
                            'New York', 'Los Angeles', 'United States', 'More Info',
                            'Vampire Slayer', 'Slayer Ugly'  # Known false positives
                        ]
                        
                        is_bad = False
                        for bad_name in bad_names:
                            if bad_name.lower() in potential_name.lower():
                                is_bad = True
                                break
                        
                        if not is_bad:
                            # Check if we already have this name
                            already_found = False
                            for existing in final_candidates:
                                if existing['name'] == potential_name:
                                    already_found = True
                                    break
                            
                            if not already_found:
                                candidate = {
                                    'name': potential_name,
                                    'source': current_file,
                                    'context_num': pos_idx + 1
                                }
                                final_candidates.append(candidate)
                                print(f"    ‚úì Found potential artist: {potential_name}")
        
    except Exception as e:
        print(f"Error processing {current_file}: {e}")
        continue

print(f"\n{'='*70}")
print("FINAL ARTIST IDENTIFICATION RESULTS")
print(f"{'='*70}")

if len(final_candidates) > 0:
    print(f"\nüé® Found {len(final_candidates)} potential artists:")
    
    # Count occurrences manually
    name_counts = {}
    for candidate in final_candidates:
        artist_name = candidate['name']
        if artist_name in name_counts:
            name_counts[artist_name] = name_counts[artist_name] + 1
        else:
            name_counts[artist_name] = 1
    
    # Convert to list and sort manually
    name_list = []
    for artist_name in name_counts:
        count = name_counts[artist_name]
        name_list.append((artist_name, count))
    
    # Manual sort by count (descending)
    name_list.sort(key=lambda x: x[1], reverse=True)
    
    print(f"\nRanked by frequency:")
    for rank in range(len(name_list)):
        artist_name, count = name_list[rank]
        print(f"  {rank + 1}. {artist_name} - {count} mention(s)")
        
        # Show sources manually
        sources = []
        for candidate in final_candidates:
            if candidate['name'] == artist_name:
                source = candidate['source']
                if source not in sources:
                    sources.append(source)
        
        print(f"     Sources: {', '.join(sources)}")
    
    # Determine top candidate
    if len(name_list) > 0:
        top_name, top_count = name_list[0]
        print(f"\n*** MOST LIKELY LITTLE MONSTERS ARTIST: {top_name} ***")
        print(f"*** CONFIDENCE: {top_count} mention(s) ***")
        
        result = {
            'status': 'SUCCESS',
            'artist': top_name,
            'confidence': top_count,
            'total_candidates': len(name_counts)
        }
    else:
        result = {
            'status': 'NO_CANDIDATES',
            'reason': 'No valid artist names extracted'
        }
else:
    print(f"\n‚ùå NO ARTIST CANDIDATES FOUND")
    print(f"\nThis could indicate:")
    print(f"1. The HTML files don't contain clear creator credits")
    print(f"2. Artist information is in a different format than expected")
    print(f"3. Little Monsters creator info may not be in these sources")
    
    result = {
        'status': 'NO_RESULTS',
        'reason': 'No artist candidates extracted from HTML sources'
    }

# Save final results
final_data = {
    'comic_title': 'Little Monsters',
    'search_method': 'Safe manual extraction',
    'timestamp': '2024-12-19',
    'files_processed': target_files,
    'candidates_found': final_candidates,
    'final_result': result
}

results_file = os.path.join(workspace_dir, 'little_monsters_final_safe_results.json')
with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(final_data, f, indent=2, ensure_ascii=False)

print(f"\n‚úì Final results saved to: {results_file}")

print(f"\n{'='*70}")
print("LITTLE MONSTERS ARTIST SEARCH COMPLETED")
print(f"{'='*70}")

if result['status'] == 'SUCCESS':
    print(f"\n‚úÖ ARTIST IDENTIFIED: {result['artist']}")
    print(f"Confidence: {result['confidence']} mentions")
    print(f"Total candidates evaluated: {result['total_candidates']}")
elif result['status'] == 'NO_CANDIDATES':
    print(f"\n‚ö†Ô∏è No valid artist candidates found")
    print(f"Reason: {result['reason']}")
else:
    print(f"\n‚ùå No results")
    print(f"Reason: {result['reason']}")
    print(f"\nNote: Previous attempts found false positives like 'Vampire Slayer'")
    print(f"which were filtered out as they're not actual artist names.")

print(f"\nComplete analysis saved to: {results_file}")
print(f"\nIf no clear artist was found, this suggests the HTML sources")
print(f"may not contain the specific Little Monsters creator information.")
```
