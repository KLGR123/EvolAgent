import json
import requests
from bs4 import BeautifulSoup
import time

# Create workspace directory
os.makedirs('workspace', exist_ok=True)

print("=== STAR TREK ALIENS COMIC SEARCH - FINAL VERIFICATION ===")
print("Goal: Verify Star Trek: Alien Spotlight - Trill comic and establish collaboration chain")

# Define headers at the very beginning to avoid scope issues
HTTP_HEADERS = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.5',
    'Connection': 'keep-alive'
}

def safe_web_request(url, timeout=20):
    """Make HTTP request with proper error handling using global headers"""
    try:
        print(f"Requesting: {url}")
        response = requests.get(url, headers=HTTP_HEADERS, timeout=timeout)
        response.raise_for_status()
        print(f"✓ Success: {response.status_code} - Content: {len(response.content)} bytes")
        return response
    except requests.exceptions.RequestException as e:
        print(f"✗ Request failed: {e}")
        return None
    except Exception as e:
        print(f"✗ Unexpected error: {e}")
        return None

# First, inspect the comprehensive analysis file to understand our hypothesis
print("\n=== INSPECTING EXISTING STAR TREK ANALYSIS ===")
analysis_file = 'workspace/star_trek_comprehensive_final_analysis.json'

if os.path.exists(analysis_file):
    print(f"Found: {analysis_file}")
    try:
        with open(analysis_file, 'r') as f:
            data = json.load(f)

        # Extract the final hypothesis safely
        if 'final_hypothesis' in data:
            hypothesis = data['final_hypothesis']
            print("\nEXISTING HYPOTHESIS:")
            print(f"  Comic: {hypothesis.get('comic_identification', {}).get('most_likely_title', 'Not specified')}")
            print(f"  Writers: {hypothesis.get('creator_hypothesis', {}).get('writers', 'Not specified')}")
            print(f"  Artist: {hypothesis.get('creator_hypothesis', {}).get('artist', 'Not specified')}")
            print(f"  Story confidence: {hypothesis.get('overall_confidence', {}).get('story_matches_criteria', 'Not specified')}")
            print(f"  Connection status: {hypothesis.get('connection_to_lemire', {}).get('verification_status', 'Not specified')}")
        else:
            print("No final_hypothesis found in file")
    except Exception as e:
        print(f"Error reading analysis file: {e}")
else:
    print("No existing comprehensive analysis found")

print("\n=== STEP 1: VERIFY STAR TREK: ALIEN SPOTLIGHT SERIES ===")
print("Searching for concrete evidence of the series and Trill issue...")

# Search targets for verification
search_targets = [
    {
        'name': 'Comic Book Database',
        'url': 'https://comicbookdb.com/search.php?form_search=Star+Trek+Alien+Spotlight+Trill',
        'focus': 'Comprehensive comic database'
    },
    {
        'name': 'Grand Comics Database',
        'url': 'https://www.comics.org/search/?target=series&method=icontains&logic=False&keywords=Star+Trek+Alien+Spotlight',
        'focus': 'Authoritative comic indexing'
    }
]

verification_results = {
    'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'search_attempts': [],
    'evidence_found': {
        'alien_spotlight_series': False,
        'trill_issue': False,
        'tipton_brothers': False,
        'jk_woodward': False
    },
    'key_discoveries': []
}

print("\nConducting verification searches...")
for target in search_targets:
    print(f"\n--- {target['name']} ---")
    print(f"URL: {target['url']}")
    
    response = safe_web_request(target['url'])
    
    search_result = {
        'source': target['name'],
        'url': target['url'],
        'success': response is not None,
        'findings': {},
        'content_length': 0
    }
    
    if response:
        try:
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Remove script and style elements
            for element in soup(["script", "style"]):
                element.decompose()
            
            # Extract clean text
            text = soup.get_text()
            lines = (line.strip() for line in text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
            clean_text = ' '.join(chunk for chunk in chunks if chunk)
            
            search_result['content_length'] = len(clean_text)
            print(f"  Content extracted: {len(clean_text)} characters")
            
            # Search for key terms
            search_terms = {
                'alien_spotlight': ['alien spotlight', 'alien-spotlight'],
                'trill': ['trill'],
                'scott_tipton': ['scott tipton'],
                'david_tipton': ['david tipton'],
                'tipton': ['tipton'],
                'jk_woodward': ['j.k. woodward', 'jk woodward', 'j k woodward'],
                'woodward': ['woodward'],
                'idw': ['idw publishing', 'idw']
            }
            
            for term_category, variations in search_terms.items():
                found = False
                found_variation = None
                
                for variation in variations:
                    if variation.lower() in clean_text.lower():
                        found = True
                        found_variation = variation
                        print(f"  ✓ Found '{variation}' in content")
                        
                        # Update evidence tracking
                        if term_category == 'alien_spotlight':
                            verification_results['evidence_found']['alien_spotlight_series'] = True
                        elif term_category == 'trill':
                            verification_results['evidence_found']['trill_issue'] = True
                        elif 'tipton' in term_category:
                            verification_results['evidence_found']['tipton_brothers'] = True
                        elif 'woodward' in term_category:
                            verification_results['evidence_found']['jk_woodward'] = True
                        
                        break
                
                search_result['findings'][term_category] = {
                    'found': found,
                    'variation': found_variation if found else None
                }
            
            # Save content sample for inspection
            filename = f"workspace/{target['name'].lower().replace(' ', '_')}_search_content.txt"
            with open(filename, 'w', encoding='utf-8') as f:
                f.write(f"{target['name']} Search Results\n")
                f.write(f"URL: {target['url']}\n")
                f.write(f"Retrieved: {time.strftime('%Y-%m-%d %H:%M:%S')}\n")
                f.write(f"{'='*50}\n\n")
                f.write(clean_text[:25000])  # Save first 25k characters
            
            print(f"  Content saved to: {filename}")
            
        except Exception as e:
            print(f"  Error processing content: {e}")
    
    verification_results['search_attempts'].append(search_result)
    time.sleep(3)  # Be respectful with requests
