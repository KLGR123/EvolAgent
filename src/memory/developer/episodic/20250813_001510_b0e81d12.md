# Attempt to access Philippine government sources for verification
def search_philippine_official_sources():
    print("Attempting to access Philippine Official Gazette and government sources...")
    
    # Try to access Philippine government websites
    philippine_sources = [
        "https://www.officialgazette.gov.ph",
        "https://www.gov.ph",

    ]

        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
    }
    
    access_results = {}
    
    for source in philippine_sources:
        try:
            print(f"Accessing: {source}")
            response = requests.get(source, headers=headers, timeout=10)
            if response.status_code == 200:
                print(f"Successfully accessed: {source}")
                access_results[source] = True
                
                # Parse content for relevant information
                soup = BeautifulSoup(response.text, 'html.parser')
                title = soup.title.get_text() if soup.title else "No title"
                print(f"Page title: {title}")
                
                # Look for search functionality or archives
                search_forms = soup.find_all('form')
                print(f"Found {len(search_forms)} search forms")
                
            else:
                print(f"Access failed for {source}: Status {response.status_code}")
                access_results[source] = False
                
        except requests.exceptions.RequestException as e:
            print(f"Error accessing {source}: {str(e)}")
            access_results[source] = False
        
        time.sleep(1)  # Respectful delay between requests
    
    return access_results
