                start = pos + 1

            year_contexts = 0

                print(f'\n  {i}. Mention at position {pos}:')

                if has_governor:

            results['contexts_with_years'] = year_contexts

                results['confidence'] = 'Medium'

                results['confidence'] = 'Low-Medium'
