# Function to search using DuckDuckGo (which allows programmatic access)
def search_web_content(query, max_results=5):
    """Search for content using web search and extract relevant information"""
    search_results = []
    
    # Use DuckDuckGo HTML search (more accessible than Google for automated queries)
    search_url = f'https://html.duckduckgo.com/html/?q={query.replace(" ", "+")}'
    
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
    }
    
    try:
        print(f'üîç Searching for: "{query}"')
        response = requests.get(search_url, headers=headers, timeout=30)
        
        if response.status_code == 200:
            print(f'‚úÖ Search successful (Status: {response.status_code})')
            
            # Save search results for analysis
            search_filename = f'workspace/search_results_{query.replace(" ", "_").replace(",", "")[:50]}.html'
            with open(search_filename, 'w', encoding='utf-8') as f:
                f.write(response.text)
            print(f'üíæ Search results saved to: {search_filename}')
            
            # Basic analysis of search results
            content_lower = response.text.lower()
            zamboanga_mentions = content_lower.count('zamboanga del sur')
            governor_mentions = content_lower.count('governor')
            election_mentions = content_lower.count('election')
            
            # Look for potential governor names (common Filipino surnames)
            potential_names = ['antonio', 'jose', 'maria', 'juan', 'pedro', 'miguel', 'francisco', 'luis', 'carlos', 'manuel']
            name_mentions = {}
            for name in potential_names:
                count = content_lower.count(name)
                if count > 0:
                    name_mentions[name] = count
            
            search_results.append({
                'query': query,
                'file': search_filename,
                'zamboanga_mentions': zamboanga_mentions,
                'governor_mentions': governor_mentions,
                'election_mentions': election_mentions,
                'potential_names': name_mentions,
                'content_length': len(response.text)
            })
            
            print(f'üìä Search analysis:')
            print(f'   - Zamboanga del Sur mentions: {zamboanga_mentions}')
            print(f'   - Governor mentions: {governor_mentions}')
            print(f'   - Election mentions: {election_mentions}')
            if name_mentions:
                print(f'   - Potential names found: {list(name_mentions.keys())}')
            
        else:
            print(f'‚ùå Search failed (Status: {response.status_code})')
            
    except Exception as e:
        print(f'‚ùå Search error: {str(e)}')
    
    return search_results
