import requests
from bs4 import BeautifulSoup
import json
from datetime import datetime
import time
import re

# First, let's properly inspect the final heritage findings file to understand current status
final_results_file = 'workspace/china_stone_carving_final_heritage_findings.json'

print("Inspecting final heritage findings file...")
print("=" * 60)

try:
    with open(final_results_file, 'r', encoding='utf-8') as f:
        final_data = json.load(f)
    
    print("Final heritage findings file structure:")
    for key in final_data.keys():
        print(f"- {key}: {type(final_data[key])}")
    
    print(f"\nResearch timestamp: {final_data.get('research_timestamp', 'N/A')}")
    print(f"Total heritage sites researched: {final_data.get('total_heritage_sites_researched', 'N/A')}")
    
    if 'dazu_rock_carvings_findings' in final_data:
        dazu_status = "Found" if final_data['dazu_rock_carvings_findings'] else "Not found"
        print(f"Dazu Rock Carvings findings: {dazu_status}")
    
    if 'additional_heritage_sites' in final_data:
        print(f"Additional heritage sites: {len(final_data['additional_heritage_sites'])}")
        for site in final_data['additional_heritage_sites']:
            print(f"  - {site.get('name', 'Unknown')}: {site.get('region', 'Unknown region')}")
    
except FileNotFoundError:
    print("Final heritage findings file not found, will conduct fresh research...")
    final_data = {}
except json.JSONDecodeError as e:
    print(f"Error reading JSON file: {e}")
    final_data = {}

print("\n" + "=" * 90)
print("CONDUCTING FIXED COMPREHENSIVE STONE CARVING HERITAGE RESEARCH")
print("=" * 90)

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.5',
    'Accept-Encoding': 'gzip, deflate',
    'Connection': 'keep-alive',
    'Upgrade-Insecure-Requests': '1',
}

# Define comprehensive stone carving heritage sites to research
heritage_sites = [
    {
        'name': 'Dazu Rock Carvings',
        'url': 'https://en.wikipedia.org/wiki/Dazu_Rock_Carvings',
        'region': 'Dazu District, Chongqing',
        'type': 'Ancient Buddhist rock carvings and stone sculptures',
        'expected_status': 'UNESCO World Heritage Site'
    },
    {
        'name': 'Longmen Grottoes',
        'url': 'https://en.wikipedia.org/wiki/Longmen_Grottoes',
        'region': 'Luoyang, Henan Province',
        'type': 'Buddhist cave temples and stone carvings',
        'expected_status': 'UNESCO World Heritage Site'
    },
    {
        'name': 'Yungang Grottoes',
        'url': 'https://en.wikipedia.org/wiki/Yungang_Grottoes',
        'region': 'Datong, Shanxi Province',
        'type': 'Ancient Buddhist cave complexes with stone sculptures',
        'expected_status': 'UNESCO World Heritage Site'
    },
    {
        'name': 'Mogao Caves',
        'url': 'https://en.wikipedia.org/wiki/Mogao_Caves',
        'region': 'Dunhuang, Gansu Province',
        'type': 'Buddhist art and stone carvings',
        'expected_status': 'UNESCO World Heritage Site'
    }
]

# Define search terms
heritage_indicators = [
    'unesco world heritage', 'world heritage site', 'unesco',
    'intangible cultural heritage', 'cultural heritage',
    'heritage list', 'world heritage list', 'heritage status',
    'world heritage committee', 'heritage designation'
]

stone_carving_terms = [
    'stone carving', 'rock carving', 'stone sculpture',
    'carving technique', 'sculptural art', 'stone art',
    'rock art', 'carved stone', 'stone relief'
]

# Store comprehensive findings
comprehensive_heritage_findings = []

print("\n🎯 SYSTEMATIC HERITAGE SITE RESEARCH WITH FIXED ERROR HANDLING:")
print("=" * 70)

for i, site in enumerate(heritage_sites, 1):
    print(f"\n🏛️ Site {i}/{len(heritage_sites)}: {site['name']}")
    print(f"Region: {site['region']}")
    print(f"Type: {site['type']}")
    print(f"Expected Status: {site['expected_status']}")
    print(f"URL: {site['url']}")
    print("-" * 70)
    
    try:
        print(f"Accessing {site['name']}...")
        response = requests.get(site['url'], headers=headers, timeout=30)
        
        if response.status_code == 200:
            print(f"✅ Successfully accessed (Status: {response.status_code})")
            
            # Parse HTML content
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Get page title
            title = soup.find('title')
            page_title = title.get_text().strip() if title else 'No title'
            print(f"Page title: {page_title}")
            
            # Remove script and style elements
            for element in soup(["script", "style"]):
                element.decompose()
            
            # Get clean text
            text = soup.get_text()
            lines = (line.strip() for line in text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
            clean_text = ' '.join(chunk for chunk in chunks if chunk)
            
            print(f"Content length: {len(clean_text)} characters")
            
            # Analyze content for heritage and stone carving information
            text_lower = clean_text.lower()
            
            # Find heritage indicators
            found_heritage_indicators = [indicator for indicator in heritage_indicators if indicator in text_lower]
            found_stone_terms = [term for term in stone_carving_terms if term in text_lower]
            
            print(f"🏆 Heritage indicators found: {found_heritage_indicators}")
            print(f"🗿 Stone carving terms found: {found_stone_terms}")
            
            # Extract relevant sentences (FIXED ERROR HANDLING)
            if found_heritage_indicators and found_stone_terms:
                print("\n🎯 HIGHLY RELEVANT HERITAGE SITE CONFIRMED!")
                
                # Split text into sentences
                sentences = re.split(r'[.!?]+', clean_text)
                unesco_sentences = []
                heritage_sentences = []
                stone_carving_sentences = []
                
                for sentence in sentences:
                    sentence_clean = sentence.strip()
                    if len(sentence_clean) > 20:  # Avoid very short fragments
                        sentence_lower = sentence_clean.lower()  # FIXED: consistent variable naming
                        
                        # UNESCO-specific sentences
                        if 'unesco' in sentence_lower and ('world heritage' in sentence_lower or 'heritage site' in sentence_lower):
                            unesco_sentences.append(sentence_clean)
                        
                        # Heritage-related sentences
                        elif any(indicator in sentence_lower for indicator in heritage_indicators):
                            heritage_sentences.append(sentence_clean)
                        
                        # Stone carving technique sentences
                        elif any(term in sentence_lower for term in stone_carving_terms):
                            stone_carving_sentences.append(sentence_clean)
                
                print(f"\n📝 UNESCO SENTENCES EXTRACTED ({len(unesco_sentences)} found):")
                for j, sentence in enumerate(unesco_sentences[:3], 1):
                    print(f"{j}. {sentence}")
                
                print(f"\n🏛️ HERITAGE SENTENCES EXTRACTED ({len(heritage_sentences)} found):")
                for j, sentence in enumerate(heritage_sentences[:2], 1):
                    print(f"{j}. {sentence}")
                
                print(f"\n🗿 STONE CARVING SENTENCES EXTRACTED ({len(stone_carving_sentences)} found):")
                for j, sentence in enumerate(stone_carving_sentences[:2], 1):
                    print(f"{j}. {sentence}")
                
                # Determine heritage status
                unesco_confirmed = 'unesco world heritage' in text_lower or 'world heritage site' in text_lower
                heritage_status = 'UNESCO World Heritage Site' if unesco_confirmed else 'Cultural Heritage Site'
                
                print(f"\n✅ HERITAGE STATUS CONFIRMED: {heritage_status}")
                
                # Store comprehensive finding
                site_finding = {
                    'name': site['name'],
                    'region': site['region'],
                    'type': site['type'],
                    'heritage_status': heritage_status,
                    'unesco_confirmed': unesco_confirmed,
                    'heritage_indicators_found': found_heritage_indicators,
                    'stone_carving_terms_found': found_stone_terms,
                    'unesco_sentences': unesco_sentences[:3],
                    'heritage_sentences': heritage_sentences[:2],
                    'stone_carving_sentences': stone_carving_sentences[:2],
                    'page_title': page_title,
                    'source_url': site['url'],
                    'content_length': len(clean_text),
                    'relevance_score': len(found_heritage_indicators) + len(found_stone_terms),
                    'research_timestamp': datetime.now().isoformat()
                }
                
                comprehensive_heritage_findings.append(site_finding)
                
            elif found_heritage_indicators or found_stone_terms:
                print("\nℹ️ Partially relevant content found")
                print(f"   Heritage indicators: {found_heritage_indicators}")
                print(f"   Stone carving terms: {found_stone_terms}")
            else:
                print("\nℹ️ Limited relevance to stone carving heritage")
                
        else:
            print(f"❌ Failed to access (Status: {response.status_code})")
            
    except Exception as e:
        print(f"❌ Error researching {site['name']}: {str(e)}")
    
    print("\n" + "=" * 90)
    time.sleep(2)  # Respectful delay between requests

print(f"\n\n📊 FINAL COMPREHENSIVE RESEARCH SUMMARY:")
