```
import os
import json
from bs4 import BeautifulSoup
import re

print('=== TARGETED ANALYSIS FOR 2009 MARTINEAU-ATKINSON PUBLISHER ===') 
print('Objective: Extract 2009 reissue publisher from workspace files')
print('Book: Letters on the Laws of Man\'s Nature and Development')
print('Authors: Harriet Martineau and Henry George Atkinson')
print('\n' + '='*100 + '\n')

# Check workspace directory
if not os.path.exists('workspace'):
    print('❌ Workspace directory does not exist')
    exit()

workspace_files = os.listdir('workspace')
print(f'Total files in workspace: {len(workspace_files)}')

# Find files with relevant keywords using proper loop structure
relevant_files = []
for file in workspace_files:
    file_lower = file.lower()  # Define within loop scope
    if any(term in file_lower for term in ['martineau', 'atkinson', '2009', 'publisher']):
        relevant_files.append(file)

print(f'\nRelevant files found: {len(relevant_files)}')
for file in relevant_files[:10]:  # Show first 10
    file_path = os.path.join('workspace', file)
    file_size = os.path.getsize(file_path)
    print(f'  - {file} ({file_size:,} bytes)')

print('\n=== STEP 1: ANALYZING HTML FILES FOR 2009 PUBLISHER INFORMATION ===') 

# Focus on HTML files that might contain search results
html_files = [f for f in relevant_files if f.endswith('.html')]
print(f'\nHTML files to analyze: {len(html_files)}')

publisher_findings = []

# Analyze HTML files for 2009 publisher information
for html_file in html_files[:10]:  # Analyze first 10 HTML files
    print(f'\n--- Analyzing: {html_file} ---')
    
    try:
        file_path = os.path.join('workspace', html_file)
        
        with open(file_path, 'r', encoding='utf-8') as f:
            html_content = f.read()
        
        # Parse HTML and extract text
        soup = BeautifulSoup(html_content, 'html.parser')
        
        # Remove script and style elements
        for element in soup(['script', 'style']):
            element.decompose()
        
        text_content = soup.get_text()
        text_lower = text_content.lower()
        
        # Check relevance
        has_2009 = '2009' in text_lower
        has_martineau = 'martineau' in text_lower
        has_publisher = any(term in text_lower for term in ['publisher', 'published', 'press', 'publishing'])
        
        relevance_score = sum([has_2009, has_martineau, has_publisher])
        print(f'Relevance: {relevance_score}/3 (2009={has_2009}, Martineau={has_martineau}, Publisher={has_publisher})')
        
        if relevance_score >= 2 and has_2009:  # Must have 2009 and at least one other term
            print('✓ High relevance with 2009 - analyzing for publishers')
            
            # Look for text around 2009 mentions
            positions_2009 = []
            start = 0
            while True:
                pos = text_lower.find('2009', start)
                if pos == -1:
                    break
                positions_2009.append(pos)
                start = pos + 1
            
            print(f'  Found {len(positions_2009)} instances of "2009"')
            
            # Extract context around each 2009 mention
            for i, pos in enumerate(positions_2009[:5], 1):  # Check first 5 instances
                context_start = max(0, pos - 300)
                context_end = min(len(text_content), pos + 400)
                context = text_content[context_start:context_end]
                context_lower = context.lower()
                
                # Check if this context contains publisher information
                publisher_terms = ['publisher', 'published', 'press', 'publishing', 'books', 'reprint', 'reissue']
                has_pub_terms = any(term in context_lower for term in publisher_terms)
                
                if has_pub_terms:
                    print(f'  Context {i} around 2009 (contains publisher terms):')
                    
                    # Clean up the context for better readability
                    clean_context = ' '.join(context.split())
                    print(f'    {clean_context[:250]}...')
                    
                    # Look for specific publisher names in this context
                    known_publishers = [
                        'Nabu Press', 'Kessinger Publishing', 'Forgotten Books', 'BiblioLife',
                        'Palala Press', 'Wentworth Press', 'Franklin Classics', 'Elibron Classics',
                        'Gale ECCO', 'Making of Modern Law', 'CreateSpace', 'Lightning Source',
                        'Cambridge University Press', 'Oxford University Press', 'Harvard University Press',
                        'Yale University Press', 'Princeton University Press', 'Routledge'
                    ]
                    
                    found_publishers = []
                    for publisher in known_publishers:
                        if publisher.lower() in context_lower:
                            found_publishers.append(publisher)
                    
                    if found_publishers:
                        print(f'    🎯 Publishers found in 2009 context: {found_publishers}')
                        
                        for publisher in found_publishers:
                            publisher_findings.append({
                                'file': html_file,
                                'publisher': publisher,
                                'context': clean_context,
                                'context_type': '2009_vicinity',
                                'relevance': 'high'
                            })
                    
                    # Also look for publisher patterns using regex
                    publisher_patterns = [
                        r'published by ([A-Z][^\n,]{5,50})',
                        r'publisher[:\s]+([A-Z][^\n,]{5,50})',
                        r'([A-Z][a-z]+ (?:Press|Publishing|Books))',
                        r'reprinted by ([A-Z][^\n,]{5,50})'
                    ]
                    
                    for pattern in publisher_patterns:
                        matches = re.findall(pattern, context, re.IGNORECASE)
                        for match in matches:
                            if isinstance(match, tuple):
                                match = match[0] if match[0] else match[1] if len(match) > 1 else ''
                            
                            match = match.strip()
                            if len(match) > 5 and match not in [f['publisher'] for f in publisher_findings]:
                                print(f'    📚 Pattern match: {match}')
                                publisher_findings.append({
                                    'file': html_file,
                                    'publisher': match,
                                    'context': clean_context,
                                    'context_type': '2009_pattern_match',
                                    'relevance': 'medium'
                                })
        else:
            print('- Low relevance or no 2009 mentions')
            
    except Exception as e:
        print(f'❌ Error analyzing {html_file}: {str(e)}')

print('\n=== STEP 2: ANALYZING LARGE HTML FILES FOR DETAILED CONTENT ===') 

# Focus on the largest HTML files which might contain more comprehensive search results
html_with_sizes = []
for html_file in html_files:
    file_path = os.path.join('workspace', html_file)
    file_size = os.path.getsize(file_path)
    html_with_sizes.append((html_file, file_size))

# Sort by size (largest first)
html_with_sizes.sort(key=lambda x: x[1], reverse=True)

print('\nLargest HTML files (most likely to contain comprehensive results):')
for i, (file, size) in enumerate(html_with_sizes[:5], 1):
    print(f'  {i}. {file} ({size:,} bytes)')

# Analyze the largest files more thoroughly
for html_file, file_size in html_with_sizes[:3]:  # Top 3 largest files
    if file_size > 200000:  # Only files larger than 200KB
        print(f'\n--- DETAILED ANALYSIS: {html_file} ({file_size:,} bytes) ---')
        
        try:
            file_path = os.path.join('workspace', html_file)
            
            with open(file_path, 'r', encoding='utf-8') as f:
                html_content = f.read()
            
            # Parse HTML
            soup = BeautifulSoup(html_content, 'html.parser')
            
            # Look for specific elements that might contain book information
            # Common selectors for search results
            result_selectors = [
                '.result', '.search-result', '.item', '.record', '.citation',
                '.book', '.publication', '.entry', '.listing', '.product'
            ]
            
            found_results = []
            for selector in result_selectors:
                elements = soup.select(selector)
                if elements:
                    print(f'  Found {len(elements)} elements with selector: {selector}')
                    found_results.extend(elements[:10])  # Take first 10 of each type
            
            if found_results:
                print(f'  Analyzing {len(found_results)} result elements for 2009 publisher info...')
                
                for i, element in enumerate(found_results[:20], 1):  # Check first 20 results
                    element_text = element.get_text().strip()
                    element_lower = element_text.lower()
                    
                    # Check if this result mentions 2009 and publisher terms
                    if ('2009' in element_lower and 
                        any(term in element_lower for term in ['martineau', 'atkinson']) and
                        any(term in element_lower for term in ['publisher', 'published', 'press'])):
                        
                        print(f'\n    🎯 RELEVANT RESULT {i}:')
                        print(f'    {element_text[:300]}...')
                        
                        # Extract publisher information from this result
                        publisher_patterns = [
                            r'publisher[:\s]*([^\n,]{10,60})',
                            r'published by ([^\n,]{10,60})',
                            r'([A-Z][a-z]+ (?:Press|Publishing|Books))',
                            r'(\b(?:Nabu|Kessinger|Forgotten|BiblioLife|Palala|Wentworth|Franklin)\b[^\n]{0,30})'
                        ]
                        
                        for pattern in publisher_patterns:
                            matches = re.findall(pattern, element_text, re.IGNORECASE)
                            for match in matches:
                                if isinstance(match, tuple):
                                    match = match[0] if match[0] else ''
                                
                                match = match.strip()
                                if len(match) > 5:
                                    print(f'      📚 Publisher extracted: {match}')
                                    publisher_findings.append({
                                        'file': html_file,
                                        'publisher': match,
                                        'context': element_text[:400],
                                        'context_type': 'search_result_element',
                                        'relevance': 'very_high'
                                    })
            else:
                print('  No structured result elements found, searching full text...')
                
                # If no structured results, search full text for 2009 + Martineau + publisher
                full_text = soup.get_text()
                lines = full_text.split('\n')
                
                relevant_lines = []
                for line in lines:
                    line_lower = line.lower().strip()
                    if (len(line.strip()) > 50 and  # Substantial content
                        '2009' in line_lower and
                        any(author in line_lower for author in ['martineau', 'atkinson']) and
                        any(term in line_lower for term in ['publisher', 'published', 'press'])):
                        relevant_lines.append(line.strip())
                
                if relevant_lines:
                    print(f'  Found {len(relevant_lines)} relevant lines with 2009 + authors + publisher')
                    for i, line in enumerate(relevant_lines[:3], 1):
                        print(f'    {i}. {line[:200]}...')
                        
                        # Extract publishers from these lines
                        for publisher in ['Nabu Press', 'Kessinger Publishing', 'Forgotten Books', 'BiblioLife', 'Palala Press']:
                            if publisher.lower() in line.lower():
                                print(f'      🎯 FOUND: {publisher} in line {i}')
                                publisher_findings.append({
                                    'file': html_file,
                                    'publisher': publisher,
                                    'context': line,
                                    'context_type': 'full_text_line',
                                    'relevance': 'high'
                                })
                
        except Exception as e:
            print(f'❌ Error in detailed analysis of {html_file}: {str(e)}')

print('\n=== STEP 3: CONSOLIDATING AND RANKING PUBLISHER FINDINGS ===') 

print(f'Total publisher findings: {len(publisher_findings)}')

if publisher_findings:
    # Group by publisher and calculate scores
    publisher_scores = {}
    
    for finding in publisher_findings:
        publisher = finding['publisher']
        relevance = finding['relevance']
        
        if publisher not in publisher_scores:
            publisher_scores[publisher] = {
                'total_mentions': 0,
                'high_relevance': 0,
                'very_high_relevance': 0,
                'findings': []
            }
        
        publisher_scores[publisher]['total_mentions'] += 1
        publisher_scores[publisher]['findings'].append(finding)
        
        if relevance == 'high':
            publisher_scores[publisher]['high_relevance'] += 1
        elif relevance == 'very_high':
            publisher_scores[publisher]['very_high_relevance'] += 1
    
    # Calculate final scores (weight very_high > high > medium)
    scored_publishers = []
    for publisher, data in publisher_scores.items():
        score = (data['very_high_relevance'] * 10 + 
                data['high_relevance'] * 5 + 
                data['total_mentions'])
        
        scored_publishers.append({
            'publisher': publisher,
            'score': score,
            'data': data
        })
    
    # Sort by score (highest first)
    scored_publishers.sort(key=lambda x: x['score'], reverse=True)
    
    print('\n🏆 PUBLISHER RANKING BY RELEVANCE:')
    for i, result in enumerate(scored_publishers, 1):
        publisher = result['publisher']
        score = result['score']
        data = result['data']
        
        print(f'{i}. {publisher} (Score: {score})')
        print(f'   Total mentions: {data["total_mentions"]}')
        print(f'   High relevance: {data["high_relevance"]}, Very high: {data["very_high_relevance"]}')
        
        # Show evidence for top 3 publishers
        if i <= 3:
            print('   Top evidence:')
            for j, finding in enumerate(data['findings'][:2], 1):
                context_preview = finding['context'][:120].replace('\n', ' ')
                print(f'     {j}. {finding["file"]} - {context_preview}...')
        print()
    
    if scored_publishers:
        top_publisher = scored_publishers[0]['publisher']
        top_score = scored_publishers[0]['score']
        
        print(f'🎯 MOST LIKELY 2009 PUBLISHER: {top_publisher}')
        print(f'Confidence score: {top_score}')
        
        # Check if it's a known reprint publisher
        specialty_publishers = ['Nabu Press', 'Kessinger Publishing', 'Forgotten Books', 'BiblioLife', 'Palala Press', 'Wentworth Press', 'Franklin Classics']
        
        if top_publisher in specialty_publishers:
            print(f'✓ CONFIRMED: {top_publisher} is a known specialty reprint publisher')
        
        # Save final answer
        final_answer = {
            'question': 'What publisher reissued "Letters on the Laws of Man\'s Nature and Development" by Harriet Martineau and Henry George Atkinson in 2009?',
            'answer': top_publisher,
            'confidence_score': top_score,
            'evidence_summary': {
                'total_mentions': scored_publishers[0]['data']['total_mentions'],
                'high_relevance_mentions': scored_publishers[0]['data']['high_relevance'],
                'very_high_relevance_mentions': scored_publishers[0]['data']['very_high_relevance'],
                'source_files': list(set(f['file'] for f in scored_publishers[0]['data']['findings'])),
                'publisher_type': 'specialty_reprint' if top_publisher in specialty_publishers else 'academic_commercial'
            },
            'all_evidence': scored_publishers[0]['data']['findings'],
            'analysis_timestamp': '2025-01-21 16:30:00'
        }
        
        answer_file = 'workspace/final_2009_publisher_identification.json'
        with open(answer_file, 'w', encoding='utf-8') as f:
            json.dump(final_answer, f, indent=2, ensure_ascii=False)
        
        print(f'\n✓ Final answer saved to: {answer_file}')
        
else:
    print('\n⚠ No publisher findings with 2009 context found')
    print('The 2009 publisher information may not be clearly present in the HTML search results')

print('\n*** TARGETED ANALYSIS COMPLETE ***')
print('✓ Analyzed HTML files for 2009 publisher information')
print('✓ Focused on content around 2009 mentions')
print('✓ Extracted and ranked publishers by relevance')

if 'top_publisher' in locals():
    print(f'\n🏆 FINAL CONCLUSION: The 2009 reissue publisher is {top_publisher}')
else:
    print('\n⚠ CONCLUSION: Unable to identify 2009 publisher from available search results')
```
