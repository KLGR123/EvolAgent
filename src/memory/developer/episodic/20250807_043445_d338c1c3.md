Search for keywords including 'China national intangible cultural heritage stone carving', 'Chinese stone carving techniques origin', 'traditional stone carving China UNESCO', and 'intangible cultural heritage stone sculpture China'. Focus on identifying the specific city, region, or province where these traditional stone carving techniques first developed and are recognized as originating from.

- Automated UNESCO nomination dossier assembly and heritage status compliance reporting for government cultural heritage departments
- Comparative art history research and statistical analysis of regional stone carving methodologies for academic publications in archaeology and anthropology
- AI-driven provenance verification and authenticity scoring of Chinese stone sculptures for international auction houses and private collectors
- Business intelligence dashboards and origin-based marketing strategies for stone carving exporters leveraging intangible heritage data
- Digital archiving and semantic search implementation for intangible cultural heritage platforms preserving China’s traditional stone carving knowledge

```
import requests
from bs4 import BeautifulSoup
import json
from datetime import datetime
import time
import re

# First, let's inspect the existing research file to understand what we've found so far
research_file = 'workspace/china_stone_carving_direct_research.json'

print("Inspecting previous research results...")
print("=" * 60)

try:
    with open(research_file, 'r', encoding='utf-8') as f:
        previous_research = json.load(f)
    
    print("Previous research file structure:")
    print(f"Keys: {list(previous_research.keys())}")
    print(f"Research timestamp: {previous_research.get('research_timestamp', 'N/A')}")
    print(f"Sources investigated: {previous_research.get('sources_investigated', 'N/A')}")
    print(f"Relevant findings: {previous_research.get('relevant_findings', 'N/A')}")
    print(f"Regional findings: {len(previous_research.get('regional_findings', []))}")
    
    if 'regional_findings' in previous_research:
        print("\nRegional findings from previous research:")
        for finding in previous_research['regional_findings']:
            print(f"- {finding.get('name', 'Unknown')}: {finding.get('region', 'Unknown region')}")
    
except FileNotFoundError:
    print("Previous research file not found, starting fresh...")
    previous_research = {}

print("\n" + "=" * 90)
print("TARGETING SPECIFIC CHINESE CULTURAL HERITAGE DATABASES")
print("=" * 90)

# Target more specific Chinese heritage and cultural websites
specific_sources = [
    {
        'name': 'China National Committee for UNESCO',
        'url': 'https://en.unesco.org/countries/china',
        'description': 'UNESCO China country page with heritage listings'
    },
    {
        'name': 'China Intangible Cultural Heritage Network',
        'url': 'http://www.ihchina.cn/en/',
        'description': 'Official Chinese intangible heritage database'
    },
    {
        'name': 'Chinese Government Heritage Portal',
        'url': 'http://www.gov.cn/english/',
        'description': 'Chinese government official portal'
    },
    {
        'name': 'Cultural China - Traditional Arts',
        'url': 'http://traditions.cultural-china.com/',
        'description': 'Cultural China traditional arts database'
    }
]

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.5',
    'Accept-Encoding': 'gzip, deflate',
    'Connection': 'keep-alive',
    'Upgrade-Insecure-Requests': '1',
}

# Store detailed research results
detailed_results = []
stone_carving_heritage_findings = []

print("\n🎯 CONDUCTING TARGETED HERITAGE DATABASE RESEARCH:")
print("=" * 60)

for i, source in enumerate(specific_sources, 1):
    print(f"\nSource {i}: {source['name']}")
    print(f"URL: {source['url']}")
    print(f"Focus: {source['description']}")
    print("-" * 70)
    
    try:
        print(f"Accessing {source['name']}...")
        response = requests.get(source['url'], headers=headers, timeout=30)
        
        if response.status_code == 200:
            print(f"✅ Successfully accessed (Status: {response.status_code})")
            
            # Parse HTML content
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Get page title
            title = soup.find('title')
            page_title = title.get_text().strip() if title else 'No title'
            print(f"Page title: {page_title}")
            
            # Remove script and style elements
            for element in soup(["script", "style"]):
                element.decompose()
            
            # Get clean text
            text = soup.get_text()
            lines = (line.strip() for line in text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
            clean_text = ' '.join(chunk for chunk in chunks if chunk)
            
            print(f"Content length: {len(clean_text)} characters")
            
            # Enhanced keyword search for stone carving heritage
            stone_carving_keywords = [
                'stone carving', 'stone sculpture', 'rock carving', 'stone craft',
                'sculptural art', 'carving technique', 'stone art', 'traditional carving'
            ]
            
            heritage_keywords = [
                'intangible cultural heritage', 'UNESCO', 'national heritage',
                'cultural heritage', 'traditional craft', 'heritage list',
                'intangible heritage', 'cultural protection'
            ]
            
            chinese_regions = [
                'quyang', 'hebei', 'shoushan', 'fujian', 'dazu', 'chongqing',
                'shandong', 'zhejiang', 'jiangsu', 'guangdong', 'yunnan',
                'beijing', 'shanghai', 'anhui', 'hunan', 'hubei'
            ]
            
            text_lower = clean_text.lower()
            
            # Find matches
            stone_matches = [kw for kw in stone_carving_keywords if kw in text_lower]
            heritage_matches = [kw for kw in heritage_keywords if kw in text_lower]
            region_matches = [region for region in chinese_regions if region in text_lower]
            
            print(f"Stone carving keywords found: {stone_matches}")
            print(f"Heritage keywords found: {heritage_matches}")
            print(f"Chinese regions mentioned: {region_matches}")
            
            # If we found relevant content, extract and analyze
            if stone_matches and heritage_matches:
                print("🎯 HIGHLY RELEVANT CONTENT FOUND!")
                
                # Extract sentences containing stone carving and heritage terms
                sentences = re.split(r'[.!?]+', clean_text)
                relevant_sentences = []
                
                for sentence in sentences:
                    sentence_lower = sentence.lower()
                    if (any(stone_kw in sentence_lower for stone_kw in stone_carving_keywords) and
                        any(heritage_kw in sentence_lower for heritage_kw in heritage_keywords)):
                        relevant_sentences.append(sentence.strip())
                
                print(f"\n📝 RELEVANT SENTENCES EXTRACTED ({len(relevant_sentences)} found):")
                for j, sentence in enumerate(relevant_sentences[:5], 1):  # Show first 5
                    print(f"{j}. {sentence}")
                
                if len(relevant_sentences) > 5:
                    print(f"... and {len(relevant_sentences) - 5} more relevant sentences")
                
                # Store detailed finding
                stone_carving_heritage_findings.append({
                    'source_name': source['name'],
                    'source_url': source['url'],
                    'page_title': page_title,
                    'stone_keywords_found': stone_matches,
                    'heritage_keywords_found': heritage_matches,
                    'regions_mentioned': region_matches,
                    'relevant_sentences': relevant_sentences,
                    'content_length': len(clean_text),
                    'relevance_score': len(stone_matches) + len(heritage_matches) + len(region_matches)
                })
                
            elif stone_matches or heritage_matches:
                print("ℹ️ Partially relevant content found")
                if len(stone_matches) > 0:
                    print(f"   Stone carving content: {stone_matches}")
                if len(heritage_matches) > 0:
                    print(f"   Heritage content: {heritage_matches}")
            else:
                print("ℹ️ No directly relevant stone carving heritage content found")
            
            # Store general result
            detailed_results.append({
                'source_name': source['name'],
                'source_url': source['url'],
                'status_code': response.status_code,
                'page_title': page_title,
                'content_length': len(clean_text),
                'stone_keywords_found': stone_matches,
                'heritage_keywords_found': heritage_matches,
                'regions_mentioned': region_matches,
                'is_highly_relevant': bool(stone_matches and heritage_matches),
                'is_partially_relevant': bool(stone_matches or heritage_matches)
            })
            
        else:
            print(f"❌ Failed to access (Status: {response.status_code})")
            detailed_results.append({
                'source_name': source['name'],
                'source_url': source['url'],
                'status_code': response.status_code,
                'error': f'HTTP {response.status_code}',
                'is_highly_relevant': False,
                'is_partially_relevant': False
            })
            
    except Exception as e:
        print(f"❌ Error accessing {source['name']}: {str(e)}")
        detailed_results.append({
            'source_name': source['name'],
            'source_url': source['url'],
            'error': str(e),
            'is_highly_relevant': False,
            'is_partially_relevant': False
        })
    
    print("\n" + "=" * 90)
    time.sleep(2)  # Respectful delay

# Now let's research the specific regions we identified earlier with more targeted searches
print("\n🏛️ DETAILED RESEARCH ON IDENTIFIED STONE CARVING REGIONS:")
print("=" * 70)

# Research specific heritage sites and techniques
regional_heritage_sites = [
    {
        'name': 'Quyang Stone Carving Heritage',
        'region': 'Quyang County, Hebei Province',
        'search_url': 'https://en.wikipedia.org/wiki/Quyang_County',
        'heritage_type': 'Traditional stone carving techniques',
        'significance': 'Known as hometown of Chinese stone carving'
    },
    {
        'name': 'Shoushan Stone Carving',
        'region': 'Fuzhou, Fujian Province', 
        'search_url': 'https://en.wikipedia.org/wiki/Shoushan_stone',
        'heritage_type': 'Seal carving and stone sculpture',
        'significance': 'Famous for Shoushan stone seal carving'
    },
    {
        'name': 'Dazu Rock Carvings',
        'region': 'Dazu District, Chongqing',
        'search_url': 'https://en.wikipedia.org/wiki/Dazu_Rock_Carvings',
        'heritage_type': 'Ancient Buddhist rock carvings',
        'significance': 'UNESCO World Heritage Site since 1999'
    }
]

regional_detailed_findings = []

for region_info in regional_heritage_sites:
    print(f"\n🏮 Researching: {region_info['name']}")
    print(f"Region: {region_info['region']}")
    print(f"Heritage Type: {region_info['heritage_type']}")
    print(f"URL: {region_info['search_url']}")
    print("-" * 60)
    
    try:
        print(f"Accessing detailed information...")
        response = requests.get(region_info['search_url'], headers=headers, timeout=30)
        
        if response.status_code == 200:
            print(f"✅ Successfully accessed (Status: {response.status_code})")
            
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Get page title
            title = soup.find('title')
            page_title = title.get_text().strip() if title else 'No title'
            
            # Remove script and style elements
            for element in soup(["script", "style"]):
                element.decompose()
            
            # Get clean text
            text = soup.get_text()
            lines = (line.strip() for line in text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
            clean_text = ' '.join(chunk for chunk in chunks if chunk)
            
            print(f"Content length: {len(clean_text)} characters")
            
            # Extract key information about this heritage site
            heritage_info = {
                'name': region_info['name'],
                'region': region_info['region'],
                'heritage_type': region_info['heritage_type'],
                'significance': region_info['significance'],
                'source_url': region_info['search_url'],
                'page_title': page_title,
                'content_length': len(clean_text),
                'research_timestamp': datetime.now().isoformat()
            }
            
            # Look for specific heritage status information
            text_lower = clean_text.lower()
            heritage_indicators = [
                'unesco', 'world heritage', 'intangible cultural heritage',
                'national heritage', 'cultural protection', 'traditional craft',
                'heritage list', 'cultural significance'
            ]
            
            found_indicators = [indicator for indicator in heritage_indicators if indicator in text_lower]
            heritage_info['heritage_indicators_found'] = found_indicators
            
            if found_indicators:
                print(f"🎯 Heritage status indicators found: {found_indicators}")
                
                # Extract relevant sentences about heritage status
                sentences = re.split(r'[.!?]+', clean_text)
                heritage_sentences = []
                
                for sentence in sentences:
                    sentence_lower = sentence.lower()
                    if any(indicator in sentence_lower for indicator in heritage_indicators):
                        heritage_sentences.append(sentence.strip())
                
                heritage_info['heritage_sentences'] = heritage_sentences[:3]  # Top 3 most relevant
                
                print(f"📝 Key heritage information extracted:")
                for j, sentence in enumerate(heritage_sentences[:3], 1):
                    print(f"{j}. {sentence}")
            else:
                print("ℹ️ No specific heritage status indicators found")
                heritage_info['heritage_sentences'] = []
            
            regional_detailed_findings.append(heritage_info)
            
        else:
            print(f"❌ Failed to access (Status: {response.status_code})")
            
    except Exception as e:
        print(f"❌ Error researching {region_info['name']}: {str(e)}")
    
    time.sleep(2)  # Respectful delay

# Save comprehensive research results
comprehensive_file = 'workspace/china_stone_carving_comprehensive_research.json'
with open(comprehensive_file, 'w', encoding='utf-8') as f:
    json.dump({
        'research_timestamp': datetime.now().isoformat(),

        'stone_carving_heritage_findings': stone_carving_heritage_findings,
        'regional_detailed_findings': regional_detailed_findings,

    }, f, indent=2, ensure_ascii=False)

print(f"\n\n📊 COMPREHENSIVE RESEARCH SUMMARY:")
print(f"Heritage databases investigated: {len(specific_sources)}")

# Provide current assessment based on research
print("\n\n🎯 CURRENT ASSESSMENT - CHINA'S STONE CARVING INTANGIBLE HERITAGE:")
print("=" * 80)

if len(stone_carving_heritage_findings) > 0:
    print("✅ SPECIFIC HERITAGE FINDINGS DISCOVERED:")
    for finding in stone_carving_heritage_findings:

for finding in regional_detailed_findings:
    heritage_status = "UNESCO/Heritage Status Found" if finding.get('heritage_indicators_found') else "Traditional Significance"
    print(f"\n🏮 {finding['name']} ({finding['region']})")
    print(f"   Type: {finding['heritage_type']}")
    print(f"   Status: {heritage_status}")
    print(f"   Significance: {finding['significance']}")

print(f"\nResearch continues to identify specific intangible cultural heritage designations...")
```
