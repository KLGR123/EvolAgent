import os
import json
from datetime import datetime
from ddgs import DDGS

print('=== HARRY SCHWARZ DEATH CAUSE INVESTIGATION ===')
print('Searching for information about Harry Schwarz\'s death and cause of death')
print('Target: Harry Schwarz - South African politician and ambassador')
print('Objective: Find cause of death and death details\n')

# Create workspace directory if it doesn't exist
workspace_dir = 'workspace'
if not os.path.exists(workspace_dir):
    os.makedirs(workspace_dir)
    print(f'✅ Created workspace directory: {workspace_dir}')
else:
    print(f'✅ Workspace directory exists: {workspace_dir}')

# Initialize search results storage
search_results = {
    'timestamp': datetime.now().isoformat(),
    'target_person': 'Harry Schwarz',
    'objective': 'Find cause of death and death details',
    'queries': [],
    'results': [],
    'total_results_found': 0
}

# Define targeted search queries for Harry Schwarz death cause
search_queries = [
    'Harry Schwarz death cause',
    'Harry Schwarz obituary cause of death',
    'Harry Schwarz South African politician died how',
    'Harry Schwarz ambassador France death details',
    'Harry Schwarz died 2010 cause',
    'Harry Schwarz South Africa death reason',
    'Harry Schwarz obituary 2010',
    'Harry Schwarz politician death circumstances'
]

print(f'📋 PLANNED SEARCHES ({len(search_queries)} queries):')
for i, query in enumerate(search_queries, 1):
    print(f'  {i}. "{query}"')

# Initialize DDGS searcher
searcher = DDGS(timeout=15)

print(f'\n🔍 EXECUTING SEARCHES:')
print('=' * 60)

# Execute each search query
for i, query in enumerate(search_queries, 1):
    print(f'\nSearch {i}/{len(search_queries)}: "{query}"')
    print('-' * 50)
    
    try:
        # Perform search with multiple backends
        results = searcher.text(
            query, 
            max_results=10, 
            page=1, 
            backend=["google", "duckduckgo", "bing", "yahoo"], 
            safesearch="off", 
            region="en-us"
        )
        
        if results:
            print(f'✅ Found {len(results)} results')
            
            # Store query and results
            query_data = {
                'query': query,
                'results_count': len(results),
                'results': results
            }
            search_results['queries'].append(query_data)
            search_results['results'].extend(results)
            search_results['total_results_found'] += len(results)
            
            # Show first few results
            for j, result in enumerate(results[:3], 1):
                title = result.get('title', 'No title')[:80]
                url = result.get('href', 'No URL')[:60]
                snippet = result.get('body', 'No snippet')[:150]
                
                print(f'  {j}. {title}...')
                print(f'     URL: {url}...')
                print(f'     Snippet: {snippet}...')
                print()
        else:
            print(f'❌ No results found for "{query}"')
            search_results['queries'].append({
                'query': query,
                'results_count': 0,
                'results': []
            })
            
    except Exception as e:
        print(f'❌ Error searching for "{query}": {str(e)}')
        search_results['queries'].append({
            'query': query,
            'results_count': 0,
            'error': str(e),
            'results': []
        })

print(f'\n📊 SEARCH SUMMARY:')
print(f'   • Total queries executed: {len(search_results["queries"])}')
print(f'   • Total results found: {search_results["total_results_found"]}')
print(f'   • Unique results collected: {len(search_results["results"])}')

# Save search results to JSON file
results_file = os.path.join(workspace_dir, 'harry_schwarz_death_search_results.json')
with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(search_results, f, indent=2, ensure_ascii=False)

print(f'\n💾 Search results saved to: {results_file}')

# Analyze results for Harry Schwarz death information
print(f'\n🔍 INITIAL ANALYSIS OF SEARCH RESULTS:')
print('=' * 60)

if search_results['total_results_found'] > 0:
    print('✅ Search results found - analyzing for Harry Schwarz death information')
    
    # Count mentions of key terms
    death_keywords = ['died', 'death', 'obituary', 'passed away', 'cause of death', 'illness', 'disease']
    harry_schwarz_keywords = ['harry schwarz', 'schwarz']
    
    relevant_results = []
    
    for result in search_results['results']:
        title = result.get('title', '').lower()
        snippet = result.get('body', '').lower()
        combined_text = f'{title} {snippet}'
        
        # Check if result mentions Harry Schwarz
        has_harry_schwarz = any(keyword in combined_text for keyword in harry_schwarz_keywords)
        
        # Check if result mentions death-related terms
        has_death_info = any(keyword in combined_text for keyword in death_keywords)
        
        if has_harry_schwarz and has_death_info:
            relevant_results.append({
                'title': result.get('title', ''),
                'url': result.get('href', ''),
                'snippet': result.get('body', ''),
                'relevance_score': sum(1 for keyword in death_keywords if keyword in combined_text)
            })
    
    print(f'📋 Relevant results found: {len(relevant_results)}')
    
    if relevant_results:
        # Sort by relevance score
        relevant_results.sort(key=lambda x: x['relevance_score'], reverse=True)
        
        print(f'\n🎯 TOP RELEVANT RESULTS:')
        for i, result in enumerate(relevant_results[:5], 1):
            print(f'\n{i}. TITLE: {result["title"]}')
            print(f'   URL: {result["url"]}')
            print(f'   SNIPPET: {result["snippet"][:200]}...')
            print(f'   RELEVANCE SCORE: {result["relevance_score"]}')
    else:
        print('❓ No results specifically mention both Harry Schwarz and death information')
        
        # Show general Harry Schwarz results
        general_results = []
        for result in search_results['results']:
            title = result.get('title', '').lower()
            snippet = result.get('body', '').lower()
            combined_text = f'{title} {snippet}'
            
            if any(keyword in combined_text for keyword in harry_schwarz_keywords):
                general_results.append(result)
        
        if general_results:
            print(f'\n📋 General Harry Schwarz results found: {len(general_results)}')
            for i, result in enumerate(general_results[:3], 1):
                print(f'\n{i}. {result.get("title", "No title")}')
                print(f'   {result.get("body", "No snippet")[:150]}...')
else:
    print('❌ No search results found')
    print('This could indicate:')
    print('• Network connectivity issues')
    print('• Search API limitations')
    print('• Very specific or rare information')

print(f'\n🎯 NEXT STEPS:')
if search_results['total_results_found'] > 0:
    print('1. Analyze search results in detail')
    print('2. Extract specific death cause information')
    print('3. Verify information from multiple sources')
    print('4. Look for biographical and obituary details')
else:
    print('1. Try alternative search approaches')
    print('2. Search for biographical information first')
    print('3. Look for South African political archives')
    print('4. Search for obituaries from 2010 timeframe')

print('\n=== HARRY SCHWARZ DEATH CAUSE SEARCH COMPLETE ===')
