                                'body': result.get('body', ''),
                                'url': url,
                                'relevance_score': combined_text.count('little monsters') + sum(1 for term in creator_terms if term in combined_text)
                            })

                    # Sort by relevance
                    promising_results.sort(key=lambda x: x['relevance_score'], reverse=True)
                    
                    print("\nTop promising results:")

                        print(f"\n  {i+1}. {result['title'][:60]}...")
                        print(f"     URL: {result['url'][:70]}...")
                        print(f"     Snippet: {result['body'][:100]}...")
                        print(f"     Relevance score: {result['relevance_score']}")
                        
                        # Manual analysis of the text for artist names
                        full_text = f"{result['title']} {result['body']}"
                        words = full_text.split()
                        
                        # Look for capitalized names that could be artists
                        potential_names = []
                        for j, word in enumerate(words):
                            if (len(word) > 2 and word[0].isupper() and word.isalpha() and
                                word not in ['Little', 'Monsters', 'Gold', 'Key', 'Comics', 'Comic', 'Book']):
                                
                                # Check if next word is also a name (surname)

                                    if (len(next_word) > 2 and next_word[0].isupper() and 
                                        next_word.isalpha() and
                                        next_word not in ['Little', 'Monsters', 'Gold', 'Key', 'Comics', 'Comic', 'Book']):
                                        
                                        full_name = f"{word} {next_word}"
                                        if full_name not in potential_names:
                                            potential_names.append(full_name)
                        
                        if potential_names:
                            print(f"     Potential artist names: {potential_names[:3]}")
