def safe_get(data, *keys, default='N/A'):
    """Safely extract nested dictionary values with fallback"""
    try:
        result = data
        for key in keys:
            if isinstance(result, dict) and key in result:
                result = result[key]
            else:
                return default
        return result if result is not None else default
    except:
        return default
