import os
import json
import requests
from urllib.parse import quote
import re

print("=== THE FROGS BAND MEMBERS RESEARCH ===\n")
print("Objective: Find which Frogs member filled in as keyboardist for The_Family_(band)")
print("Focus: Jimmy Flemion, Dennis Flemion, and their musical collaborations\n")

# First, let's search for basic information about The Frogs band members
search_queries = [
    "The Frogs band Jimmy Flemion Dennis Flemion members biography",
    "The Frogs band keyboardist Prince The Family band collaboration",
    "Jimmy Flemion Dennis Flemion Prince side project The Family",
    "The Frogs musicians substitute keyboardist The Family band Prince",
    "The Family band Prince keyboardist replacement member",
    "The Frogs band members musical collaborations Prince projects"
]

# Get SerpAPI key from environment variables
api_key = os.getenv("SERPAPI_API_KEY")

if api_key is None:
    print("Error: Missing API key. Make sure you have SERPAPI_API_KEY in your environment variables.")
    print("Cannot proceed with search without API key.")
else:
    print(f"API key found. Starting search with {len(search_queries)} targeted queries\n")
    
    all_search_results = []
    frogs_member_info = []
    family_band_connections = []
    
    for i, query in enumerate(search_queries, 1):
        print(f"=== SEARCH {i}/{len(search_queries)} ===")
        print(f"Query: {query}")
        
        # Prepare API request parameters
        params = {
            "q": query,
            "api_key": api_key,
            "engine": "google",
            "google_domain": "google.com",
            "safe": "off",
            "num": 10,
            "type": "search",
        }
        
        try:
            # Make API request to SerpAPI
            response = requests.get("https://serpapi.com/search.json", params=params, timeout=30)
            
            if response.status_code == 200:
                results = response.json()
                organic_results = results.get('organic_results', [])
                print(f"Search successful - found {len(organic_results)} organic results")
                
                # Store results with query context
                search_result = {
                    'query_number': i,
                    'query_text': query,
                    'results_data': results,
                    'organic_count': len(organic_results)
                }
                all_search_results.append(search_result)
                
                # Analyze results immediately for relevant information
                if organic_results:
                    print("\nAnalyzing results for Frogs members and Family band connections:\n")
                    
                    for j, result in enumerate(organic_results[:8], 1):
                        title = result.get('title', 'No title')
                        link = result.get('link', 'No link')
                        snippet = result.get('snippet', 'No snippet')
                        
                        # Create combined text for analysis
                        combined_text = f"{title} {snippet}".lower()
                        
                        print(f"Result {j}:")
                        print(f"Title: {title}")
                        print(f"URL: {link}")
                        print(f"Snippet: {snippet[:200]}{'...' if len(snippet) > 200 else ''}")
                        
                        # Look for The Frogs band indicators
                        frogs_indicators = ['the frogs', 'frogs band', 'jimmy flemion', 'dennis flemion', 'flemion brothers']
                        found_frogs = [kw for kw in frogs_indicators if kw in combined_text]
                        
                        # Look for The Family band indicators
                        family_indicators = ['the family', 'family band', 'prince side project', 'prince family']
                        found_family = [kw for kw in family_indicators if kw in combined_text]
                        
                        # Look for keyboardist/musician role indicators
                        musician_indicators = ['keyboardist', 'keyboard', 'musician', 'filled in', 'substitute', 'replacement', 'session']
                        found_musician = [kw for kw in musician_indicators if kw in combined_text]
                        
                        # Look for Prince connection indicators
                        prince_indicators = ['prince', 'prince rogers nelson', 'paisley park', 'minneapolis sound']
                        found_prince = [kw for kw in prince_indicators if kw in combined_text]
                        
                        # Highlight significant matches
                        if found_frogs:
                            print(f"*** THE FROGS INDICATORS: {', '.join(found_frogs)} ***")
                        
                        if found_family:
                            print(f"*** THE FAMILY BAND INDICATORS: {', '.join(found_family)} ***")
                        
                        if found_musician:
                            print(f"*** MUSICIAN ROLE INDICATORS: {', '.join(found_musician)} ***")
                        
                        if found_prince:
                            print(f"*** PRINCE CONNECTION INDICATORS: {', '.join(found_prince)} ***")
                        
                        # Calculate relevance score
                        relevance_score = len(found_frogs) + len(found_family) + len(found_musician) + len(found_prince)
                        if relevance_score >= 2:
                            print(f"*** RELEVANT RESULT (Score: {relevance_score}/4) ***")
                            
                            # Store highly relevant results
                            if found_frogs and (found_family or found_prince):
                                frogs_member_info.append({
                                    'title': title,
                                    'url': link,
                                    'snippet': snippet,
                                    'frogs_indicators': found_frogs,
                                    'family_indicators': found_family,
                                    'musician_indicators': found_musician,
                                    'prince_indicators': found_prince,
                                    'relevance_score': relevance_score,
                                    'query_source': query
                                })
                        
                        # Look for specific member names and roles
                        name_patterns = [
                            r'jimmy flemion.*keyboardist',
                            r'dennis flemion.*keyboardist',
                            r'keyboardist.*jimmy flemion',
                            r'keyboardist.*dennis flemion',
                            r'flemion.*the family',
                            r'the family.*flemion'
                        ]
                        
                        for pattern in name_patterns:
                            if re.search(pattern, combined_text, re.IGNORECASE):
                                print(f"*** SPECIFIC MEMBER-ROLE MATCH: {pattern} ***")
                        
                        print("-" * 60)
                
                print(f"\nCompleted search {i}/{len(search_queries)}\n")
                
            else:
                print(f"Search failed with status {response.status_code}: {response.text}")
                
        except Exception as e:
            print(f"Error during search {i}: {str(e)}")
        
        print("=" * 70)
        print()
    
    # Save comprehensive search results
    comprehensive_results = {
        'search_objective': 'Find which Frogs member filled in as keyboardist for The_Family_(band)',
        'search_timestamp': '2024-12-19',
        'total_queries_executed': len(search_queries),
        'queries_used': search_queries,
        'all_search_results': all_search_results,
        'total_organic_results': sum([sr['organic_count'] for sr in all_search_results]),
        'frogs_member_info': frogs_member_info,
        'family_band_connections': family_band_connections
    }
    
    with open('workspace/frogs_family_band_search.json', 'w', encoding='utf-8') as f:
        json.dump(comprehensive_results, f, indent=2, ensure_ascii=False)
    
    print(f"\n=== SEARCH ANALYSIS SUMMARY ===\n")
    print(f"Total searches conducted: {len(search_queries)}")
    print(f"Total organic results collected: {comprehensive_results['total_organic_results']}")
    print(f"Frogs member info entries found: {len(frogs_member_info)}")
    print(f"Results saved to: workspace/frogs_family_band_search.json")
    
    # Display summary of relevant findings
    if frogs_member_info:
        print("\n=== RELEVANT FROGS MEMBER FINDINGS ===\n")
        for i, info in enumerate(frogs_member_info, 1):
            print(f"Finding {i}:")
            print(f"  Title: {info['title']}")
            print(f"  URL: {info['url']}")
            print(f"  Frogs Indicators: {info['frogs_indicators']}")
            print(f"  Family Indicators: {info['family_indicators']}")
            print(f"  Musician Indicators: {info['musician_indicators']}")
            print(f"  Prince Indicators: {info['prince_indicators']}")
            print(f"  Relevance Score: {info['relevance_score']}/4")
            print(f"  Snippet: {info['snippet'][:150]}...")
            print()
    else:
        print("\n=== NO DIRECT CONNECTIONS FOUND ===\n")
        print("Initial search did not find direct connections between")
        print("The Frogs members and The Family band keyboardist role.")
        print("\nThis suggests either:")
        print("1. The connection is not well-documented online")
        print("2. The information requires more specific search terms")
        print("3. The connection may be indirect or through other musicians")
    
    print("\n=== NEXT STEPS ===\n")
    print("1. Analyze collected search results for any subtle connections")
    print("2. Search specifically for The Family band keyboardist history")
    print("3. Research individual Frogs members' session work")
    print("4. Cross-reference Prince collaborator databases")
    
    print("\n=== SEARCH PHASE COMPLETE ===\n")
    print("Ready for detailed analysis of search results to identify the specific connection.")
