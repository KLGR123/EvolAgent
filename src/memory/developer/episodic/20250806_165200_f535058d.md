import requests
import json
import os
from datetime import datetime
from urllib.parse import quote

# Get SerpAPI key from environment variables
api_key = os.getenv("SERPAPI_API_KEY")

if api_key is None:
    print("Error: Missing API key. Make sure you have SERPAPI_API_KEY in your environment variables.")
else:
    print("API key found, proceeding with LibreTexts chemistry search...")
    print("=" * 80)
    print("SEARCHING FOR LIBRETEXTS INTRODUCTORY CHEMISTRY MATERIALS")
    print("Target: Materials by Marisa Alviar-Agnew & Henry Agnew, CK-12 license, 08/21/2023")
    print("Objective: Find '1.E Exercises' section mentioning equine veterinarian")
    print("=" * 80)
    
    # Define targeted search queries for LibreTexts chemistry materials
    search_queries = [
        'site:chem.libretexts.org "Marisa Alviar-Agnew" "Henry Agnew" "Introductory Chemistry"',
        'site:libretexts.org "Introductory Chemistry" "Marisa Alviar-Agnew" "Henry Agnew" "CK-12"',
        '"LibreTexts Introductory Chemistry" "Marisa Alviar-Agnew" "Henry Agnew" "08/21/2023"',
        'site:chem.libretexts.org "1.E Exercises" "equine veterinarian"',
        '"LibreTexts" "Introductory Chemistry" "CK-12 license" "Marisa Alviar-Agnew"',
        'site:libretexts.org "1.E Exercises" chemistry equine veterinarian',
        '"Marisa Alviar-Agnew" "Henry Agnew" chemistry textbook "CK-12" license',
        'site:chem.libretexts.org "Introductory Chemistry" exercises equine veterinarian'
    ]
    
    print(f"Executing {len(search_queries)} targeted searches for LibreTexts materials...\n")
    
    # Store all search results
    all_search_results = []
    
    for i, query in enumerate(search_queries, 1):
        print(f"Search {i}/{len(search_queries)}: {query}")
        print("-" * 60)
        
        # Prepare API request parameters
        params = {
            "q": query,
            "api_key": api_key,
            "engine": "google",
            "google_domain": "google.com",
            "safe": "off",
            "num": 15,
            "type": "search"
        }
        
        try:
            # Make API request to SerpAPI
            response = requests.get("https://serpapi.com/search.json", params=params, timeout=30)
            
            if response.status_code == 200:
                results = response.json()
                
                if results.get("organic_results"):
                    print(f"Found {len(results['organic_results'])} results")
                    
                    # Store results with query context
                    query_results = {
                        'query': query,
                        'results': results['organic_results'],
                        'search_number': i,
                        'timestamp': datetime.now().isoformat()
                    }
                    all_search_results.append(query_results)
                    
                    # Analyze and display results for LibreTexts materials
                    libretexts_found = False
                    for j, result in enumerate(results['organic_results'][:10], 1):
                        title = result.get('title', 'No title')
                        link = result.get('link', 'No URL')
                        snippet = result.get('snippet', 'No snippet')
                        
                        # Check for LibreTexts indicators
                        title_lower = title.lower()
                        snippet_lower = snippet.lower()
                        link_lower = link.lower()
                        combined_text = f"{title_lower} {snippet_lower} {link_lower}"
                        
                        # Key indicators for the target materials
                        key_indicators = [
                            'libretexts',
                            'marisa alviar-agnew',
                            'henry agnew',
                            'introductory chemistry',
                            'ck-12',
                            '1.e exercises',
                            'equine veterinarian',
                            'exercises',
                            'chemistry'
                        ]
                        
                        matching_indicators = [ind for ind in key_indicators if ind in combined_text]
                        
                        print(f"\n  Result {j}:")
                        print(f"    Title: {title}")
                        print(f"    URL: {link}")
                        print(f"    Snippet: {snippet[:300]}{'...' if len(snippet) > 300 else ''}")
                        
                        if matching_indicators:
                            print(f"    ‚≠ê MATCHING INDICATORS: {', '.join(matching_indicators)}")
                            libretexts_found = True
                            
                            # Special attention to LibreTexts chemistry domain
                            if 'libretexts.org' in link_lower:
                                print(f"    üéØ LIBRETEXTS DOMAIN: Direct access to LibreTexts materials")
                                
                                # Check for specific chapter or exercise content
                                if '1.e' in combined_text or 'exercises' in combined_text:
                                    print(f"    üìö EXERCISE CONTENT: Potential match for 1.E Exercises section")
                                
                                # Check for author mentions
                                if 'alviar-agnew' in combined_text or 'henry agnew' in combined_text:
                                    print(f"    üë• AUTHOR MATCH: Contains target authors")
                                
                                # Check for equine veterinarian mention
                                if 'equine' in combined_text or 'veterinarian' in combined_text:
                                    print(f"    üêé EQUINE VETERINARIAN: Direct mention found!")
                    
                    if not libretexts_found:
                        print("    No LibreTexts materials found in this search")
                        
                else:
                    print("No organic results found for this query")
                    all_search_results.append({
                        'query': query,
                        'results': [],
                        'search_number': i,
                        'timestamp': datetime.now().isoformat()
                    })
            else:
                print(f"Error: API request failed with status {response.status_code}: {response.text}")
                
        except Exception as e:
            print(f"Error during search {i}: {str(e)}")
            continue
        
        print("\n")
    
    print("\n" + "=" * 80)
    print("SEARCH RESULTS ANALYSIS")
    print("=" * 80)
    
    # Save all search results to workspace for detailed analysis
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    results_file = f"workspace/libretexts_chemistry_search_results_{timestamp}.json"
    
    with open(results_file, 'w', encoding='utf-8') as f:
        json.dump({
            'search_target': 'LibreTexts Introductory Chemistry by Marisa Alviar-Agnew & Henry Agnew',
            'objective': 'Find 1.E Exercises section mentioning equine veterinarian',
            'search_timestamp': timestamp,
            'total_queries': len(search_queries),
            'queries_executed': search_queries,
            'search_results': all_search_results
        }, f, indent=2, ensure_ascii=False)
    
    print(f"Search results saved to: {results_file}")
    
    # Analyze findings
    total_results = sum(len(query_result['results']) for query_result in all_search_results)
    print(f"Total search results collected: {total_results}")
    
    # Look for direct LibreTexts matches
    libretexts_matches = []
    author_matches = []
    exercise_matches = []
    equine_veterinarian_matches = []
    
    for query_result in all_search_results:
        for result in query_result['results']:
            title = result.get('title', '').lower()
            snippet = result.get('snippet', '').lower()
            link = result.get('link', '').lower()
            combined = f"{title} {snippet} {link}"
            
            # LibreTexts domain matches
            if 'libretexts.org' in link:
                libretexts_matches.append({
                    'query': query_result['query'],
                    'title': result.get('title'),
                    'link': result.get('link'),
                    'snippet': result.get('snippet'),
                    'match_type': 'LibreTexts Domain'
                })
            
            # Author matches
            if 'alviar-agnew' in combined or 'henry agnew' in combined:
                author_matches.append({
                    'query': query_result['query'],
                    'title': result.get('title'),
                    'link': result.get('link'),
                    'snippet': result.get('snippet'),
                    'match_type': 'Author Match'
                })
            
            # Exercise content matches
            if '1.e' in combined or ('exercises' in combined and 'chemistry' in combined):
                exercise_matches.append({
                    'query': query_result['query'],
                    'title': result.get('title'),
                    'link': result.get('link'),
                    'snippet': result.get('snippet'),
                    'match_type': 'Exercise Content'
                })
            
            # Equine veterinarian matches
            if 'equine' in combined or 'veterinarian' in combined:
                equine_veterinarian_matches.append({
                    'query': query_result['query'],
                    'title': result.get('title'),
                    'link': result.get('link'),
                    'snippet': result.get('snippet'),
                    'match_type': 'Equine Veterinarian Mention'
                })
    
    print(f"\nüìä SEARCH FINDINGS:")
    print(f"LibreTexts domain matches: {len(libretexts_matches)}")
    print(f"Author matches: {len(author_matches)}")
    print(f"Exercise content matches: {len(exercise_matches)}")
    print(f"Equine veterinarian mentions: {len(equine_veterinarian_matches)}")
    
    # Display top matches for each category
    if libretexts_matches:
        print(f"\nüéØ TOP LIBRETEXTS DOMAIN MATCHES:")
        for i, match in enumerate(libretexts_matches[:5], 1):
            print(f"  {i}. {match['title']}")
            print(f"     URL: {match['link']}")
            print(f"     Query: {match['query']}")
            print(f"     Snippet: {match['snippet'][:200]}...")
            print()
    
    if author_matches:
        print(f"\nüë• AUTHOR MATCHES:")
        for i, match in enumerate(author_matches[:3], 1):
            print(f"  {i}. {match['title']}")
            print(f"     URL: {match['link']}")
            print(f"     Snippet: {match['snippet'][:150]}...")
            print()
    
    if exercise_matches:
        print(f"\nüìö EXERCISE CONTENT MATCHES:")
        for i, match in enumerate(exercise_matches[:3], 1):
            print(f"  {i}. {match['title']}")
            print(f"     URL: {match['link']}")
            print(f"     Snippet: {match['snippet'][:150]}...")
            print()
    
    if equine_veterinarian_matches:
        print(f"\nüêé EQUINE VETERINARIAN MATCHES:")
        for i, match in enumerate(equine_veterinarian_matches[:3], 1):
            print(f"  {i}. {match['title']}")
            print(f"     URL: {match['link']}")
            print(f"     Snippet: {match['snippet'][:150]}...")
            print()
    
    print(f"\nüìã INITIAL SEARCH STATUS:")
    if libretexts_matches and author_matches:
        print("‚úÖ LIBRETEXTS MATERIALS IDENTIFIED with author matches")
        print("   - LibreTexts chemistry content found")
        print("   - Target authors mentioned")
        print("   - Ready for detailed content examination")
    elif libretexts_matches:
        print("‚ö†Ô∏è LIBRETEXTS MATERIALS FOUND but limited author verification")
        print("   - LibreTexts chemistry content found")
        print("   - May need more specific searches for authors")
    else:
        print("‚ùå LIMITED LIBRETEXTS RESULTS")
        print("   - May need alternative search strategies")
        print("   - Consider direct LibreTexts site navigation")
    
    # Save categorized matches for next analysis phase
    matches_summary = {
        'timestamp': timestamp,
        'libretexts_matches': libretexts_matches,
        'author_matches': author_matches,
        'exercise_matches': exercise_matches,
        'equine_veterinarian_matches': equine_veterinarian_matches,
        'search_summary': {
            'total_results': total_results,
            'libretexts_count': len(libretexts_matches),
            'author_count': len(author_matches),
            'exercise_count': len(exercise_matches),
            'equine_count': len(equine_veterinarian_matches)
        }
    }
    
    summary_file = f"workspace/libretexts_search_summary_{timestamp}.json"
    with open(summary_file, 'w', encoding='utf-8') as f:
        json.dump(matches_summary, f, indent=2, ensure_ascii=False)
    
    print(f"\nCategorized matches saved to: {summary_file}")
    print("Ready for next phase: detailed examination of LibreTexts materials.")
