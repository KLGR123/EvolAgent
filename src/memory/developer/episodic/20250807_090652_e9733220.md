```
import os
import json
from bs4 import BeautifulSoup
import re

print('=== EXTRACTING SPECIFIC ARTIST/SUFFRAGETTE NAMES FROM SUFFOLK RESEARCH FILES ===')
print('Previous searches identified promising Suffolk locations - now extracting individual names')
print('Objective: Find specific people who were both artists and suffragettes from Suffolk supernatural locations\n')

# Ensure workspace directory exists
os.makedirs('workspace', exist_ok=True)

# Initialize results storage
name_extraction_results = {
    'timestamp': '2024-12-19',
    'objective': 'Extract specific artist and suffragette names from Suffolk location research',
    'files_analyzed': [],
    'individuals_found': [],
    'analysis_summary': {}
}

print('=== PHASE 1: INSPECTING AVAILABLE RESEARCH FILES ===')
print('=' * 70)

# List all files in workspace to understand what we have
all_files = os.listdir('workspace')
print(f'Total files in workspace: {len(all_files)}')

# Categorize files
json_files = [f for f in all_files if f.endswith('.json')]
html_files = [f for f in all_files if f.endswith('.html')]
txt_files = [f for f in all_files if f.endswith('.txt')]

print(f'JSON files: {len(json_files)}')
for f in json_files:
    print(f'  • {f}')

print(f'\nHTML files: {len(html_files)}')
for f in html_files[:10]:  # Show first 10
    print(f'  • {f}')
if len(html_files) > 10:
    print(f'  ... and {len(html_files) - 10} more')

print(f'\nTXT files: {len(txt_files)}')
for f in txt_files:
    print(f'  • {f}')

# First, let's examine any comprehensive research JSON file
research_json_files = [f for f in json_files if 'research' in f.lower() or 'suffolk' in f.lower()]

if research_json_files:
    print(f'\n=== EXAMINING RESEARCH JSON FILES ===')
    for json_file in research_json_files:
        print(f'\nAnalyzing: {json_file}')
        try:
            json_path = os.path.join('workspace', json_file)
            with open(json_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            
            print(f'File structure:')
            for key in data.keys():
                if isinstance(data[key], list):
                    print(f'  • {key}: {len(data[key])} items')
                elif isinstance(data[key], dict):
                    print(f'  • {key}: {len(data[key])} keys')
                else:
                    print(f'  • {key}: {type(data[key]).__name__}')
            
            # Look for artist/suffragette research data
            if 'artist_suffragette_leads' in data:
                leads = data['artist_suffragette_leads']
                print(f'\nArtist/Suffragette leads found: {len(leads)}')
                
                for i, lead in enumerate(leads, 1):
                    location = lead.get('location', 'Unknown')
                    promising = lead.get('promising', False)
                    print(f'  {i}. {location} - Promising: {promising}')
                    
                    # Check for associated files
                    if 'artist_search' in lead and 'file' in lead['artist_search']:
                        artist_file = lead['artist_search']['file']
                        print(f'     Artist file: {artist_file}')
                        name_extraction_results['files_analyzed'].append({
                            'file': artist_file,
                            'location': location,
                            'type': 'artist'
                        })
                    
                    if 'suffragette_search' in lead and 'file' in lead['suffragette_search']:
                        suffragette_file = lead['suffragette_search']['file']
                        print(f'     Suffragette file: {suffragette_file}')
                        name_extraction_results['files_analyzed'].append({
                            'file': suffragette_file,
                            'location': location,
                            'type': 'suffragette'
                        })
        
        except Exception as e:
            print(f'Error reading {json_file}: {str(e)}')

print('\n=== PHASE 2: IDENTIFYING SUFFOLK LOCATION HTML FILES ===')
print('=' * 70)

# Look for HTML files related to Suffolk locations
suffolk_locations = ['aldeburgh', 'bury', 'lavenham', 'melford', 'woodbridge']
suffolk_html_files = []

for html_file in html_files:
    for location in suffolk_locations:
        if location in html_file.lower():
            suffolk_html_files.append({
                'file': html_file,
                'location': location.title(),
                'type': 'artist' if 'artist' in html_file.lower() else 'suffragette'
            })
            break

print(f'Found {len(suffolk_html_files)} Suffolk-related HTML files:')
for file_info in suffolk_html_files:
    print(f'  • {file_info["file"]} ({file_info["location"]}, {file_info["type"]})')

# Combine with files from JSON analysis
all_target_files = name_extraction_results['files_analyzed'] + suffolk_html_files

# Remove duplicates
seen_files = set()
unique_files = []
for file_info in all_target_files:
    file_key = file_info['file']
    if file_key not in seen_files:
        seen_files.add(file_key)
        unique_files.append(file_info)

print(f'\nTotal unique files to analyze: {len(unique_files)}')

print('\n=== PHASE 3: EXTRACTING NAMES FROM HTML FILES ===')
print('=' * 70)

# Process each HTML file to extract potential names
for i, file_info in enumerate(unique_files, 1):
    filename = file_info['file']
    location = file_info.get('location', 'Unknown')
    search_type = file_info.get('type', 'unknown')
    
    print(f'\n--- Analyzing File {i}/{len(unique_files)}: {filename} ---')
    print(f'Location: {location}, Type: {search_type}')
    
    try:
        filepath = os.path.join('workspace', filename)
        
        # Check if file exists
        if not os.path.exists(filepath):
            print(f'❌ File not found: {filepath}')
            continue
        
        with open(filepath, 'r', encoding='utf-8') as f:
            html_content = f.read()
        
        print(f'File size: {len(html_content):,} characters')
        
        # Parse HTML
        soup = BeautifulSoup(html_content, 'html.parser')
        
        # Extract text content
        page_text = soup.get_text()
        
        # Look for potential names using multiple strategies
        potential_names = set()
        
        # Strategy 1: Find names in titles and headings
        for heading in soup.find_all(['h1', 'h2', 'h3', 'h4', 'title']):
            heading_text = heading.get_text().strip()
            # Look for capitalized names (First Last)
            name_matches = re.findall(r'\b[A-Z][a-z]+ [A-Z][a-z]+\b', heading_text)
            potential_names.update(name_matches)
        
        # Strategy 2: Find names in context with relevant keywords
        relevant_keywords = {
            'artist': ['artist', 'painter', 'sculptor', 'born', 'lived', 'painted', 'artwork'],
            'suffragette': ['suffragette', 'suffrage', 'women', 'rights', 'activist', 'campaign', 'vote']
        }
        
        keywords_to_use = relevant_keywords.get(search_type, [])
        if not keywords_to_use:
            keywords_to_use = relevant_keywords['artist'] + relevant_keywords['suffragette']
        
        # Split text into sentences and look for names in relevant contexts
        sentences = re.split(r'[.!?]', page_text)
        for sentence in sentences:
            sentence = sentence.strip()
            if 20 <= len(sentence) <= 300:  # Reasonable sentence length
                sentence_lower = sentence.lower()
                
                # Check if sentence contains relevant keywords
                if any(keyword in sentence_lower for keyword in keywords_to_use):
                    # Extract potential names from this sentence
                    name_matches = re.findall(r'\b[A-Z][a-z]+ [A-Z][a-z]+\b', sentence)
                    potential_names.update(name_matches)
        
        # Strategy 3: Find names in link text
        for link in soup.find_all('a'):
            link_text = link.get_text().strip()
            if 5 <= len(link_text) <= 100:
                name_matches = re.findall(r'\b[A-Z][a-z]+ [A-Z][a-z]+\b', link_text)
                potential_names.update(name_matches)
        
        # Filter out common false positives
        false_positives = {
            'New York', 'United States', 'Great Britain', 'Long Island', 'New England',
            'World War', 'First World', 'Second World', 'High School', 'Art Gallery',
            'Art Museum', 'Women Rights', 'Human Rights', 'Civil Rights', 'Royal Academy',
            'British Museum', 'National Gallery', 'Tate Gallery', 'Victoria Albert',
            'East Anglia', 'West Suffolk', 'East Suffolk', 'South Suffolk', 'North Suffolk'
        }
        
        # Filter and validate names
        filtered_names = set()
        for name in potential_names:
            if name not in false_positives:
                parts = name.split()
                if len(parts) == 2:
                    first_name, last_name = parts
                    # Basic validation for likely person names
                    if (len(first_name) >= 3 and len(last_name) >= 3 and 
                        first_name.isalpha() and last_name.isalpha()):
                        filtered_names.add(name)
        
        print(f'Raw names found: {len(potential_names)}')
        print(f'Filtered names: {len(filtered_names)}')
        
        if filtered_names:
            print('Top names found:')
            for name in list(filtered_names)[:5]:
                print(f'  • {name}')
            
            # Add to results
            for name in filtered_names:
                individual_entry = {
                    'name': name,
                    'location': location,
                    'type': search_type,
                    'source_file': filename
                }
                name_extraction_results['individuals_found'].append(individual_entry)
        else:
            print('❌ No clear names found after filtering')
    
    except Exception as e:
        print(f'Error processing {filename}: {str(e)}')

print('\n=== PHASE 4: ANALYZING EXTRACTED INDIVIDUALS ===')
print('=' * 70)

total_individuals = len(name_extraction_results['individuals_found'])
print(f'Total individuals extracted: {total_individuals}')

if total_individuals > 0:
    # Group by name to find individuals appearing in multiple contexts
    by_name = {}
    for individual in name_extraction_results['individuals_found']:
        name = individual['name']
        if name not in by_name:
            by_name[name] = []
        by_name[name].append(individual)
    
    # Group by location
    by_location = {}
    for individual in name_extraction_results['individuals_found']:
        location = individual['location']
        if location not in by_location:
            by_location[location] = []
        by_location[location].append(individual)
    
    # Group by type
    by_type = {'artist': [], 'suffragette': [], 'unknown': []}
    for individual in name_extraction_results['individuals_found']:
        type_key = individual.get('type', 'unknown')
        if type_key not in by_type:
            by_type[type_key] = []
        by_type[type_key].append(individual)
    
    print('\n📊 ANALYSIS BY LOCATION:')
    for location, individuals in by_location.items():
        unique_names = set(i['name'] for i in individuals)
        print(f'  {location}: {len(unique_names)} unique individuals')
        
        # Show sample names
        sample_names = list(unique_names)[:3]
        if sample_names:
            print(f'    Sample: {', '.join(sample_names)}')
    
    print('\n📊 ANALYSIS BY TYPE:')
    for type_name, individuals in by_type.items():
        unique_names = set(i['name'] for i in individuals)
        print(f'  {type_name.title()}: {len(unique_names)} unique individuals')
    
    # Find individuals who appear as both artists and suffragettes
    artist_names = set(i['name'] for i in by_type['artist'])
    suffragette_names = set(i['name'] for i in by_type['suffragette'])
    
    dual_individuals = artist_names.intersection(suffragette_names)
    
    print('\n🎯 KEY FINDINGS:')
    if dual_individuals:
        print(f'INDIVIDUALS WHO ARE BOTH ARTISTS AND SUFFRAGETTES ({len(dual_individuals)}):')
        for name in dual_individuals:
            # Find their locations
            locations = set()
            for individual in name_extraction_results['individuals_found']:
                if individual['name'] == name:
                    locations.add(individual['location'])
            print(f'  ⭐ {name} (from {', '.join(locations)})')
    else:
        print('No individuals found who appear in both artist and suffragette searches')
    
    # Show top candidates from each category
    print('\n🎨 TOP ARTISTS:')
    unique_artists = list(set(i['name'] for i in by_type['artist']))
    for name in unique_artists[:5]:
        locations = set(i['location'] for i in by_type['artist'] if i['name'] == name)
        print(f'  • {name} (from {', '.join(locations)})')
    
    print('\n🗳️ TOP SUFFRAGETTES:')
    unique_suffragettes = list(set(i['name'] for i in by_type['suffragette']))
    for name in unique_suffragettes[:5]:
        locations = set(i['location'] for i in by_type['suffragette'] if i['name'] == name)
        print(f'  • {name} (from {', '.join(locations)})')
    
    # Identify most frequently mentioned individuals
    name_frequency = {}
    for name, occurrences in by_name.items():
        name_frequency[name] = len(occurrences)
    
    most_frequent = sorted(name_frequency.items(), key=lambda x: x[1], reverse=True)
    
    print('\n📈 MOST FREQUENTLY MENTIONED:')
    for name, count in most_frequent[:5]:
        print(f'  • {name}: {count} mentions')

else:
    print('❌ No individual names were successfully extracted')
    print('\nPossible reasons:')
    print('  • HTML files may not contain specific biographical information')
    print('  • Search results may be too general')
    print('  • Name extraction patterns may need refinement')

# Save extraction results
extraction_file = os.path.join('workspace', 'suffolk_individuals_extracted.json')

name_extraction_results['analysis_summary'] = {
    'total_files_analyzed': len(unique_files),
    'total_individuals': total_individuals,
    'unique_names': len(set(i['name'] for i in name_extraction_results['individuals_found'])),
    'artists_found': len(set(i['name'] for i in name_extraction_results['individuals_found'] if i['type'] == 'artist')),
    'suffragettes_found': len(set(i['name'] for i in name_extraction_results['individuals_found'] if i['type'] == 'suffragette')),
    'dual_individuals': len(dual_individuals) if total_individuals > 0 else 0,
    'locations_covered': len(set(i['location'] for i in name_extraction_results['individuals_found']))
}

with open(extraction_file, 'w', encoding='utf-8') as f:
    json.dump(name_extraction_results, f, indent=2, ensure_ascii=False)

print(f'\n💾 EXTRACTION RESULTS SAVED TO: {extraction_file}')

print('\n=== PHASE 5: FINAL RECOMMENDATIONS ===')
print('=' * 60)

if total_individuals > 0:
    print('✅ SUCCESS: Individual names extracted from Suffolk research files')
    print('\n🎯 NEXT STEPS TO COMPLETE THE RESEARCH:')
    print('1. 📚 Conduct detailed biographical verification of extracted individuals')
    print('2. ✅ Confirm their actual connections to art and suffrage movements')
    print('3. 📍 Verify their association with the identified Suffolk locations')
    print('4. 🔍 Cross-reference with historical databases and archives')
    
    if dual_individuals:
        print(f'5. ⭐ Priority research on dual individuals: {', '.join(list(dual_individuals)[:3])}')
    else:
        print('5. 🔄 Research potential connections between artists and suffragettes from same locations')
else:
    print('⚠️ ALTERNATIVE APPROACHES NEEDED:')
    print('1. 🔍 More targeted searches for specific Suffolk artists and suffragettes')
    print('2. 📚 Direct consultation of historical databases and biographical dictionaries')
    print('3. 🏛️ Contact Suffolk local history societies and museums')
    print('4. 📖 Search specialized art history and women\'s history resources')

print('\n📊 FINAL STATISTICS:')
for key, value in name_extraction_results['analysis_summary'].items():
    print(f'   • {key.replace("_", " ").title()}: {value}')

print('\n=== SUFFOLK ARTIST/SUFFRAGETTE NAME EXTRACTION COMPLETE ===')
```
