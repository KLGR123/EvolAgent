def analyze_cricket_content(html_content, query):
    """Extract and analyze cricket-specific search results from HTML content"""
    soup = BeautifulSoup(html_content, 'html.parser')

            link = link_elem.get('href') if link_elem else 'No link'
            
            # Extract snippet/description
            snippet_elem = container.find(['p', 'span', 'div'], class_=lambda x: x and 'snippet' in str(x).lower()) or container.find('p')
            snippet = snippet_elem.get_text().strip() if snippet_elem else 'No snippet'
            
            # Skip if no meaningful content
            if len(title) < 5 or title == 'No title':
                continue
                
            # Calculate relevance score for Greg Chappell cricket information
            combined_text = f'{title} {snippet} {link}'.lower()
            
            relevance_score = 0
            matched_terms = []
            death_indicators = []
            alive_indicators = []
            
            # Key terms for Greg Chappell cricket information
            key_terms = {
                'greg chappell': 8,
                'gregory chappell': 8,
                'australian cricket': 5,
                'australia captain': 5,
                'cricket legend': 4,
                'death': 6,

                'cricket coach': 3,
                'india coach': 3,
                'cricket australia': 4,
                'test cricket': 3,
                'batting': 2,
                'captain': 3
            }
            
            # Look for specific death/alive year patterns
            import re
            year_patterns = re.findall(r'\b(19\d{2}|20\d{2})\b', combined_text)
            
            for term, weight in key_terms.items():
                if term in combined_text:
                    relevance_score += weight
                    matched_terms.append(term)
            
            # Check for death-related information with years
            death_words = ['death', 'died', 'obituary', 'passed away', 'demise']
            alive_words = ['still alive', 'living', 'current', 'recent', 'active']
            
            for year in year_patterns:
                for death_word in death_words:
                    if death_word in combined_text:
                        # Check if year appears near death word (within 50 characters)
                        death_pos = combined_text.find(death_word)
                        year_pos = combined_text.find(year)
                        if abs(death_pos - year_pos) < 50:
                            death_indicators.append(f'{year} (near "{death_word}")')
                            relevance_score += 5
                
                for alive_word in alive_words:
                    if alive_word in combined_text:
                        alive_pos = combined_text.find(alive_word)
                        year_pos = combined_text.find(year)
                        if abs(alive_pos - year_pos) < 50:
                            alive_indicators.append(f'{year} (near "{alive_word}")')
                            relevance_score += 4
            
            # Boost score for cricket-specific sources
            cricket_sources = ['cricket.com.au', 'espncricinfo', 'cricbuzz', 'cricket australia', 'wisden']
            for source in cricket_sources:
                if source in combined_text:
                    relevance_score += 3
                    matched_terms.append(f'cricket_source:{source}')
            
            if relevance_score > 0:  # Only include results with some relevance
                results.append({
                    'title': title[:300],
                    'link': link,
                    'snippet': snippet[:500],

                    'death_indicators': death_indicators,
                    'alive_indicators': alive_indicators,
                    'years_mentioned': year_patterns,
                    'query': query
                })
                
        except Exception as e:
            continue  # Skip problematic results
    
    return results
