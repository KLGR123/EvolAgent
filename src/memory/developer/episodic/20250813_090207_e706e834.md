Look for official publication details, creator credits, and any collaborative relationships mentioned in connection with this work.

```
from ddgs import DDGS
import json
import os
import time
import re

print("=== LITTLE MONSTERS COMIC ARTIST SEARCH ===")
print("Using reliable search service to find artist information")
print("=" * 60)

# Create workspace directory
workspace_dir = 'workspace'
os.makedirs(workspace_dir, exist_ok=True)

# Search queries specifically targeting Little Monsters comic artist information
search_queries = [
    'Little Monsters comic Gold Key artist creator',
    'Little Monsters Gold Key Comics illustrator artist',
    'Little Monsters comic book artist writer credits',
    'Gold Key Little Monsters comic series creator artist',
    'Little Monsters Harvey Comics artist illustrator'
]

print("\nüîç CONDUCTING SEARCHES FOR LITTLE MONSTERS ARTIST...")
print("-" * 50)

searcher = DDGS(timeout=15)
all_search_results = []
artist_mentions = []

for query_num, query in enumerate(search_queries, 1):
    print(f"\n{query_num}. Searching: '{query}'")
    
    try:
        # Search with multiple backends for reliability
        results = searcher.text(
            query, 
            max_results=10, 
            backend=["google", "duckduckgo", "bing"],
            safesearch="off",
            region="en-us"
        )
        
        if results:
            print(f"   ‚úì Found {len(results)} results")
            
            for result_idx, result in enumerate(results, 1):
                title = result.get('title', '')
                body = result.get('body', '')
                href = result.get('href', '')
                
                print(f"     {result_idx}. {title[:60]}...")
                print(f"        URL: {href[:70]}...")
                print(f"        Snippet: {body[:80]}...")
                
                # Look for artist/creator information in title and body
                combined_text = f"{title} {body}".lower()
                
                # Check if this result is about Little Monsters comic
                if 'little monsters' in combined_text:
                    print(f"        ‚úì Contains Little Monsters content")
                    
                    # Look for creator patterns
                    creator_patterns = [
                        r'artist[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)',
                        r'created by[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)',
                        r'art by[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)',
                        r'illustrated by[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)',
                        r'drawn by[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)',
                        r'writer[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)',
                        r'story[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)'
                    ]
                    
                    # Also look for names that appear after common comic terms
                    full_text = f"{title} {body}"
                    
                    for pattern in creator_patterns:
                        matches = re.findall(pattern, full_text, re.IGNORECASE)
                        for match in matches:
                            # Clean up the match
                            clean_name = match.strip()
                            
                            # Filter out obvious non-names
                            exclude_terms = [
                                'Little Monsters', 'Gold Key', 'Comics', 'Comic',
                                'Harvey', 'Publishing', 'Books', 'Series'
                            ]
                            
                            is_valid_name = True
                            for exclude in exclude_terms:
                                if exclude.lower() in clean_name.lower():
                                    is_valid_name = False
                                    break
                            
                            if is_valid_name and len(clean_name) > 3:
                                artist_mentions.append({
                                    'name': clean_name,
                                    'source_url': href,
                                    'source_title': title,
                                    'context': pattern,
                                    'query': query
                                })
                                print(f"        ‚Üí Found potential artist: {clean_name}")
                    
                    # Also look for names in specific contexts
                    # Sometimes artist names appear without explicit labels
                    words = full_text.split()
                    for i, word in enumerate(words):
                        if word.lower() in ['by', 'artist', 'creator', 'illustrator']:
                            # Look at the next few words for potential names
                            for j in range(1, 4):  # Check next 3 words
                                if i + j < len(words):
                                    potential_name = words[i + j]
                                    # Check if it looks like a name (capitalized, alphabetic)
                                    if (potential_name and 
                                        len(potential_name) > 2 and 
                                        potential_name[0].isupper() and 
                                        potential_name.isalpha()):
                                        
                                        # Check if next word is also a name (full name)
                                        if i + j + 1 < len(words):
                                            next_word = words[i + j + 1]
                                            if (next_word and 
                                                len(next_word) > 2 and 
                                                next_word[0].isupper() and 
                                                next_word.isalpha()):
                                                
                                                full_name = f"{potential_name} {next_word}"
                                                
                                                # Filter out common false positives
                                                if (full_name not in ['Little Monsters', 'Gold Key', 'Comic Book'] and
                                                    'Comic' not in full_name and
                                                    'Book' not in full_name):
                                                    
                                                    artist_mentions.append({
                                                        'name': full_name,
                                                        'source_url': href,
                                                        'source_title': title,
                                                        'context': f'after_{word}',
                                                        'query': query
                                                    })
                                                    print(f"        ‚Üí Found potential artist: {full_name}")
                
                # Store all results for analysis
                all_search_results.append({
                    'query': query,
                    'result_index': result_idx,
                    'title': title,
                    'body': body,
                    'url': href
                })
        else:
            print(f"   ‚úó No results found for this query")
        
        # Be respectful with request timing
        time.sleep(2)
        
    except Exception as e:
        print(f"   ‚úó Search failed: {e}")
        continue

print(f"\n{'=' * 60}")
print("ARTIST SEARCH RESULTS ANALYSIS")
print(f"{'=' * 60}")

if artist_mentions:
    print(f"\nüé® Found {len(artist_mentions)} potential artist mentions:")
    
    # Count frequency of each artist name
    artist_counts = {}
    for mention in artist_mentions:
        name = mention['name']
        if name in artist_counts:
            artist_counts[name] += 1
        else:
            artist_counts[name] = 1
    
    # Sort by frequency
    sorted_artists = sorted(artist_counts.items(), key=lambda x: x[1], reverse=True)
    
    print(f"\nArtist candidates ranked by frequency:")
    for rank, (name, count) in enumerate(sorted_artists, 1):
        print(f"  {rank}. {name} - {count} mention(s)")
        
        # Show sources for this artist
        sources = [m['source_url'] for m in artist_mentions if m['name'] == name]
        unique_sources = list(set(sources))
        print(f"     Found in {len(unique_sources)} unique source(s)")
        
        # Show context information
        contexts = [m['context'] for m in artist_mentions if m['name'] == name]
        unique_contexts = list(set(contexts))
        print(f"     Contexts: {', '.join(unique_contexts)}")
        
        # Show first source as example
        if unique_sources:
            print(f"     Example source: {unique_sources[0][:60]}...")
    
    # Determine most likely artist
    if sorted_artists:
        top_artist, top_count = sorted_artists[0]
        
        # Get detailed info about top candidate
        top_mentions = [m for m in artist_mentions if m['name'] == top_artist]
        top_sources = list(set([m['source_url'] for m in top_mentions]))
        
        print(f"\n*** MOST LIKELY LITTLE MONSTERS ARTIST: {top_artist} ***")
        print(f"*** CONFIDENCE: {top_count} mention(s) across {len(top_sources)} source(s) ***")
        
        final_result = {
            'status': 'SUCCESS',
            'artist_identified': top_artist,
            'confidence_score': top_count,
            'sources_count': len(top_sources),
            'total_candidates': len(artist_counts),
            'search_method': 'Web search with pattern matching'
        }
    else:
        final_result = {
            'status': 'INCONCLUSIVE',
            'reason': 'Multiple candidates found but no clear consensus'
        }
else:
    print(f"\n‚ùå NO ARTIST INFORMATION FOUND")
    print(f"\nThis could indicate:")
    print(f"1. Little Monsters comic artist information is not widely available online")
    print(f"2. Artist information may be in archives or specialized databases")
    print(f"3. The comic may have had multiple artists or uncredited work")
    print(f"4. Different search terms or approaches may be needed")
    
    final_result = {
        'status': 'NO_RESULTS',
        'reason': 'No artist information found in web searches',
        'searches_conducted': len(search_queries),
        'total_results_found': len(all_search_results)
    }

# Save comprehensive results
search_data = {
    'comic_title': 'Little Monsters',
    'search_method': 'Multi-backend web search with pattern matching',
    'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'search_queries': search_queries,
    'total_search_results': len(all_search_results),
    'artist_mentions': artist_mentions,
    'all_search_results': all_search_results,
    'final_result': final_result
}

results_file = os.path.join(workspace_dir, 'little_monsters_comprehensive_search.json')
with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(search_data, f, indent=2, ensure_ascii=False)

print(f"\n‚úì Comprehensive search results saved to: {results_file}")

# Also create a summary file for easy reference
summary = {
    'comic_title': 'Little Monsters',
    'search_date': time.strftime('%Y-%m-%d'),
    'final_result': final_result,
    'top_candidates': sorted_artists[:3] if artist_mentions else [],
    'search_summary': {
        'queries_used': len(search_queries),
        'total_results': len(all_search_results),
        'artist_mentions_found': len(artist_mentions),
        'unique_artists': len(set([m['name'] for m in artist_mentions])) if artist_mentions else 0
    }
}

summary_file = os.path.join(workspace_dir, 'little_monsters_search_summary.json')
with open(summary_file, 'w', encoding='utf-8') as f:
    json.dump(summary, f, indent=2, ensure_ascii=False)

print(f"‚úì Search summary saved to: {summary_file}")

print(f"\n{'=' * 60}")
print("LITTLE MONSTERS ARTIST SEARCH CONCLUSION")
print(f"{'=' * 60}")

if final_result['status'] == 'SUCCESS':
    print(f"\n‚úÖ ARTIST IDENTIFIED: {final_result['artist_identified']}")
    print(f"Confidence: {final_result['confidence_score']} mentions")
    print(f"Sources: {final_result['sources_count']} unique source(s)")
    print(f"Total candidates evaluated: {final_result['total_candidates']}")
    print(f"Method: {final_result['search_method']}")
elif final_result['status'] == 'INCONCLUSIVE':
    print(f"\n‚ö†Ô∏è INCONCLUSIVE RESULTS")
    print(f"Reason: {final_result['reason']}")
    if artist_mentions:
        print(f"Found {len(artist_mentions)} artist mentions but no clear consensus")
else:
    print(f"\n‚ùå NO DEFINITIVE ARTIST FOUND")
    print(f"Status: {final_result['status']}")
    print(f"Reason: {final_result['reason']}")
    if 'searches_conducted' in final_result:
        print(f"Searches conducted: {final_result['searches_conducted']}")
    if 'total_results_found' in final_result:
        print(f"Total search results analyzed: {final_result['total_results_found']}")

print(f"\nDetailed results saved to: {results_file}")
print(f"Summary saved to: {summary_file}")
print(f"\nNote: This search used reliable web search services to find")
print(f"actual Little Monsters comic artist information, avoiding the")
print(f"false positives found in previous HTML file analysis attempts.")
```
