# Function to extract potential governor names from text
def extract_governor_names(text):
    """Extract potential governor names using various patterns"""
    potential_names = set()
    
    for pattern in name_patterns:
        matches = re.findall(pattern, text, re.IGNORECASE)
        for match in matches:
            # Clean up the match
            name = match.strip()
            # Filter out common words that aren't names
            exclude_words = ['the', 'of', 'and', 'for', 'in', 'on', 'at', 'to', 'from', 'with', 'by']
            if name and len(name.split()) <= 4 and not any(word.lower() in exclude_words for word in name.split()):
                if len(name) > 3:  # Avoid single letters or very short strings
                    potential_names.add(name)
    
    return list(potential_names)
