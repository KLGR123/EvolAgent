import json

search_queries = [

searcher = DDGS(timeout=20)

all_search_results = []
total_results_found = 0
definitive_connections = []
pioneer_vogel_evidence = []

for i, query in enumerate(search_queries, 1):
    print(f"=== FINAL SEARCH {i}/{len(search_queries)} ===")
    print(f"Query: {query}")
    
    try:
        # Focused search with multiple backends
        results = searcher.text(
            query, 
            max_results=10, 
            page=1, 
            backend=["google", "duckduckgo", "bing", "yahoo"], 
            safesearch="off", 
            region="en-us"
        )
        
        if results:
            print(f"Search successful - found {len(results)} results")
            total_results_found += len(results)
            
            # Store results
            search_result = {
                'query_number': i,
                'query_text': query,
                'results_data': results,
                'organic_count': len(results)
            }
            all_search_results.append(search_result)
            
            print("\nAnalyzing results for definitive Pioneer Pictures + Victor Vogel connection:\n")
            
            for j, result in enumerate(results[:5], 1):  # Check first 5 results
                title = result.get('title', 'No title')
                link = result.get('href', 'No link')
                snippet = result.get('body', 'No snippet')
                
                combined_text = f"{title} {snippet}".lower()
                
                print(f"Result {j}:")
                print(f"Title: {title}")
                print(f"URL: {link}")
                print(f"Snippet: {snippet[:300]}{'...' if len(snippet) > 300 else ''}")
                
                # Look for definitive Pioneer Pictures + Victor Vogel connections
                has_pioneer_pictures = any(term in combined_text for term in ['pioneer pictures', 'pioneer film', 'pioneer studio'])
                has_victor_vogel = any(term in combined_text for term in ['victor vogel', 'vic vogel'])
                has_employment_terms = any(term in combined_text for term in ['worked for', 'employed by', 'staff', 'composer for', 'music director'])
                has_film_terms = any(term in combined_text for term in ['film', 'movie', 'soundtrack', 'score', 'composer'])
                
                # Calculate connection strength
                connection_score = 0
                if has_pioneer_pictures: connection_score += 5
                if has_victor_vogel: connection_score += 3
                if has_employment_terms: connection_score += 2
                if has_film_terms: connection_score += 1
                
                # Highlight findings
                if has_pioneer_pictures:
                    print("*** PIONEER PICTURES MENTIONED ***")
                
                if has_victor_vogel:
                    print("*** VICTOR VOGEL MENTIONED ***")
                
                if has_employment_terms:
                    print("*** EMPLOYMENT/WORK RELATIONSHIP TERMS ***")
                
                if has_film_terms:
                    print("*** FILM INDUSTRY TERMS ***")
                
                if connection_score >= 8:  # High threshold for definitive connection
                    print(f"*** DEFINITIVE CONNECTION FOUND (Score: {connection_score}) ***")
                    definitive_connections.append({
                        'title': title,
                        'url': link,
                        'snippet': snippet,
                        'connection_score': connection_score,
                        'query': query
                    })
                
                elif connection_score >= 5:  # Medium threshold for potential evidence
                    print(f"*** POTENTIAL EVIDENCE (Score: {connection_score}) ***")
                    pioneer_vogel_evidence.append({
                        'title': title,
                        'url': link,
                        'snippet': snippet,
                        'connection_score': connection_score,
                        'query': query
                    })
                
                # Look for specific connection patterns
                connection_patterns = [
                    r'victor vogel.*pioneer pictures',
                    r'pioneer pictures.*victor vogel',
                    r'vic vogel.*pioneer pictures',
                    r'pioneer pictures.*vic vogel',
                    r'vogel.*worked.*pioneer',
                    r'pioneer.*employed.*vogel',
                    r'vogel.*composer.*pioneer'
                ]
                
                for pattern in connection_patterns:
                    if re.search(pattern, combined_text, re.IGNORECASE):
                        print(f"*** CONNECTION PATTERN FOUND: {pattern} ***")
                
                print("-" * 60)

        else:
