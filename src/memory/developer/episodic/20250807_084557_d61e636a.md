import os

story_file_priority = [
    'the_ash_tree_story.txt',
    'the_ash_tree_full_text.txt',

]

for priority_file in story_file_priority:
    if priority_file in workspace_files:
        story_filepath = os.path.join(workspace, priority_file)
        print(f'Attempting to load: {priority_file}')
        
        try:
            with open(story_filepath, 'r', encoding='utf-8') as f:
                content = f.read()
            
            # Look for the story content after metadata
            lines = content.split('\n')
            story_start_idx = 0
            
            # Skip metadata lines (usually marked with === or contain 'Source:', 'Extracted:', etc.)
            for i, line in enumerate(lines):
                if line.strip().startswith('=') or 'Source:' in line or 'Extracted:' in line:
                    continue
                elif len(line.strip()) > 50 and not line.strip().isupper():
                    # Found what looks like story content
                    story_start_idx = i
                    break
            
            # Extract the story content
            story_lines = lines[story_start_idx:]
            story_text = '\n'.join(story_lines).strip()
            
            if len(story_text) > 1000:  # Reasonable story length
                story_source = priority_file
                print(f'✓ Successfully loaded story from {priority_file}')
                print(f'  Story length: {len(story_text):,} characters')
                break
            else:
                print(f'  Story too short: {len(story_text)} characters')
                
        except Exception as e:
            print(f'  Error loading {priority_file}: {str(e)}')

if not story_text:
    print('❌ Could not load story text from existing files')

        print(f'Loading full Gutenberg text: {gutenberg_file}')

            full_text = f.read()
        
        print(f'Full text length: {len(full_text):,} characters')
        
        # Find THE ASH-TREE
        story_start = full_text.find('THE ASH-TREE')
        if story_start != -1:
            print(f'Found "THE ASH-TREE" at position {story_start}')
            
            # Find the end (next story)
            next_story_pos = full_text.find('NUMBER 13', story_start + 100)
            if next_story_pos == -1:
                next_story_pos = full_text.find('THE SCHOOL STORY', story_start + 100)
            
            if next_story_pos != -1:
                raw_story = full_text[story_start:next_story_pos]
                print(f'Raw story extracted: {len(raw_story):,} characters')
                
                # Clean the text properly
                cleaned_lines = []
                for text_line in raw_story.split('\n'):
                    text_line = text_line.strip()
                    if text_line and not text_line.startswith('*') and len(text_line) > 2:
                        cleaned_lines.append(text_line)
                
                story_text = '\n\n'.join(cleaned_lines)
                story_source = 'Extracted from full Gutenberg text'
                print(f'✓ Cleaned story: {len(story_text):,} characters')
                
                # Save the properly extracted story
                clean_story_file = os.path.join(workspace, 'ash_tree_clean_extracted.txt')
                with open(clean_story_file, 'w', encoding='utf-8') as f:
                    f.write('"The Ash Tree" by M.R. James\n')
                    f.write(f'Source: {story_source}\n')
                    f.write(f'Extracted: {time.strftime("%Y-%m-%d %H:%M:%S")}\n')
                    f.write('=' * 80 + '\n\n')
                    f.write(story_text)
                
                print(f'✓ Clean story saved to: {clean_story_file}')

# Create comprehensive search categories for Suffolk location identification
search_categories = {
    'suffolk_specific': {
        'keywords': ['suffolk', 'bury st edmunds', 'bury', 'ipswich', 'felixstowe', 'lowestoft', 
                    'newmarket', 'sudbury', 'haverhill', 'aldeburgh', 'southwold', 'woodbridge',

    },
    'estate_properties': {
        'keywords': ['hall', 'manor', 'house', 'court', 'grange', 'priory', 'abbey', 'rectory',
                    'vicarage', 'castle', 'tower', 'lodge', 'mansion', 'estate', 'park'],
        'description': 'Estate and property types'

    'ecclesiastical': {

}

story_lower = story_text.lower()
analysis_results = {}
all_findings = []

print('Analyzing story text for location and historical clues...')
print()

# Analyze each category
for category_name, category_data in search_categories.items():
    keywords = category_data['keywords']
    description = category_data['description']
    category_findings = {}
    
    print(f'Searching for {description.lower()}...')
    
    for keyword in keywords:
        if keyword in story_lower:
            # Find all occurrences
            positions = []
            start = 0
            while True:
                pos = story_lower.find(keyword, start)
                if pos == -1:
                    break
                positions.append(pos)
                start = pos + 1
            
            # Extract contexts
            contexts = []
            for pos in positions:
                context_start = max(0, pos - 200)
                context_end = min(len(story_text), pos + 200)
                context = story_text[context_start:context_end].replace('\n', ' ').strip()
                contexts.append({
                    'position': pos,
                    'context': context
                })
            
            category_findings[keyword] = {
                'count': len(positions),
                'contexts': contexts
            }
            
            # Add to all findings
            for ctx in contexts:
                all_findings.append({
                    'category': category_name,
                    'keyword': keyword,
                    'position': ctx['position'],
                    'context': ctx['context']
                })
    
    if category_findings:
        analysis_results[category_name] = category_findings
        print(f'  ✓ Found {len(category_findings)} relevant terms')
        # Show the terms found
        for term in category_findings.keys():
            count = category_findings[term]['count']
            print(f'    - "{term}": {count} occurrence(s)')
    else:
        print(f'  - No matches found')
    print()

print('=== DETAILED FINDINGS BY CATEGORY ===')

for category_name, findings in analysis_results.items():
    category_desc = search_categories[category_name]['description']
    print(f'{category_desc.upper()}:')
    
    for keyword, data in findings.items():
        print(f'  • "{keyword}" - {data["count"]} occurrence(s)')
        
        # Show context for each occurrence
        for i, context_data in enumerate(data['contexts'], 1):
            context = context_data['context']
            print(f'    Context {i}: ...{context}...')
        print()

print('=== REGEX SEARCH FOR SPECIFIC PLACE NAMES ===')
print()

# Enhanced regex patterns for identifying specific places
place_patterns = [
    (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Hall|Manor|House|Court|Grange|Priory|Abbey)', 'Estate/Manor names'),
    (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Church|Cathedral|Chapel)', 'Religious buildings'),
    (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Village|Parish|Town)', 'Settlements'),
    (r'the\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(estate|property|lands)', 'Property references'),
    (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+in\s+(Suffolk|Norfolk|Essex|Cambridgeshire)', 'County-specific places'),
    (r'Sir\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)', 'Named individuals'),
    (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(family|lineage|ancestry)', 'Family names')
]

place_matches = []

for pattern, description in place_patterns:
    matches = list(re.finditer(pattern, story_text, re.IGNORECASE))
    if matches:
        print(f'{description}: {len(matches)} match(es)')
        
        for match in matches:
            if len(match.groups()) >= 2:
                place_name = match.group(1).strip()
                place_type = match.group(2).strip()
                full_match = match.group(0)
            else:
                place_name = match.group(1).strip()
                place_type = 'Individual/Family'
                full_match = match.group(0)
            
            position = match.start()
            
            # Extract context
            context_start = max(0, position - 150)
            context_end = min(len(story_text), position + 150)
            context = story_text[context_start:context_end].replace('\n', ' ').strip()
            
            place_info = {
                'place_name': place_name,
                'place_type': place_type,
                'full_match': full_match,
                'pattern_description': description,
                'context': context,
                'position': position
            }
            
            place_matches.append(place_info)
            
            print(f'  → "{place_name}" ({place_type})')
            print(f'    Full match: "{full_match}"')
            print(f'    Context: ...{context}...')
            print()
