# Enhanced key terms with more comprehensive coverage
enhanced_key_terms = {

        '17th century', '1600s', '1650s', '1660s', '1670s', '1680s', '1690s',

comprehensive_analysis = {

# Process all HTML files
for i, filename in enumerate(html_files, 1):
    filepath = os.path.join('workspace', filename)
    print(f'\nProcessing {i}/{len(html_files)}: {filename}')
    print('-' * 70)
    
    try:
        with open(filepath, 'r', encoding='utf-8') as f:
            html_content = f.read()

        soup = BeautifulSoup(html_content, 'html.parser')
        
        # Remove script and style tags
        for element in soup(['script', 'style', 'nav', 'header', 'footer']):
            element.decompose()
        
        # Get clean text content
        page_text = soup.get_text().lower()
        
        # Initialize file analysis
        file_analysis = {
            'filename': filename,
            'file_size': len(html_content),
            'text_length': len(page_text),
            'found_terms': {},
            'relevance_score': 0,
            'key_passages': [],
            'priority_level': 'low'
        }
        
        print(f'File size: {len(html_content):,} chars, Text: {len(page_text):,} chars')
        
        # Search for terms in each category
        total_relevance = 0
        category_scores = {}
        
        for category, terms in enhanced_key_terms.items():
            found_in_category = []
            category_score = 0
            
            for term in terms:
                if term in page_text:
                    found_in_category.append(term)
                    # Enhanced scoring system
                    if category == 'irish_chronicles':
                        category_score += 8  # Highest priority
                    elif category == 'suffolk_locations':
                        category_score += 6  # High priority for Suffolk
                    elif category == 'spectral_phenomena':
                        category_score += 5
                    elif category == 'supernatural_events':
                        category_score += 3
                    elif category == 'time_periods':
                        category_score += 2
                    elif category == 'historical_indicators':
                        category_score += 1

                file_analysis['found_terms'][category] = found_in_category
                category_scores[category] = category_score
                total_relevance += category_score
                print(f'  ✅ {category}: {found_in_category[:4]} (Score: {category_score})')
        
        file_analysis['relevance_score'] = total_relevance
        file_analysis['category_scores'] = category_scores
        
        # Special handling for high-priority combinations
        has_irish_chronicle = 'irish_chronicles' in file_analysis['found_terms']
        has_suffolk = 'suffolk_locations' in file_analysis['found_terms']
        has_spectral = 'spectral_phenomena' in file_analysis['found_terms']
        
        # Determine priority level
        if has_irish_chronicle and has_suffolk:
            file_analysis['priority_level'] = 'critical'  # Both Irish and Suffolk!
            print('  🚨 CRITICAL: Contains both Irish chronicles AND Suffolk locations!')
        elif has_irish_chronicle and has_spectral:
            file_analysis['priority_level'] = 'high'  # Irish chronicle + spectral
            print('  🎯 HIGH: Contains Irish chronicles AND spectral phenomena!')
        elif has_irish_chronicle:
            file_analysis['priority_level'] = 'high'  # Irish chronicle found
            print('  📚 HIGH: Contains Irish chronicle references!')
        elif has_suffolk and has_spectral:
            file_analysis['priority_level'] = 'medium'  # Suffolk + spectral
            print('  ⭐ MEDIUM: Contains Suffolk AND spectral phenomena!')
        elif total_relevance >= 10:
            file_analysis['priority_level'] = 'medium'
            print('  📊 MEDIUM: High relevance score!')
        
        print(f'  📊 Total relevance: {total_relevance}, Priority: {file_analysis["priority_level"]}')
        
        # Extract key passages for high-relevance files
        if total_relevance >= 8:
            sentences = page_text.split('.')
            relevant_passages = []
            
            for sentence in sentences:
                sentence = sentence.strip()
                if 30 < len(sentence) < 400:  # Reasonable sentence length
                    # Count relevant terms in this sentence
                    term_matches = []
                    sentence_score = 0
                    
                    for category, terms in enhanced_key_terms.items():
                        for term in terms:
                            if term in sentence:
                                term_matches.append(term)
                                # Score based on category importance
                                if category == 'irish_chronicles':
                                    sentence_score += 5
                                elif category == 'suffolk_locations':
                                    sentence_score += 4
                                elif category == 'spectral_phenomena':
                                    sentence_score += 3
                                else:
                                    sentence_score += 1
                    
                    # Keep sentences with multiple relevant terms
                    if sentence_score >= 5:
                        relevant_passages.append({
                            'text': sentence[:250] + '...' if len(sentence) > 250 else sentence,
                            'score': sentence_score,
                            'matched_terms': list(set(term_matches))[:6]  # Unique terms, max 6
                        })
            
            # Sort by score and keep top passages
            relevant_passages.sort(key=lambda x: x['score'], reverse=True)
            file_analysis['key_passages'] = relevant_passages[:3]
            
            if relevant_passages:
                print(f'  📝 Key passages: {len(relevant_passages)} found')
                for j, passage in enumerate(relevant_passages[:2], 1):
                    print(f'     {j}. "{passage["text"][:100]}..." (Score: {passage["score"]})')

        # Store analysis results
        comprehensive_analysis['files_analyzed'].append(file_analysis)
        comprehensive_analysis['total_files_processed'] += 1

        if 'irish_chronicles' in file_analysis['found_terms']:
            comprehensive_analysis['irish_chronicle_references'].extend(
                file_analysis['found_terms']['irish_chronicles']

                file_analysis['found_terms']['suffolk_locations']

                file_analysis['found_terms']['suffolk_locations']
            )

            comprehensive_analysis['spectral_army_mentions'].extend(
                file_analysis['found_terms']['spectral_phenomena']
            )
        
        if 'supernatural_events' in file_analysis['found_terms']:
            comprehensive_analysis['supernatural_events'].extend(
                file_analysis['found_terms']['supernatural_events']
            )
        
        # Mark as potential document if high relevance
        if total_relevance >= 12 or file_analysis['priority_level'] in ['critical', 'high']:
