import os
import requests
import json
import time
from urllib.parse import quote_plus
from bs4 import BeautifulSoup
from collections import Counter

print('=== SEARCH FOR 17TH CENTURY IRISH SPECTRAL ARMY AND SUFFOLK SUPERNATURAL EVENTS ===')
print('Objective: Find historical document recording both Irish spectral army apparition and Suffolk supernatural events\n')

# Ensure workspace directory exists
os.makedirs('workspace', exist_ok=True)

# Initialize comprehensive search results
search_results = {
    'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'objective': 'Find historical document with 17th century Irish spectral army and Suffolk supernatural events',
    'target_chronicles': [
        'Chronicon Scotorum',
        'Annals of the Four Masters', 
        'Geoffrey Keating works',
        'Irish historical chronicles'
    ],
    'search_phases': [],
    'findings': [],
    'document_candidates': [],
    'suffolk_locations': [],
    'final_analysis': {}
}

print('TARGET DOCUMENT CHARACTERISTICS:')
print('• Contains 17th century spectral army apparition in Ireland')
print('• Records supernatural events in Suffolk')
print('• Likely chronicles: Chronicon Scotorum, Annals of Four Masters, Geoffrey Keating')
print('• May mention unusual arachnid/supernatural events')
print('• Should identify specific Suffolk location')
print()

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.9',
    'Connection': 'keep-alive'
}

print('=== PHASE 1: IRISH CHRONICLES WITH SPECTRAL ARMY REFERENCES ===')
print('=' * 70)

# Phase 1: Search for Irish chronicles mentioning spectral armies
phase1_queries = [
    '"Chronicon Scotorum" spectral army apparition 17th century Ireland',
    '"Annals of the Four Masters" phantom army Ireland Suffolk supernatural',
    'Geoffrey Keating spectral army Ireland Suffolk chronicle',
    'Irish chronicles 17th century phantom army apparition Suffolk',
    '"spectral army" Ireland 1600s chronicle Suffolk supernatural events'
]

print(f'Executing {len(phase1_queries)} searches for Irish chronicles:')
for i, query in enumerate(phase1_queries, 1):
    print(f'  {i}. {query}')

search_results['search_phases'].append({
    'phase': 'Irish Chronicles Search',
    'queries': phase1_queries,
    'results': []
})

for i, query in enumerate(phase1_queries, 1):
    print(f'\nPhase 1 Search {i}/{len(phase1_queries)}: {query}')
    print('-' * 60)
    
    try:
        # Use DuckDuckGo for academic searches
        search_url = f'https://html.duckduckgo.com/html/?q={quote_plus(query)}'
        print(f'URL: {search_url}')
        
        response = requests.get(search_url, headers=headers, timeout=30)
        print(f'Status: {response.status_code}')
        
        if response.status_code == 200:
            # Save HTML for reference
            filename = f'phase1_search_{i}_{query[:40].replace(" ", "_").replace('"', "")}.html'
            filepath = os.path.join('workspace', filename)
            
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(response.text)
            
            print(f'Saved: {filepath}')
            
            # Parse and analyze results
            soup = BeautifulSoup(response.text, 'html.parser')
            page_text = soup.get_text().lower()
            
            # Key terms for Irish chronicles and spectral events
            key_terms = {
                'chronicon': 5,
                'scotorum': 5,
                'annals': 4,
                'four masters': 5,
                'keating': 4,
                'spectral': 5,
                'phantom': 4,
                'apparition': 4,
                'army': 4,
                'ireland': 3,
                'suffolk': 5,
                'supernatural': 3,
                '17th century': 4,
                '1600s': 3,
                'chronicle': 3,
                'arachnid': 4,
                'spider': 3
            }
            
            found_terms = []
            relevance_score = 0
            
            for term, weight in key_terms.items():
                if term in page_text:
                    found_terms.append(term)
                    relevance_score += weight
            
            print(f'Relevance score: {relevance_score}')
            print(f'Found terms: {", ".join(found_terms[:8])}')
            
            # Look for specific chronicle names and Suffolk locations
            chronicle_matches = []
            suffolk_locations = []
            
            chronicles = ['chronicon scotorum', 'annals of the four masters', 'keating', 'foras feasa']
            for chronicle in chronicles:
                if chronicle in page_text:
                    chronicle_matches.append(chronicle)
                    print(f'  ✓ Chronicle found: {chronicle}')
            
            # Look for Suffolk place names
            suffolk_places = ['ipswich', 'bury st edmunds', 'lowestoft', 'felixstowe', 'sudbury', 'haverhill', 'newmarket', 'woodbridge', 'aldeburgh', 'southwold']
            for place in suffolk_places:
                if place in page_text:
                    suffolk_locations.append(place)
                    print(f'  ✓ Suffolk location found: {place}')
            
            # Store finding
            finding = {
                'phase': 'Irish Chronicles',
                'query': query,
                'relevance_score': relevance_score,
                'found_terms': found_terms,
                'chronicle_matches': chronicle_matches,
                'suffolk_locations': suffolk_locations,
                'has_spectral': any(term in found_terms for term in ['spectral', 'phantom', 'apparition']),
                'has_suffolk': 'suffolk' in found_terms,
                'html_file': filepath
            }
            
            search_results['findings'].append(finding)
            search_results['search_phases'][0]['results'].append(finding)
            
            # If high relevance, extract detailed text
            if relevance_score >= 15:
                print('🎯 HIGH RELEVANCE - Extracting detailed information...')
                
                # Extract relevant sentences
                sentences = page_text.split('.')
                relevant_sentences = []
                
                for sentence in sentences:
                    if any(term in sentence for term in ['spectral', 'phantom', 'army', 'suffolk', 'chronicle']):
                        if 20 < len(sentence.strip()) < 300:
                            relevant_sentences.append(sentence.strip())
                
                if relevant_sentences:
                    print('Key sentences found:')
                    for j, sentence in enumerate(relevant_sentences[:3], 1):
                        print(f'  {j}. {sentence[:150]}...')
                    
                    finding['key_sentences'] = relevant_sentences[:5]
            
            # Update Suffolk locations list
            search_results['suffolk_locations'].extend(suffolk_locations)
            
        else:
            print(f'Failed with status {response.status_code}')
    
    except Exception as e:
        print(f'Error: {str(e)}')
    
    time.sleep(3)  # Rate limiting

print('\n' + '=' * 80)
print('PHASE 2: SUFFOLK SUPERNATURAL EVENTS AND ARACHNID PHENOMENA')
print('=' * 80)

# Phase 2: Search for Suffolk supernatural events, particularly arachnid-related
phase2_queries = [
    'Suffolk supernatural events 17th century arachnid spider phenomena',
    'Suffolk unusual spider events historical chronicle Irish connection',
    'Suffolk paranormal arachnid 1600s Irish chronicles spectral army',
    '"Suffolk" supernatural spider arachnid historical document Ireland',
    'Suffolk county supernatural events chronicle Irish phantom army'
]

print(f'Executing {len(phase2_queries)} searches for Suffolk supernatural events:')
for i, query in enumerate(phase2_queries, 1):
    print(f'  {i}. {query}')

search_results['search_phases'].append({
    'phase': 'Suffolk Supernatural Search',
    'queries': phase2_queries,
    'results': []
})

for i, query in enumerate(phase2_queries, 1):
    print(f'\nPhase 2 Search {i}/{len(phase2_queries)}: {query}')
    print('-' * 60)
    
    try:
        search_url = f'https://html.duckduckgo.com/html/?q={quote_plus(query)}'
        print(f'URL: {search_url}')
        
        response = requests.get(search_url, headers=headers, timeout=30)
        print(f'Status: {response.status_code}')
        
        if response.status_code == 200:
            filename = f'phase2_search_{i}_{query[:40].replace(" ", "_").replace('"', "")}.html'
            filepath = os.path.join('workspace', filename)
            
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(response.text)
            
            print(f'Saved: {filepath}')
            
            soup = BeautifulSoup(response.text, 'html.parser')
            page_text = soup.get_text().lower()
            
            # Terms specific to Suffolk supernatural events
            suffolk_terms = {
                'suffolk': 5,
                'arachnid': 5,
                'spider': 4,
                'supernatural': 4,
                'paranormal': 3,
                'unusual': 2,
                'phenomena': 3,
                'chronicle': 4,
                'historical': 2,
                'ireland': 3,
                'irish': 3,
                'spectral': 4,
                'phantom': 4,
                '17th century': 4,
                '1600s': 3
            }
            
            found_terms = []
            relevance_score = 0
            
            for term, weight in suffolk_terms.items():
                if term in page_text:
                    found_terms.append(term)
                    relevance_score += weight
            
            print(f'Relevance score: {relevance_score}')
            print(f'Found terms: {", ".join(found_terms[:8])}')
            
            # Look for specific Suffolk locations mentioned
            suffolk_locations = []
            suffolk_places = ['ipswich', 'bury st edmunds', 'lowestoft', 'felixstowe', 'sudbury', 'haverhill', 'newmarket', 'woodbridge', 'aldeburgh', 'southwold', 'beccles', 'eye', 'framlingham', 'stowmarket']
            
            for place in suffolk_places:
                if place in page_text:
                    suffolk_locations.append(place)
                    print(f'  ✓ Suffolk location: {place}')
            
            # Store finding
            finding = {
                'phase': 'Suffolk Supernatural',
                'query': query,
                'relevance_score': relevance_score,
                'found_terms': found_terms,
                'suffolk_locations': suffolk_locations,
                'has_arachnid': any(term in found_terms for term in ['arachnid', 'spider']),
                'has_supernatural': any(term in found_terms for term in ['supernatural', 'paranormal', 'phenomena']),
                'has_irish_connection': any(term in found_terms for term in ['ireland', 'irish', 'spectral', 'phantom']),
                'html_file': filepath
            }
            
            search_results['findings'].append(finding)
            search_results['search_phases'][1]['results'].append(finding)
            search_results['suffolk_locations'].extend(suffolk_locations)
            
        else:
            print(f'Failed with status {response.status_code}')
    
    except Exception as e:
        print(f'Error: {str(e)}')
    
    time.sleep(3)

print('\n' + '=' * 80)
print('COMPREHENSIVE ANALYSIS OF SEARCH RESULTS')
print('=' * 80)

total_findings = len(search_results['findings'])
print(f'Total search results collected: {total_findings}')

if search_results['findings']:
    # Sort by relevance score
    search_results['findings'].sort(key=lambda x: x['relevance_score'], reverse=True)
    
    print('\n📊 RELEVANCE ANALYSIS:')
    print('-' * 40)
    
    high_relevance = [f for f in search_results['findings'] if f['relevance_score'] >= 15]
    moderate_relevance = [f for f in search_results['findings'] if 8 <= f['relevance_score'] < 15]
    
    print(f'High relevance results (15+ points): {len(high_relevance)}')
    print(f'Moderate relevance results (8-14 points): {len(moderate_relevance)}')
    
    if high_relevance:
        print('\n🎯 HIGH RELEVANCE FINDINGS:')
        for i, finding in enumerate(high_relevance, 1):
            print(f'\n{i}. Phase: {finding["phase"]}')
            print(f'   Query: {finding["query"]}')
            print(f'   Score: {finding["relevance_score"]}')
            print(f'   Terms: {", ".join(finding["found_terms"][:6])}')
            if finding.get('chronicle_matches'):
                print(f'   Chronicles: {", ".join(finding["chronicle_matches"])}')
            if finding.get('suffolk_locations'):
                print(f'   Suffolk locations: {", ".join(finding["suffolk_locations"])}')
            if finding.get('key_sentences'):
                print(f'   Key info: {finding["key_sentences"][0][:120]}...')
    
    # Analyze chronicle mentions
    all_chronicles = []
    for finding in search_results['findings']:
        if finding.get('chronicle_matches'):
            all_chronicles.extend(finding['chronicle_matches'])
    
    if all_chronicles:
        chronicle_counts = Counter(all_chronicles)
        print('\n📚 CHRONICLE ANALYSIS:')
        print('-' * 30)
        for chronicle, count in chronicle_counts.most_common():
            print(f'  • {chronicle}: {count} mentions')
        
        search_results['final_analysis']['most_mentioned_chronicle'] = chronicle_counts.most_common(1)[0][0] if chronicle_counts else None
    
    # Analyze Suffolk locations
    all_suffolk = list(set(search_results['suffolk_locations']))
    if all_suffolk:
        suffolk_counts = Counter(search_results['suffolk_locations'])
        print('\n🏘️ SUFFOLK LOCATIONS ANALYSIS:')
        print('-' * 35)
        for location, count in suffolk_counts.most_common(10):
            print(f'  • {location}: {count} mentions')
        
        search_results['final_analysis']['top_suffolk_location'] = suffolk_counts.most_common(1)[0][0] if suffolk_counts else None
    
    # Evidence compilation
    evidence = {
        'irish_chronicle_found': any(f.get('chronicle_matches') for f in search_results['findings']),
        'spectral_army_referenced': any(f.get('has_spectral') for f in search_results['findings']),
        'suffolk_connection': any(f.get('has_suffolk') for f in search_results['findings']),
        'supernatural_events': any(f.get('has_supernatural') for f in search_results['findings']),
        'arachnid_phenomena': any(f.get('has_arachnid') for f in search_results['findings']),
        'suffolk_locations_identified': len(all_suffolk) > 0
    }
    
    print('\n🔍 EVIDENCE SUMMARY:')
    print('-' * 30)
    for key, found in evidence.items():
        status = '✅' if found else '❌'
        print(f'{status} {key.replace("_", " ").title()}: {found}')
    
    search_results['final_analysis']['evidence'] = evidence
    
    # Calculate confidence
    confirmed_evidence = sum(evidence.values())
    confidence = (confirmed_evidence / len(evidence)) * 100
    print(f'\n📈 OVERALL CONFIDENCE: {confidence:.1f}% ({confirmed_evidence}/{len(evidence)} criteria met)')
    search_results['final_analysis']['confidence_percentage'] = confidence

else:
    print('❌ No search results collected')

# Save comprehensive results
results_file = os.path.join('workspace', 'irish_spectral_army_suffolk_supernatural_search.json')
with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(search_results, f, indent=2, ensure_ascii=False)

print(f'\n💾 COMPREHENSIVE RESULTS SAVED TO: {results_file}')

        print(f'\n🎯 PRIMARY CHRONICLE: {search_results["final_analysis"]["most_mentioned_chronicle"]}')
    
    if search_results['final_analysis'].get('top_suffolk_location'):
        print(f'🏘️ TOP SUFFOLK LOCATION: {search_results["final_analysis"]["top_suffolk_location"]}')
        print('\n📋 NEXT RESEARCH STEPS:')
        print(f'   1. Research artists from {search_results["final_analysis"]["top_suffolk_location"]}')

        print('   3. Verify chronicle references in saved HTML files')

else:

print('\n🔍 FILES TO REVIEW:')
for i, finding in enumerate(search_results['findings'][:5], 1):
    if finding.get('html_file') and finding['relevance_score'] >= 8:
        print(f'   {i}. {finding["html_file"]} (Score: {finding["relevance_score"]})')

print('\n=== SEARCH FOR HISTORICAL DOCUMENT COMPLETE ===')
