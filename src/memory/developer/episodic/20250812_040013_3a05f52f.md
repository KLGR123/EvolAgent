### Development Step 10: Finding Gold Medalists’ Birth Countries at the Finland/Norway-Excluded Cortina FIS Nordic Championships

**Description**: Search for information about FIS Nordic World Ski Championships held in Cortina d'Ampezzo that did not include participants from Finland or Norway. Focus on identifying which specific year this championship took place, then find the complete list of gold medalists from that event. Extract the birth countries of all gold medalists to identify which country produced a gold medalist at this particular championship.

**Use Cases**:
- Sports historians automating the retrieval of niche championship data to document the 1941 Cortina d’Ampezzo event that uniquely excluded Finland and Norway participants
- Data journalists extracting gold medalists’ birth countries to illustrate mid-20th-century shifts in Nordic skiing dominance for a feature article
- National skiing federations analyzing past medal distributions to benchmark athlete development strategies against non-traditional podium countries
- Broadcast production teams generating on-air graphics for live coverage by automatically scraping unique event narratives and champion metadata
- Academic researchers building a wartime sports dataset for statistical modeling by programmatically harvesting archived results and participant anomalies
- Genealogists compiling athlete family histories by extracting birthplace information of medalists from historical sports records
- Sports memorabilia authentication services verifying champion identities and nationalities from primary source tables to ensure auction catalog accuracy
- Regional tourism boards profiling lesser-known gold medalists and their birthplaces to design heritage skiing tours in Cortina d’Ampezzo and medalists’ home regions

```
import os
import json
from bs4 import BeautifulSoup
import re

# Create workspace directory if it doesn't exist
if not os.path.exists('workspace'):
    os.makedirs('workspace')

print("=== FIS NORDIC WORLD SKI CHAMPIONSHIPS: WORKING WITH COLLECTED DATA ===")
print("Objective: Find Cortina d'Ampezzo championship WITHOUT Finland or Norway participants")
print("Then: Extract all gold medalists and their birth countries\n")

# First, let's check what files we have in workspace
print("Checking workspace contents...")
if os.path.exists('workspace'):
    workspace_files = os.listdir('workspace')
    print(f"Found {len(workspace_files)} files in workspace:")
    for file in workspace_files:
        print(f"  - {file}")
else:
    print("No workspace directory found")
    exit()

# Look for the saved HTML files from previous attempts
html_files = [f for f in workspace_files if f.endswith('.html')]
print(f"\nFound {len(html_files)} HTML files:")
for file in html_files:
    print(f"  - {file}")

# Check for any analysis files
analysis_files = [f for f in workspace_files if f.endswith('.json') or f.endswith('.txt')]
print(f"\nFound {len(analysis_files)} analysis files:")
for file in analysis_files:
    print(f"  - {file}")

# Let's start by examining the 1941 FIS Nordic World Ski Championships file
# Based on HISTORY, this is likely our target since it's specifically about a championship in Cortina
championship_1941_file = None
for file in html_files:
    if '1941' in file and 'FIS' in file:
        championship_1941_file = f'workspace/{file}'
        break

if championship_1941_file and os.path.exists(championship_1941_file):
    print(f"\n=== ANALYZING 1941 FIS NORDIC WORLD SKI CHAMPIONSHIPS ===\n")
    print(f"File: {championship_1941_file}")
    
    with open(championship_1941_file, 'r', encoding='utf-8') as f:
        html_content = f.read()
    
    soup = BeautifulSoup(html_content, 'html.parser')
    
    # Get page title
    title = soup.find('title')
    title_text = title.get_text().strip() if title else 'No title found'
    print(f"Title: {title_text}")
    
    # Get all text content
    page_text = soup.get_text()
    
    # Check for Finland and Norway mentions
    finland_mentions = page_text.lower().count('finland')
    norway_mentions = page_text.lower().count('norway')
    cortina_mentions = page_text.lower().count('cortina')
    
    print(f"Cortina mentions: {cortina_mentions}")
    print(f"Finland mentions: {finland_mentions}")
    print(f"Norway mentions: {norway_mentions}")
    
    # Look for specific evidence that Finland/Norway did NOT participate
    print("\n=== SEARCHING FOR PARTICIPATION EVIDENCE ===\n")
    
    # Find all tables - these likely contain results
    tables = soup.find_all('table')
    print(f"Found {len(tables)} tables")
    
    # Analyze each table for medal/result information
    medal_tables = []
    for i, table in enumerate(tables):
        # Get table text safely
        table_content = table.get_text()
        table_lower = table_content.lower()
        
        # Check if this table contains medal/result information
        medal_keywords = ['gold', 'silver', 'bronze', 'medal', 'winner', 'champion', '1st', '2nd', '3rd']
        has_medal_info = any(keyword in table_lower for keyword in medal_keywords)
        
        # Check for country information
        has_countries = any(country in table_lower for country in ['germany', 'italy', 'austria', 'sweden', 'switzerland'])
        
        # Check specifically for Finland and Norway
        table_has_finland = 'finland' in table_lower
        table_has_norway = 'norway' in table_lower
        
        if has_medal_info or has_countries:
            print(f"\nTable {i+1}:")
            print(f"  Has medal info: {has_medal_info}")
            print(f"  Has countries: {has_countries}")
            print(f"  Contains Finland: {table_has_finland}")
            print(f"  Contains Norway: {table_has_norway}")
            
            # Get table headers to understand structure
            headers = table.find_all('th')
            if headers:
                header_texts = [th.get_text().strip() for th in headers]
                print(f"  Headers: {header_texts[:5]}")  # Show first 5 headers
            
            # Get first few rows to see data structure
            rows = table.find_all('tr')
            print(f"  Total rows: {len(rows)}")
            
            if len(rows) > 1:  # Has data beyond headers
                # Show first data row
                first_data_row = rows[1] if len(rows) > 1 else None
                if first_data_row:
                    cells = first_data_row.find_all(['td', 'th'])
                    cell_texts = [cell.get_text().strip() for cell in cells]
                    print(f"  Sample row: {cell_texts[:5]}")  # First 5 columns
            
            medal_tables.append({
                'index': i,
                'has_medal_info': has_medal_info,
                'has_countries': has_countries,
                'has_finland': table_has_finland,
                'has_norway': table_has_norway,
                'table_element': table,
                'row_count': len(rows)
            })
    
    print(f"\nFound {len(medal_tables)} tables with medal/country information")
    
    # Now let's extract gold medalists from the most promising tables
    print("\n=== EXTRACTING GOLD MEDALISTS ===\n")
    
    all_gold_medalists = []
    
    for table_info in medal_tables:
        if table_info['has_medal_info']:
            print(f"Analyzing Table {table_info['index'] + 1} for gold medalists...")
            
            table = table_info['table_element']
            rows = table.find_all('tr')
            
            # Try to identify gold medal winners
            for row_idx, row in enumerate(rows):
                cells = row.find_all(['td', 'th'])
                if len(cells) >= 2:  # Need at least 2 columns
                    cell_texts = [cell.get_text().strip() for cell in cells]
                    row_text = ' '.join(cell_texts).lower()
                    
                    # Look for gold medal indicators
                    if 'gold' in row_text or '1st' in row_text or (row_idx == 1 and table_info['has_medal_info']):
                        print(f"  Potential gold medalist row: {cell_texts}")
                        
                        # Try to extract name and country
                        # Look for patterns like "Name (Country)" or separate columns
                        name = None
                        country = None
                        
                        for cell_text in cell_texts:
                            # Check if this cell contains a country in parentheses
                            country_match = re.search(r'\(([^)]+)\)', cell_text)
                            if country_match:
                                country = country_match.group(1)
                                name = cell_text.replace(f'({country})', '').strip()
                                break
                        
                        # If no parentheses pattern, try to identify name and country from separate cells
                        if not name and len(cell_texts) >= 2:
                            # Usually name is in first or second column, country might be in another
                            potential_name = cell_texts[0] if cell_texts[0] and not cell_texts[0].isdigit() else cell_texts[1] if len(cell_texts) > 1 else None
                            
                            # Look for country names in the cells
                            countries = ['germany', 'italy', 'austria', 'sweden', 'switzerland', 'czechoslovakia', 'hungary', 'poland']
                            for cell in cell_texts:
                                for country_name in countries:
                                    if country_name in cell.lower():
                                        country = cell
                                        name = potential_name
                                        break
                                if country:
                                    break
                        
                        if name or country:
                            gold_medalist = {
                                'name': name or 'Name not clearly identified',
                                'country': country or 'Country not clearly identified',
                                'event': 'Unknown event',  # Would need more context
                                'table_index': table_info['index'] + 1,
                                'row_data': cell_texts
                            }
                            all_gold_medalists.append(gold_medalist)
    
    print(f"\n=== GOLD MEDALISTS EXTRACTED ===\n")
    print(f"Total gold medalists found: {len(all_gold_medalists)}")
    
    for i, medalist in enumerate(all_gold_medalists, 1):
        print(f"{i}. Name: {medalist['name']}")
        print(f"   Country: {medalist['country']}")
        print(f"   Raw data: {medalist['row_data']}")
        print()
    
    # Check if this championship had no Finland/Norway participation
    total_finland = sum(1 for table in medal_tables if table['has_finland'])
    total_norway = sum(1 for table in medal_tables if table['has_norway'])
    
    print(f"=== FINLAND/NORWAY PARTICIPATION CHECK ===\n")
    print(f"Tables mentioning Finland: {total_finland}")
    print(f"Tables mentioning Norway: {total_norway}")
    
    if total_finland == 0 and total_norway == 0:
        print("\n*** CONFIRMED: 1941 FIS Nordic World Ski Championships in Cortina d'Ampezzo ***")
        print("*** NO FINLAND OR NORWAY PARTICIPATION FOUND IN RESULTS TABLES ***")
        print("\nThis appears to be our target championship!")
        
        # Save the results
        results = {
            'championship': '1941 FIS Nordic World Ski Championships',
            'location': 'Cortina d\'Ampezzo',
            'year': 1941,
            'finland_participation': False,
            'norway_participation': False,
            'gold_medalists': all_gold_medalists,
            'total_medal_tables': len(medal_tables),
            'analysis_notes': 'No Finland or Norway found in any results tables'
        }
        
        with open('workspace/1941_championship_results.json', 'w') as f:
            json.dump(results, f, indent=2)
        
        print(f"\nResults saved to: workspace/1941_championship_results.json")
        
        # Extract unique countries that produced gold medalists
        countries_with_gold = set()
        for medalist in all_gold_medalists:
            if medalist['country'] != 'Country not clearly identified':
                countries_with_gold.add(medalist['country'])
        
        print(f"\n=== COUNTRIES THAT PRODUCED GOLD MEDALISTS ===\n")
        if countries_with_gold:
            for country in sorted(countries_with_gold):
                print(f"- {country}")
        else:
            print("Need to refine extraction - countries not clearly identified from current data")
            print("Raw data shows potential countries, but need better parsing")
    else:
        print(f"\nThis championship had Finland/Norway participation - not our target")
        print(f"Need to check other years")

else:
    print("\n1941 championship file not found. Checking other sources...")
    
    # Let's check what other files we have
    for file in html_files:
        print(f"\nChecking: {file}")
        filepath = f'workspace/{file}'
        
        with open(filepath, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # Quick check for championship years
        if 'cortina' in content.lower():
            print(f"  Contains Cortina mentions")
            finland_count = content.lower().count('finland')
            norway_count = content.lower().count('norway')
            print(f"  Finland mentions: {finland_count}")
            print(f"  Norway mentions: {norway_count}")

print("\n=== ANALYSIS COMPLETE ===\n")
print("Next steps: Refine gold medalist extraction and identify birth countries")
```
