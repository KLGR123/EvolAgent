# Function to get web content with proper headers
def get_page_content(url, timeout=30):
    """Fetch web content with appropriate headers and error handling"""
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
        'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
        'Accept-Language': 'en-US,en;q=0.5',
        'Accept-Encoding': 'gzip, deflate',
        'Connection': 'keep-alive',
        'Upgrade-Insecure-Requests': '1'
    }
    
    try:
        print(f'üì° Attempting to fetch: {url}')
        response = requests.get(url, headers=headers, timeout=timeout)
        response.raise_for_status()
        print(f'‚úÖ Successfully fetched: {url} (Status: {response.status_code})')
        print(f'   Content length: {len(response.text)} characters')
        return response.text
    except requests.exceptions.RequestException as e:
        print(f'‚ùå Error fetching {url}: {str(e)}')
        return None
    except Exception as e:
        print(f'‚ùå Unexpected error fetching {url}: {str(e)}')
        return None

research_targets = [

    research_results['sources_attempted'].append({
        'name': target['name'],
        'url': target['url'],
        'attempt_time': datetime.now().isoformat()
    })
    
    # Attempt to fetch the content
    content = get_page_content(target['url'])
    
    if content:
        research_results['successful_sources'].append({
            'name': target['name'],
            'url': target['url'],
            'content_length': len(content),
            'access_time': datetime.now().isoformat()
        })
        
        # Save the content for analysis
        filename = f'workspace/{target["name"].lower().replace(" ", "_").replace("-", "_")}_content.html'
        try:
            with open(filename, 'w', encoding='utf-8') as f:
                f.write(content)
            print(f'üíæ Saved content to: {filename}')
            
            # Basic content analysis
            content_lower = content.lower()
            zamboanga_mentions = content_lower.count('zamboanga del sur')

            election_mentions = content_lower.count('election')
            
            print(f'üìä Content Analysis:')

            print(f'   - "Governor" mentions: {governor_mentions}')
            print(f'   - "Election" mentions: {election_mentions}')
            
            # Look for specific years
            year_mentions = {}

                count = content_lower.count(year)
                if count > 0:
                    year_mentions[year] = count

                'zamboanga_mentions': zamboanga_mentions,
                'governor_mentions': governor_mentions,
                'election_mentions': election_mentions,
                'year_mentions': year_mentions,

            'reason': 'Could not access website'
        })
