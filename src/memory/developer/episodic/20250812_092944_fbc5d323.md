searcher = DDGS(timeout=15)

for query_num, query in enumerate(search_queries, 1):
    print(f'\nüîç SEARCH {query_num}/{len(search_queries)}: {query}')
    print('-' * 70)
    
    try:
        # Search with focus on Brazilian Portuguese sources
        results = searcher.text(
            query,
            max_results=10,
            page=1,
            backend=["google", "duckduckgo", "bing"],
            safesearch="off",
            region="pt-br"  # Brazilian Portuguese region
        )
        
        if results == []:
            print(f'‚ùå No results for: "{query}"')
        else:
            print(f'‚úÖ Found {len(results)} results')
            successful_searches += 1
            total_results += len(results)
            
            # Process each result
            for result_num, result in enumerate(results, 1):
                title = result.get('title', 'No title')
                body = result.get('body', 'No description')
                href = result.get('href', 'No URL')
                
                # Store result data
                result_data = {
                    'query': query,
                    'result_number': result_num,
                    'title': title,
                    'url': href,
                    'description': body,
                    'relevance_analysis': {}
                }
                
                # Analyze content for biographical and religious information
                combined_text = f'{title.lower()} {body.lower()}'
                
                # Check for individual mentions
                anivaldo_mentioned = 'anivaldo miranda' in combined_text or 'anivaldo' in combined_text
                maciel_mentioned = 'maciel oliveira' in combined_text or 'maciel' in combined_text

                for portuguese_term, english_term in religious_keywords.items():
                    if portuguese_term in combined_text:
                        found_religious_terms.append(f'{portuguese_term} ({english_term})')
                
                # Check for CBHSF and advocacy context
                cbhsf_mentioned = 'cbhsf' in combined_text or 'comit√™' in combined_text or 'bacia hidrogr√°fica' in combined_text
                sobradinho_mentioned = 'sobradinho' in combined_text or 'barragem' in combined_text or 'deslocados' in combined_text
                
                # Calculate relevance score
                relevance_score = 0
                relevance_factors = []
                
                if anivaldo_mentioned:
                    relevance_score += 5
                    relevance_factors.append('Anivaldo Miranda mentioned')
                
                if maciel_mentioned:
                    relevance_score += 5
                    relevance_factors.append('Maciel Oliveira mentioned')
                
                if found_religious_terms:
                    relevance_score += len(found_religious_terms) * 2
                    relevance_factors.append(f'Religious terms: {found_religious_terms}')
                
                if cbhsf_mentioned:
                    relevance_score += 3
                    relevance_factors.append('CBHSF context mentioned')
                
                if sobradinho_mentioned:
                    relevance_score += 3

                    'factors': relevance_factors,
                    'anivaldo_mentioned': anivaldo_mentioned,
                    'maciel_mentioned': maciel_mentioned,
                    'religious_terms_found': found_religious_terms,

                    print(f'     Title: {title}')
                    print(f'     URL: {href}')
                    print(f'     Factors: {relevance_factors}')
                    if found_religious_terms:
                        print(f'     Religious terms: {found_religious_terms}')
                    print(f'     Description: {body[:200]}...')
                    
                    # Store in biographical findings

    'queries_attempted': len(search_queries),
    'successful_searches': successful_searches,
    'total_results': total_results,
    'anivaldo_biographical_findings': len(search_data['biographical_findings']['anivaldo_miranda']),
    'maciel_biographical_findings': len(search_data['biographical_findings']['maciel_oliveira']),
    'anivaldo_religious_terms': list(set(search_data['religious_affiliations']['anivaldo_miranda'])),
    'maciel_religious_terms': list(set(search_data['religious_affiliations']['maciel_oliveira']))

print('üìä BIOGRAPHICAL SEARCH RESULTS ANALYSIS')
print('=' * 90)

print(f'\nüìà SEARCH STATISTICS:')
print(f'  ‚Ä¢ Queries attempted: {search_summary["queries_attempted"]}')
print(f'  ‚Ä¢ Successful searches: {search_summary["successful_searches"]}')
print(f'  ‚Ä¢ Total results collected: {search_summary["total_results"]}')

print(f'\nüë§ ANIVALDO MIRANDA FINDINGS:')

    print('\n  üìã Top Anivaldo Miranda Results:')
    anivaldo_results = sorted(search_data['biographical_findings']['anivaldo_miranda'],
