import os
from datetime import datetime

print("Examining workspace for LibreTexts search result files...")
print("=" * 80)

workspace_files = []
if os.path.exists('workspace'):

        if 'libretexts' in file.lower() and 'search' in file.lower():
            workspace_files.append(file)
            print(f"Found: {file}")
else:
    print("No workspace directory found")

if not workspace_files:

else:
    print(f"\nTotal search files found: {len(workspace_files)}")
    
    # Load and inspect the most recent search summary file
    summary_files = [f for f in workspace_files if 'summary' in f]
    
    if summary_files:

                print(f"  {key}: {value}")

                print(f"Title: {match['title']}")
                print(f"URL: {match['link']}")
                print(f"Authors confirmed: Marisa Alviar-Agnew & Henry Agnew")
                print(f"License: CK-12")
                break
        
        if target_exercise_url:
            print(f"\n" + "=" * 80)
            print("ACCESSING 1.E EXERCISES PAGE TO FIND EQUINE VETERINARIAN")
            print("=" * 80)

            # Since we can't directly scrape, let's try a more specific search

            import requests
            
            # Get SerpAPI key from environment variables
            api_key = os.getenv("SERPAPI_API_KEY")
            
            if api_key is None:
                print("Error: Missing API key for content search.")
            else:
                print(f"Searching for equine veterinarian content within the 1.E Exercises page...")
                
                # Define targeted searches for the specific page content
                content_search_queries = [
                    f'site:chem.libretexts.org/Courses/Chabot_College/Introduction_to_General_Organic_and_Biochemistry/01%3A_Chemistry_in_our_Lives/1.E%3A_Exercises "equine veterinarian"',
                    f'site:chem.libretexts.org "1.E Exercises" "equine veterinarian" "Marisa Alviar-Agnew"',
                    f'"chem.libretexts.org" "1.E Exercises" equine veterinarian chemistry',

                ]

                    print(f"Content Search {i}/{len(content_search_queries)}: {query}")
                    print("-" * 60)

                        "q": query,
                        "api_key": api_key,
                        "engine": "google",
                        "google_domain": "google.com",
                        "safe": "off",
                        "num": 10,
                        "type": "search"
                    }

                                # Store results with query context

                                    'results': [],
                                    'search_number': i,

                                })
                        else:

                    print("\n")

                content_results_file = f"workspace/libretexts_content_search_results_{timestamp}.json"
