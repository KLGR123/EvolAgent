    # Define specific search queries to find the 2016 Wiley paper

            "engine": "google",
            "google_domain": "google.com",
            "safe": "off",
            "num": 15,
            "type": "search"
        }
        
        try:

                        snippet_lower = snippet.lower()
                        combined_text = f"{title_lower} {snippet_lower}"
