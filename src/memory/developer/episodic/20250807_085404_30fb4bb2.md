        chronicle_matches = []

                # Check for pattern combinations using the properly scoped variable

                has_supernatural = any(re.search(pattern, sentence_lower_var) for pattern in supernatural_patterns)
                has_time = any(re.search(pattern, sentence_lower_var) for pattern in time_patterns)
                has_irish = any(re.search(pattern, sentence_lower_var) for pattern in irish_patterns)
                
                # Chronicle references (chronicle + supernatural/time/irish)

                    if has_time or has_army:
