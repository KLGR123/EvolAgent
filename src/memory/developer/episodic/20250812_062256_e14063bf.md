# Function to get HTML content from a URL with proper headers
def get_page_content(url, max_retries=3):
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
    }
    
    for attempt in range(max_retries):
        try:
            print(f"Fetching: {url} (Attempt {attempt + 1}/{max_retries})")
            response = requests.get(url, headers=headers, timeout=15)
            response.raise_for_status()
            print(f"Successfully fetched: {url}")
            return response.text
        except Exception as e:
            print(f"Error fetching {url}: {str(e)}")
            if attempt < max_retries - 1:
                time.sleep(2)
    return None

# Function to clean up text
def clean_text(text):
    if text is None:
        return ""
    cleaned = re.sub(r'\s+', ' ', text)
    cleaned = re.sub(r'\[\d+\]', '', cleaned)
    return cleaned.strip()

            external_research_results['successful_retrievals'].append(url)
            
            # Save the content for potential analysis
            url_parts = url.split('/')
            filename = f"workspace/external_source_{url_parts[-1]}.html"
            try:
                with open(filename, 'w', encoding='utf-8') as f:
                    f.write(content)
                print(f"Saved content to {filename}")
                
                # Try to parse for tax-related votes
                soup = BeautifulSoup(content, 'html.parser')
                text_content = soup.get_text().lower()
                
                # Look for tax-related keywords
                tax_keywords = ['tax cut', 'tax relief', 'economic growth', 'jobs and growth', 'h.r. 1836', 'h.r. 2']
                found_keywords = []
                
                for keyword in tax_keywords:
                    if keyword in text_content:
                        found_keywords.append(keyword)
                        print(f"Found keyword '{keyword}' in content")
                
                if found_keywords:
                    external_research_results['data_found'] = True
                    external_research_results['keywords_found'] = found_keywords
                
            except Exception as e:
                print(f"Error saving content: {e}")
        else:
            print(f"Failed to retrieve content from {url}")
            external_research_results['failed_retrievals'].append(url)
    
    # Save external research results
    with open('workspace/external_voting_research_results.json', 'w') as f:
        json.dump(external_research_results, f, indent=2)
    
    print(f"\nSaved external research results to workspace/external_voting_research_results.json")
