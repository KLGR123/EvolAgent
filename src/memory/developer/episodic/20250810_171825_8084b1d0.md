import os
import requests
import json
import time
from bs4 import BeautifulSoup
from urllib.parse import quote_plus

print('=== RESEARCHING FEMALE MYCOLOGIST AND BEATRIX POTTER CONNECTION ===')
print('Objective: Find mycologist who wrote "On the Germination of the Spores of the Agaricineae"')
print('Connection: Linnean Society 1897, letter to Noel Moore, possible Beatrix Potter link\n')

# Ensure workspace exists
os.makedirs('workspace', exist_ok=True)

# Initialize research data
research_data = {
    'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'objective': 'Research female mycologist 1897 Linnean Society paper and Beatrix Potter connection',
    'search_phases': [],
    'findings': {},
    'connections_verified': {}
}

print('=== PHASE 1: RESEARCHING THE 1897 LINNEAN SOCIETY PAPER ===')
print('=' * 70)

# The clues strongly suggest Beatrix Potter, who was indeed a mycologist
print('HYPOTHESIS: The mycologist is Beatrix Potter')
print('EVIDENCE TO VERIFY:')
print('1. Beatrix Potter submitted mycological research to Linnean Society')
print('2. Paper titled "On the Germination of the Spores of the Agaricineae" in 1897')
print('3. Read by male colleague (women couldn\'t present directly)')
print('4. "The Tale of Peter Rabbit" conceived in letter to Noel Moore')
print('5. Book translated into specific language by publisher with Jewish Bible connection\n')

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8'
}

# Research Beatrix Potter mycology work
print('--- Researching Beatrix Potter Mycology Work ---')
try:
    potter_query = '"Beatrix Potter" mycologist "Linnean Society" 1897 "Agaricineae"'
    potter_url = f'https://duckduckgo.com/?q={quote_plus(potter_query)}'
    print(f'Search URL: {potter_url}')
    
    potter_response = requests.get(potter_url, headers=headers, timeout=30)
    print(f'Search status: {potter_response.status_code}')
    
    if potter_response.status_code == 200:
        potter_filename = 'beatrix_potter_mycology_research.html'
        potter_filepath = os.path.join('workspace', potter_filename)
        
        with open(potter_filepath, 'w', encoding='utf-8') as f:
            f.write(potter_response.text)
        
        print(f'✅ Potter mycology research saved: {potter_filepath}')
        
        # Analyze content for key terms
        potter_soup = BeautifulSoup(potter_response.text, 'html.parser')
        potter_text = potter_soup.get_text().lower()
        
        # Count relevant terms
        mycology_count = potter_text.count('mycology') + potter_text.count('mycologist')
        linnean_count = potter_text.count('linnean')
        agaricineae_count = potter_text.count('agaricineae')
        spores_count = potter_text.count('spores')
        
        potter_analysis = {
            'mycology_mentions': mycology_count,
            'linnean_mentions': linnean_count,
            'agaricineae_mentions': agaricineae_count,
            'spores_mentions': spores_count,
            'research_file': potter_filepath
        }
        
        research_data['findings']['potter_mycology'] = potter_analysis
        
        print(f'Potter mycology analysis:')
        print(f'  • Mycology mentions: {mycology_count}')
        print(f'  • Linnean Society mentions: {linnean_count}')
        print(f'  • Agaricineae mentions: {agaricineae_count}')
        print(f'  • Spores mentions: {spores_count}')
        
except Exception as e:
    print(f'Error researching Potter mycology: {str(e)}')

time.sleep(3)

try:

    translation_url = f'https://duckduckgo.com/?q={quote_plus(translation_query)}'

            count = translation_text.count(lang)

        translation_analysis = {
            'language_mentions': language_counts,
