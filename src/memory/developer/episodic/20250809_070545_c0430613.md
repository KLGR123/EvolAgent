### Development Step 5: Locate 1994 Example for 'caminata' in Collins Dictionary and Identify Its Source Title for Translation

**Description**: Search for the Spanish word 'caminata' in the Collins Spanish-to-English dictionary online to locate the 1994 example sentence. Focus on finding the Collins dictionary entry that contains an example sentence from 1994, then identify the source title of that example sentence and prepare it for Google translation.

```
import os
import requests
from bs4 import BeautifulSoup
import time

print('=== ACCESSING COLLINS DICTIONARY CAMINATA ENTRY FOR 1994 EXAMPLE ===') 
print('Attempting direct access to Collins dictionary URLs found in search results\n')

# Create workspace directory if it doesn't exist
if not os.path.exists('workspace'):
    os.makedirs('workspace')

# Based on search results, we found these Collins dictionary URLs for caminata
collins_urls = [
    'https://www.collinsdictionary.com/dictionary/spanish-english/caminata',
    'https://www.collinsdictionary.com/us/sentences/spanish/caminata'
]

headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.5',
    'Accept-Encoding': 'gzip, deflate',
    'Connection': 'keep-alive',
    'Upgrade-Insecure-Requests': '1'
}

successful_pages = []
failed_pages = []

for i, url in enumerate(collins_urls, 1):
    print(f'=== ACCESSING COLLINS URL {i}/2 ===')
    print(f'URL: {url}')
    
    try:
        # Try with session and different approach
        session = requests.Session()
        session.headers.update(headers)
        
        response = session.get(url, timeout=30, allow_redirects=True)
        print(f'Response status: {response.status_code}')
        
        if response.status_code == 200:
            print(f'✓ Successfully accessed Collins dictionary page')
            
            # Save the content
            filename = f'collins_caminata_page_{i}.html'
            filepath = os.path.join('workspace', filename)
            
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(response.text)
            
            print(f'Content saved to: {filepath}')
            print(f'Content length: {len(response.text):,} characters')
            
            # Parse and analyze for 1994 content
            soup = BeautifulSoup(response.content, 'html.parser')
            page_text = soup.get_text()
            
            # Check for 1994 references
            has_1994 = '1994' in page_text
            print(f'Contains "1994": {has_1994}')
            
            if has_1994:
                print('\n*** 1994 CONTENT FOUND - ANALYZING ***')
                
                # Find lines containing 1994
                lines = page_text.split('\n')
                lines_with_1994 = []
                
                for line_num, line in enumerate(lines, 1):
                    if '1994' in line:
                        lines_with_1994.append((line_num, line.strip()))
                
                print(f'Found {len(lines_with_1994)} lines with "1994":')
                for line_num, line_text in lines_with_1994:
                    if line_text:  # Skip empty lines
                        print(f'  Line {line_num}: {line_text}')
                        
                        # Look for source titles or publication names
                        if any(indicator in line_text.lower() for indicator in ['source:', 'from:', 'title:', 'book:', 'publication:', 'newspaper:', 'magazine:']):
                            print(f'    *** POTENTIAL SOURCE TITLE FOUND ***')
                
                # Look for HTML elements that might contain examples with sources
                print('\n--- Searching for example sentence structures with 1994 ---')
                
                # Find all text elements that contain 1994
                elements_with_1994 = soup.find_all(text=lambda text: text and '1994' in str(text))
                
                for element in elements_with_1994:
                    parent = element.parent
                    if parent:
                        parent_text = parent.get_text().strip()
                        if parent_text and len(parent_text) > 20:  # Skip very short text
                            print(f'Element: {parent.name if parent.name else "text"}')
                            print(f'Content: {parent_text[:200]}...' if len(parent_text) > 200 else f'Content: {parent_text}')
                            print('---')
            
            else:
                print('No 1994 content found in this page')
                
                # Still analyze the page structure for examples
                print('\n--- Analyzing page structure for example sentences ---')
                
                # Look for example sections
                example_sections = soup.find_all(['div', 'section', 'span'], class_=lambda x: x and any(term in x.lower() for term in ['example', 'sentence', 'usage', 'citation']))
                
                print(f'Found {len(example_sections)} potential example sections')
                
                for j, section in enumerate(example_sections[:5], 1):  # Limit to first 5
                    section_text = section.get_text().strip()
                    if section_text:
                        print(f'\nExample section {j}:')
                        print(f'Class: {section.get("class", "No class")}')
                        print(f'Content preview: {section_text[:150]}...' if len(section_text) > 150 else f'Content: {section_text}')
            
            successful_pages.append({
                'url': url,
                'filename': filepath,
                'has_1994': has_1994,
                'content_length': len(response.text)
            })
            
        else:
            print(f'✗ Failed to access - Status: {response.status_code}')
            if response.status_code == 403:
                print('  Access forbidden - may need different approach')
            elif response.status_code == 404:
                print('  Page not found')
            
            failed_pages.append({
                'url': url,
                'status': response.status_code,
                'reason': response.reason
            })
            
    except Exception as e:
        print(f'✗ Error accessing {url}: {str(e)}')
        failed_pages.append({
            'url': url,
            'error': str(e)
        })
    
    print('\n' + '='*80 + '\n')
    time.sleep(3)  # Be respectful to servers

print('=== ACCESS SUMMARY ===')
print(f'Successfully accessed: {len(successful_pages)} Collins pages')
print(f'Failed to access: {len(failed_pages)} pages')

if successful_pages:
    print('\n--- Successfully Accessed Pages ---')
    for page in successful_pages:
        print(f'✓ {page["url"]}')
        print(f'  File: {page["filename"]}')
        print(f'  Has 1994: {page["has_1994"]}')
        print(f'  Size: {page["content_length"]:,} characters')
        
        if page['has_1994']:
            print('  *** CONTAINS 1994 REFERENCE - PRIORITY FOR ANALYSIS ***')

if failed_pages:
    print('\n--- Failed Access Attempts ---')
    for page in failed_pages:
        print(f'✗ {page["url"]}')
        if 'status' in page:
            print(f'  Status: {page["status"]} - {page.get("reason", "Unknown")}')
        if 'error' in page:
            print(f'  Error: {page["error"]}')

# Save access summary
summary_file = os.path.join('workspace', 'collins_access_summary.txt')
with open(summary_file, 'w', encoding='utf-8') as f:
    f.write('COLLINS DICTIONARY CAMINATA ACCESS SUMMARY\n')
    f.write('='*50 + '\n\n')
    f.write(f'Access timestamp: {time.strftime("%Y-%m-%d %H:%M:%S")}\n')
    f.write(f'Target: Collins Spanish-English dictionary "caminata" entry\n')
    f.write(f'Objective: Find 1994 example sentence and source title\n\n')
    
    f.write(f'URLs attempted: {len(collins_urls)}\n')
    f.write(f'Successful accesses: {len(successful_pages)}\n')
    f.write(f'Failed accesses: {len(failed_pages)}\n\n')
    
    if successful_pages:
        f.write('SUCCESSFUL PAGES:\n')
        f.write('-'*30 + '\n')
        for page in successful_pages:
            f.write(f'URL: {page["url"]}\n')
            f.write(f'File: {page["filename"]}\n')
            f.write(f'Has 1994: {page["has_1994"]}\n')
            f.write(f'Content length: {page["content_length"]:,} characters\n')
            f.write('-'*20 + '\n')
    
    if failed_pages:
        f.write('\nFAILED PAGES:\n')
        f.write('-'*30 + '\n')
        for page in failed_pages:
            f.write(f'URL: {page["url"]}\n')
            if 'status' in page:
                f.write(f'Status: {page["status"]} - {page.get("reason", "Unknown")}\n')
            if 'error' in page:
                f.write(f'Error: {page["error"]}\n')
            f.write('-'*20 + '\n')

print(f'\n✓ Access summary saved to: {summary_file}')

# Next steps guidance
print('\n=== NEXT STEPS ===')
if any(page['has_1994'] for page in successful_pages):
    print('✓ Found Collins pages with 1994 references')
    print('Next: Parse the saved HTML files to extract the 1994 example sentence and source title')
else:
    print('No 1994 references found in accessible Collins pages')
    print('May need to try alternative approaches:')
    print('1. Search for Collins dictionary archives or historical versions')
    print('2. Look for academic citations referencing Collins 1994 examples')
    print('3. Try different Collins dictionary sections or related pages')
```
