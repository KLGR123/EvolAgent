import os
import json
import requests
from urllib.parse import quote
import re

print("=== BECKY SHARP (1935) PRODUCTION COMPANY SEARCH ===")
print("Objective: Identify the production company/studio that produced 'Becky Sharp' (1935)")
print("Focus: First full-length color film using three-strip Technicolor\n")

# Search queries targeting different aspects of the research
search_queries = [
    "Becky Sharp 1935 three-strip Technicolor producer production company",
    "first full-length Technicolor film production company Becky Sharp",
    "Becky Sharp movie producer studio 1935 Technicolor",
    "three-strip Technicolor Becky Sharp who produced studio",
    "Becky Sharp 1935 film production company Pioneer Pictures",
    "Becky Sharp Rouben Mamoulian producer studio Technicolor"
]

# Get SerpAPI key from environment variables
api_key = os.getenv("SERPAPI_API_KEY")

if api_key is None:
    print("Error: Missing API key. Make sure you have SERPAPI_API_KEY in your environment variables.")
else:
    print(f"Starting comprehensive search with {len(search_queries)} different query approaches\n")
    
    all_search_results = []
    
    for i, query in enumerate(search_queries, 1):
        print(f"=== SEARCH {i}/{len(search_queries)} ===")
        print(f"Query: {query}")
        
        # Prepare API request parameters
        params = {
            "q": query,
            "api_key": api_key,
            "engine": "google",
            "google_domain": "google.com",
            "safe": "off",
            "num": 12,  # Get sufficient results for comprehensive search
            "type": "search",
        }
        
        try:
            # Make API request to SerpAPI
            response = requests.get("https://serpapi.com/search.json", params=params, timeout=30)
            
            if response.status_code == 200:
                results = response.json()
                print(f"Search successful - found {len(results.get('organic_results', []))} organic results")
                
                # Store results with query context
                search_result = {
                    'query_number': i,
                    'query_text': query,
                    'results_data': results,
                    'organic_count': len(results.get('organic_results', []))
                }
                all_search_results.append(search_result)
                
                # Analyze results for relevant keywords immediately
                if results.get("organic_results"):
                    print("\nAnalyzing results for production company and Technicolor keywords:\n")
                    
                    for j, result in enumerate(results['organic_results'][:8], 1):  # Check first 8 results
                        title = result.get('title', 'No title')
                        link = result.get('link', 'No link')
                        snippet = result.get('snippet', 'No snippet')
                        
                        combined_text = f"{title} {snippet}".lower()
                        
                        print(f"Result {j}:")
                        print(f"Title: {title}")
                        print(f"URL: {link}")
                        print(f"Snippet: {snippet[:250]}{'...' if len(snippet) > 250 else ''}")
                        
                        # Look for production company keywords
                        production_keywords = ['producer', 'production', 'studio', 'company', 'produced by', 'pioneer pictures', 'rko', 'paramount', 'mgm', 'warner']
                        found_production_keywords = [kw for kw in production_keywords if kw in combined_text]
                        
                        # Look for Technicolor keywords
                        technicolor_keywords = ['technicolor', 'three-strip', 'color', 'colour', 'first full-length']
                        found_technicolor_keywords = [kw for kw in technicolor_keywords if kw in combined_text]
                        
                        # Look for Becky Sharp references
                        movie_keywords = ['becky sharp', '1935', 'rouben mamoulian', 'thackeray', 'vanity fair']
                        found_movie_keywords = [kw for kw in movie_keywords if kw in combined_text]
                        
                        # Look for historical significance keywords
                        historical_keywords = ['first', 'pioneer', 'breakthrough', 'milestone', 'historic']
                        found_historical_keywords = [kw for kw in historical_keywords if kw in combined_text]
                        
                        # Look for director/cast keywords
                        people_keywords = ['rouben mamoulian', 'miriam hopkins', 'director', 'cast']
                        found_people_keywords = [kw for kw in people_keywords if kw in combined_text]
                        
                        # Highlight significant matches
                        if found_production_keywords:
                            print(f"*** PRODUCTION KEYWORDS: {', '.join(found_production_keywords)} ***")
                        
                        if found_technicolor_keywords:
                            print(f"*** TECHNICOLOR KEYWORDS: {', '.join(found_technicolor_keywords)} ***")
                        
                        if found_movie_keywords:
                            print(f"*** BECKY SHARP KEYWORDS: {', '.join(found_movie_keywords)} ***")
                        
                        if found_historical_keywords:
                            print(f"*** HISTORICAL KEYWORDS: {', '.join(found_historical_keywords)} ***")
                        
                        if found_people_keywords:
                            print(f"*** PEOPLE KEYWORDS: {', '.join(found_people_keywords)} ***")
                        
                        # Special attention to highly relevant results
                        relevance_score = len(found_production_keywords) + len(found_technicolor_keywords) + len(found_movie_keywords)
                        if relevance_score >= 3:
                            print(f"*** HIGH RELEVANCE RESULT (Score: {relevance_score}/3) ***")
                        
                        # Look for specific production company names
                        company_patterns = [
                            r'Pioneer Pictures',
                            r'RKO',
                            r'Paramount',
                            r'MGM',
                            r'Warner.*Bros',
                            r'United Artists',
                            r'Columbia Pictures',
                            r'Universal',
                            r'20th Century Fox'
                        ]
                        
                        for pattern in company_patterns:
                            matches = re.findall(pattern, title + ' ' + snippet, re.IGNORECASE)
                            if matches:
                                print(f"*** PRODUCTION COMPANY FOUND: {matches} ***")
                        
                        # Look for producer names
                        producer_patterns = [
                            r'produced by ([A-Z][a-z]+ [A-Z][a-z]+)',
                            r'producer ([A-Z][a-z]+ [A-Z][a-z]+)',
                            r'([A-Z][a-z]+ [A-Z][a-z]+) produced'
                        ]
                        
                        for pattern in producer_patterns:
                            matches = re.findall(pattern, snippet, re.IGNORECASE)
                            if matches:
                                print(f"*** POTENTIAL PRODUCER NAMES: {matches} ***")
                        
                        # Look for specific mentions of "first full-length color film"
                        if 'first full-length' in combined_text and ('color' in combined_text or 'colour' in combined_text):
                            print(f"*** FIRST FULL-LENGTH COLOR FILM REFERENCE FOUND ***")
                        
                        print("-" * 60)
                
                print(f"\nCompleted search {i}/{len(search_queries)}\n")
                
            else:
                print(f"Search failed with status {response.status_code}: {response.text}")
                
        except Exception as e:
            print(f"Error during search {i}: {str(e)}")
        
        print("=" * 70)
        print()
    
    # Save all search results for detailed analysis
    comprehensive_results = {
        'search_objective': 'Identify production company/studio that produced Becky Sharp (1935) - first full-length three-strip Technicolor film',
        'search_timestamp': '2024-12-19',
        'total_queries_executed': len(search_queries),
        'queries_used': search_queries,
        'all_search_results': all_search_results,
        'total_organic_results': sum([sr['organic_count'] for sr in all_search_results])
    }
    
    with open('workspace/becky_sharp_production_search.json', 'w', encoding='utf-8') as f:
        json.dump(comprehensive_results, f, indent=2, ensure_ascii=False)
    
    print(f"\n=== SEARCH PHASE COMPLETE ===\n")
    print(f"Total searches conducted: {len(search_queries)}")
    print(f"Total organic results collected: {comprehensive_results['total_organic_results']}")
    print(f"Results saved to: workspace/becky_sharp_production_search.json")
    
    print("\nNext steps:")
    print("1. Analyze collected results for production company names")
    print("2. Focus on results mentioning Pioneer Pictures, RKO, or other studios")
    print("3. Cross-reference Becky Sharp (1935) production information")
    print("4. Identify the specific production company and any key producers")
    print("5. Establish connection to composer employment for next research phase")
