import os
from ddgs import DDGS
import json
from datetime import datetime

print('=== JOHN PARISH ROBERTSON BIRTHPLACE RESEARCH ===')
print('Searching for biographical information about John Parish Robertson to identify his birthplace...')
print()

# Check workspace directory
workspace_dir = 'workspace'
if not os.path.exists(workspace_dir):
    os.makedirs(workspace_dir)
    print(f'üìÅ Created workspace directory: {workspace_dir}')
else:
    print(f'üìÅ Using existing workspace directory: {workspace_dir}')
    existing_files = os.listdir(workspace_dir)
    if existing_files:
        print(f'üìÑ Existing files in workspace: {existing_files}')
    else:
        print('üìÑ Workspace directory is empty')

print()

# Define comprehensive search queries for John Parish Robertson's birthplace
search_queries = [
    'John Parish Robertson birthplace born where biography',
    'John Parish Robertson biography birth location early life',
    'Robertson brothers Paraguay birthplace origin',
    'William John Parish Robertson birth origin brothers',
    '"John Parish Robertson" birthplace born encyclopedia',
    'John Parish Robertson Scottish British birth location',
    'Robertson brothers merchants Paraguay biography birthplace',
    'John Parish Robertson biographical details birth city town'
]

print(f'üîç EXECUTING BIOGRAPHICAL SEARCH QUERIES')
print('=' * 50)
print(f'Total queries planned: {len(search_queries)}')
for i, query in enumerate(search_queries, 1):
    print(f'  {i}. {query}')

print()

# Initialize search results storage
all_search_results = []
searcher = DDGS(timeout=15)
backends = ["google", "duckduckgo", "bing", "yandex", "brave"]

# Execute each search query
for i, query in enumerate(search_queries, 1):
    print(f'\nüîç SEARCH {i}/{len(search_queries)}: {query}')
    print('-' * 70)
    
    try:
        results = searcher.text(
            query, 
            max_results=8, 
            page=1, 
            backend=backends, 
            safesearch="off", 
            region="en-us"
        )
        
        if results and len(results) > 0:
            print(f'‚úÖ Found {len(results)} results')
            
            for j, result in enumerate(results, 1):
                title = result.get('title', 'No title')
                href = result.get('href', 'No link')
                body = result.get('body', 'No snippet')
                
                print(f'\nResult {j}:')
                print(f'Title: {title}')
                print(f'Link: {href}')
                print(f'Snippet: {body[:250]}...' if len(body) > 250 else f'Snippet: {body}')
                
                # Analyze content for birthplace indicators
                content_text = f'{title.lower()} {body.lower()}'
                
                # Look for birthplace keywords
                birthplace_keywords = ['born', 'birth', 'birthplace', 'native', 'origin']
                location_keywords = ['scotland', 'scottish', 'england', 'english', 'britain', 'british', 'london', 'edinburgh', 'glasgow']
                robertson_keywords = ['john parish robertson', 'parish robertson', 'robertson brothers']
                
                birthplace_matches = [kw for kw in birthplace_keywords if kw in content_text]
                location_matches = [kw for kw in location_keywords if kw in content_text]
                robertson_matches = [kw for kw in robertson_keywords if kw in content_text]
                
                # Calculate relevance score
                relevance_score = len(birthplace_matches) + len(location_matches) + len(robertson_matches)
                
                if birthplace_matches:
                    print(f'üéÇ Birth keywords found: {birthplace_matches}')
                if location_matches:
                    print(f'üåç Location keywords found: {location_matches}')
                if robertson_matches:
                    print(f'üë§ Robertson references: {robertson_matches}')
                if relevance_score > 0:
                    print(f'üìä Relevance score: {relevance_score}')
                
                # Store result with analysis
                result_data = {
                    'query': query,
                    'query_number': i,
                    'result_number': j,
                    'title': title,
                    'link': href,
                    'snippet': body,
                    'birthplace_keywords': birthplace_matches,
                    'location_keywords': location_matches,
                    'robertson_keywords': robertson_matches,
                    'relevance_score': relevance_score,
                    'content_length': len(body)
                }
                
                all_search_results.append(result_data)
                print('-' * 40)
        else:
            print(f'‚ùå No results found for query {i}')
    
    except Exception as e:
        print(f'‚ùå Error executing search {i}: {str(e)}')
        continue

print(f'\n\nüìä SEARCH RESULTS SUMMARY')
print('=' * 30)
print(f'Total queries executed: {len(search_queries)}')
print(f'Total results collected: {len(all_search_results)}')

# Analyze and rank results by relevance
high_relevance_results = [r for r in all_search_results if r['relevance_score'] >= 3]
medium_relevance_results = [r for r in all_search_results if r['relevance_score'] == 2]
low_relevance_results = [r for r in all_search_results if r['relevance_score'] == 1]

print(f'High relevance results (3+ score): {len(high_relevance_results)}')
print(f'Medium relevance results (2 score): {len(medium_relevance_results)}')
print(f'Low relevance results (1 score): {len(low_relevance_results)}')
print(f'No relevance results (0 score): {len(all_search_results) - len(high_relevance_results) - len(medium_relevance_results) - len(low_relevance_results)}')

# Sort all results by relevance score
all_search_results.sort(key=lambda x: x['relevance_score'], reverse=True)

print(f'\nüéØ TOP RELEVANCE RESULTS FOR BIRTHPLACE ANALYSIS')
print('=' * 55)

# Display top 10 most relevant results
top_results = all_search_results[:10]
for i, result in enumerate(top_results, 1):
    print(f'\n{i}. {result["title"]}')
    print(f'   üîó Source: {result["link"]}')
    print(f'   üìä Relevance Score: {result["relevance_score"]}')
    print(f'   üéÇ Birth keywords: {result["birthplace_keywords"]}')
    print(f'   üåç Location keywords: {result["location_keywords"]}')
    print(f'   üë§ Robertson keywords: {result["robertson_keywords"]}')
    print(f'   üìù Snippet: {result["snippet"][:200]}...' if len(result["snippet"]) > 200 else f'   üìù Snippet: {result["snippet"]}')
    print('-' * 60)

# Save comprehensive search results
timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
results_file = f'workspace/john_parish_robertson_birthplace_search_{timestamp}.json'

search_data = {
    'research_timestamp': datetime.now().isoformat(),
    'research_objective': 'Find biographical information about John Parish Robertson to identify his birthplace',
    'search_queries_executed': search_queries,
    'search_summary': {
        'total_queries': len(search_queries),
        'total_results': len(all_search_results),
        'high_relevance_count': len(high_relevance_results),
        'medium_relevance_count': len(medium_relevance_results),
        'low_relevance_count': len(low_relevance_results)
    },
    'results': all_search_results
}

with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(search_data, f, indent=2, ensure_ascii=False)

print(f'\nüìÑ COMPREHENSIVE SEARCH RESULTS SAVED')
print(f'File: {results_file}')
print(f'Total results stored: {len(all_search_results)}')

# Extract potential birthplace information from high relevance results
print(f'\nüè† BIRTHPLACE ANALYSIS FROM HIGH RELEVANCE RESULTS')
print('=' * 55)

birthplace_evidence = []
for result in high_relevance_results:
    content = f'{result["title"]} {result["snippet"]}'
    
    # Look for specific birthplace patterns
    birthplace_patterns = [
        'born in', 'born at', 'birthplace', 'native of', 'from',
        'scotland', 'scottish', 'england', 'english', 'britain', 'british',
        'london', 'edinburgh', 'glasgow', 'greenock'
    ]
    
    found_patterns = []
    for pattern in birthplace_patterns:
        if pattern in content.lower():
            found_patterns.append(pattern)
    
    if found_patterns:
        birthplace_evidence.append({
            'title': result['title'],
            'link': result['link'],
            'snippet': result['snippet'],
            'birthplace_indicators': found_patterns,
            'relevance_score': result['relevance_score']
        })

if birthplace_evidence:
    print(f'Found {len(birthplace_evidence)} results with birthplace indicators:')
    for i, evidence in enumerate(birthplace_evidence, 1):
        print(f'\n{i}. {evidence["title"]}')
        print(f'   üîó {evidence["link"]}')
        print(f'   üìä Relevance: {evidence["relevance_score"]}')
        print(f'   üè† Birthplace indicators: {evidence["birthplace_indicators"]}')
        print(f'   üìù {evidence["snippet"][:300]}...' if len(evidence["snippet"]) > 300 else f'   üìù {evidence["snippet"]}')
else:
    print('No clear birthplace indicators found in high relevance results.')
    print('Additional research may be needed to identify specific birthplace.')

# Save birthplace analysis
birthplace_file = f'workspace/john_parish_robertson_birthplace_analysis_{timestamp}.json'
birthplace_analysis = {
    'analysis_timestamp': datetime.now().isoformat(),
    'subject': 'John Parish Robertson',
    'research_focus': 'Birthplace identification',
    'evidence_found': birthplace_evidence,
    'analysis_summary': {
        'high_relevance_results_analyzed': len(high_relevance_results),
        'birthplace_evidence_count': len(birthplace_evidence),
        'research_status': 'Initial search completed - may need additional targeted research'
    }
}

with open(birthplace_file, 'w', encoding='utf-8') as f:
    json.dump(birthplace_analysis, f, indent=2, ensure_ascii=False)

print(f'\nüìÑ BIRTHPLACE ANALYSIS SAVED')
print(f'File: {birthplace_file}')

print(f'\n‚úÖ INITIAL BIOGRAPHICAL SEARCH COMPLETED')
print('=' * 45)
print(f'üìä Search Statistics:')
print(f'  - Queries executed: {len(search_queries)}')
print(f'  - Total results: {len(all_search_results)}')
print(f'  - High relevance results: {len(high_relevance_results)}')
print(f'  - Birthplace evidence found: {len(birthplace_evidence)}')
print(f'\nüìÅ Files created:')
print(f'  - {results_file}')
print(f'  - {birthplace_file}')

print(f'\nüîç RESEARCH STATUS:')
if birthplace_evidence:
    print(f'‚úÖ Found {len(birthplace_evidence)} results with potential birthplace information')
    print('üìã Next step: Analyze the evidence to determine specific birthplace')
else:
    print('‚ö†Ô∏è Limited birthplace evidence found in initial search')
    print('üìã May need additional targeted searches or different search terms')

print(f'\nüìù PRELIMINARY FINDINGS:')
print('Based on the search results, further analysis is needed to determine')
print('John Parish Robertson\'s specific birthplace. The search has collected')
print(f'{len(all_search_results)} results that will be analyzed for birthplace clues.')
