import os
import requests
import json
import time
from urllib.parse import quote_plus
from bs4 import BeautifulSoup
from collections import Counter

print('=== SEARCH FOR 17TH CENTURY IRISH SPECTRAL ARMY AND SUPERNATURAL DOCUMENTS ===')
print('Objective: Find historical chronicles from 1600s documenting both spectral army apparitions')
print('in Ireland and unusual supernatural events that M.R. James may have used as inspiration\n')

# Ensure workspace directory exists
os.makedirs('workspace', exist_ok=True)

# Initialize comprehensive results storage
search_results = {
    'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'objective': 'Find 17th century documents recording spectral army in Ireland and supernatural events',
    'target_period': '1600s (17th century)',
    'search_focus': 'Chronicles, historical accounts, antiquarian records',
    'connection': 'Source material for M.R. James Suffolk spider curse inspiration',
    'search_queries': [],
    'findings': [],
    'potential_sources': [],
    'historians_identified': [],
    'analysis': {}
}

print('TARGET DOCUMENT CHARACTERISTICS:')
print('• Time Period: 17th century (1600s)')
print('• Geographic Focus: Ireland (with possible Britain connections)')
print('• Content: Spectral army apparitions AND supernatural events')
print('• Document Types: Chronicles, historical accounts, antiquarian records')
print('• Authors: Historians, chroniclers, antiquarians of the period')
print('• Purpose: Identify M.R. James source material for Suffolk spider curse')
print()

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.9',
    'Connection': 'keep-alive'
}

print('=== PHASE 1: TARGETED SEARCHES FOR 17TH CENTURY IRISH SUPERNATURAL CHRONICLES ===')
print('=' * 80)

# Comprehensive search queries targeting specific aspects
targeted_queries = [
    '17th century Ireland spectral army apparition chronicle historical account',
    '1600s Irish supernatural events chronicles historians antiquarians',
    'spectral army Ireland 17th century historical documents records',
    '17th century Ireland ghost army apparition supernatural chronicle',
    'Irish chronicles 1600s supernatural events spectral apparitions historians',
    'antiquarian records Ireland 17th century supernatural army spirits',
    '17th century British Isles supernatural chronicles spectral army Ireland',
    'M.R. James source material 17th century Irish supernatural chronicles',
    'historical accounts Ireland 1600s ghostly army apparitions supernatural',
    'chroniclers Ireland 17th century supernatural events spectral phenomena'
]

print(f'Executing {len(targeted_queries)} comprehensive searches:')
for i, query in enumerate(targeted_queries, 1):
    print(f'  {i:2d}. {query}')

for i, query in enumerate(targeted_queries, 1):
    print(f'\nSearch {i}/{len(targeted_queries)}: {query}')
    print('-' * 70)
    
    try:
        # Construct Google search URL
        google_url = f'https://www.google.com/search?q={quote_plus(query)}'
        print(f'URL: {google_url}')
        
        response = requests.get(google_url, headers=headers, timeout=25)
        print(f'Status: {response.status_code}')
        
        if response.status_code == 200:
            # Save HTML for detailed analysis
            filename = f'search_{i:02d}_{query[:50].replace(" ", "_").replace(".", "")}.html'
            filepath = os.path.join('workspace', filename)
            
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(response.text)
            
            print(f'Saved: {filepath}')
            
            # Parse and analyze results
            soup = BeautifulSoup(response.text, 'html.parser')
            page_text = soup.get_text().lower()
            
            # Define comprehensive relevance scoring system
            key_terms = {
                # Time period indicators
                '17th century': 5, '1600s': 5, 'seventeenth century': 5,
                '1600': 4, '1610': 4, '1620': 4, '1630': 4, '1640': 4, 
                '1650': 4, '1660': 4, '1670': 4, '1680': 4, '1690': 4,
                
                # Geographic indicators
                'ireland': 4, 'irish': 4, 'dublin': 3, 'cork': 3, 'ulster': 3,
                'munster': 3, 'leinster': 3, 'connacht': 3, 'britain': 2,
                
                # Supernatural phenomena
                'spectral army': 6, 'ghostly army': 6, 'phantom army': 6,
                'apparition': 4, 'supernatural': 4, 'ghost': 3, 'spirit': 3,
                'phantom': 4, 'spectral': 4, 'ghostly': 4, 'paranormal': 3,
                
                # Document types
                'chronicle': 5, 'historical account': 4, 'antiquarian': 4,
                'historian': 3, 'chronicler': 4, 'record': 3, 'manuscript': 3,
                'annals': 4, 'history': 2, 'document': 2,
                
                # M.R. James connection
                'm.r. james': 5, 'montague james': 5, 'james': 2,
                'suffolk': 3, 'spider curse': 4, 'source material': 3
            }
            
            found_terms = []
            relevance_score = 0
            
            for term, weight in key_terms.items():
                if term in page_text:
                    found_terms.append(term)
                    relevance_score += weight
            
            print(f'Relevance score: {relevance_score}')
            print(f'Found terms ({len(found_terms)}): {", ".join(found_terms[:10])}')
            
            # Look for specific historian/chronicler names
            known_chroniclers = [
                'geoffrey keating', 'seathrún céitinn', 'four masters',
                'annals of the four masters', 'john lynch', 'david rothe',
                'peter walsh', 'richard stanihurst', 'edmund campion',
                'holinshed', 'raphael holinshed', 'camden', 'william camden',
                'giraldus cambrensis', 'gerald of wales'
            ]
            
            chroniclers_found = []
            for chronicler in known_chroniclers:
                if chronicler in page_text:
                    chroniclers_found.append(chronicler)
                    relevance_score += 3
                    print(f'  ✓ Chronicler found: {chronicler}')
            
            # Store comprehensive finding
            finding = {
                'query': query,
                'relevance_score': relevance_score,
                'found_terms': found_terms,
                'chroniclers_found': chroniclers_found,
                'html_file': filepath,
                'search_index': i
            }
            
            search_results['findings'].append(finding)
            search_results['search_queries'].append(query)
            search_results['historians_identified'].extend(chroniclers_found)
            
            # Extract detailed information for high-relevance results
            if relevance_score >= 20:
                print('🎯 HIGH RELEVANCE - Extracting detailed information...')
                
                # Look for specific text snippets mentioning key combinations
                sentences = page_text.split('.')
                relevant_snippets = []
                
                for sentence in sentences:
                    sentence = sentence.strip()
                    if len(sentence) > 30 and len(sentence) < 300:
                        # Check for combinations of key terms
                        has_time = any(term in sentence for term in ['17th', '1600', 'seventeenth'])
                        has_place = any(term in sentence for term in ['ireland', 'irish'])
                        has_supernatural = any(term in sentence for term in ['spectral', 'ghost', 'apparition', 'supernatural', 'phantom'])
                        has_document = any(term in sentence for term in ['chronicle', 'account', 'record', 'historian'])
                        
                        if (has_time and has_place) or (has_supernatural and has_document) or \
                           (has_place and has_supernatural and has_document):
                            relevant_snippets.append(sentence)
                
                if relevant_snippets:
                    print(f'Key text snippets found ({len(relevant_snippets)}):')  
                    for j, snippet in enumerate(relevant_snippets[:5], 1):
                        print(f'  {j}. {snippet[:200]}...')
                    
                    finding['key_snippets'] = relevant_snippets[:10]
                    
                    # This is a potential source - add to candidates
                    search_results['potential_sources'].append({
                        'query': query,
                        'score': relevance_score,
                        'terms': found_terms[:15],
                        'chroniclers': chroniclers_found,
                        'snippets': relevant_snippets[:5],
                        'file': filepath
                    })
            
            elif relevance_score >= 10:
                print('⭐ MODERATE RELEVANCE - Notable result')
                
        else:
            print(f'Failed with status {response.status_code}')
            
    except Exception as e:
        print(f'Error: {str(e)}')
    
    time.sleep(4)  # Rate limiting

print('\n' + '=' * 90)
print('PHASE 2: COMPREHENSIVE ANALYSIS OF SEARCH RESULTS')
print('=' * 90)

total_findings = len(search_results['findings'])
print(f'Total search results collected: {total_findings}')
print(f'Potential source documents identified: {len(search_results["potential_sources"])}')
print(f'Unique chroniclers/historians found: {len(set(search_results["historians_identified"]))}')

if search_results['findings']:
    # Sort by relevance score
    search_results['findings'].sort(key=lambda x: x['relevance_score'], reverse=True)
    
    print('\n📊 RELEVANCE DISTRIBUTION:')
    print('-' * 40)
    
    high_relevance = [f for f in search_results['findings'] if f['relevance_score'] >= 20]
    moderate_relevance = [f for f in search_results['findings'] if 10 <= f['relevance_score'] < 20]
    low_relevance = [f for f in search_results['findings'] if 5 <= f['relevance_score'] < 10]
    
    print(f'High relevance results (20+ points): {len(high_relevance)}')
    print(f'Moderate relevance results (10-19 points): {len(moderate_relevance)}')
    print(f'Low relevance results (5-9 points): {len(low_relevance)}')
    
    if high_relevance:
        print('\n🎯 HIGH RELEVANCE FINDINGS:')
        print('=' * 50)
        for i, finding in enumerate(high_relevance, 1):
            print(f'\n{i}. Query: {finding["query"]}')  
            print(f'   Score: {finding["relevance_score"]}')
            print(f'   Key terms: {", ".join(finding["found_terms"][:8])}')
            if finding['chroniclers_found']:
                print(f'   Chroniclers: {", ".join(finding["chroniclers_found"])}')
            if finding.get('key_snippets'):
                print(f'   Key snippet: {finding["key_snippets"][0][:150]}...')
            print(f'   File: {finding["html_file"]}')
    
    elif moderate_relevance:
        print('\n⭐ MODERATE RELEVANCE FINDINGS:')
        print('=' * 50)
        for i, finding in enumerate(moderate_relevance[:5], 1):
            print(f'\n{i}. Query: {finding["query"]}')  
            print(f'   Score: {finding["relevance_score"]}')
            print(f'   Key terms: {", ".join(finding["found_terms"][:6])}')
            if finding['chroniclers_found']:
                print(f'   Chroniclers: {", ".join(finding["chroniclers_found"])}')
    
    # Analyze chronicler/historian patterns
    all_chroniclers = search_results['historians_identified']
    if all_chroniclers:
        chronicler_counts = Counter(all_chroniclers)
        print('\n📚 CHRONICLER/HISTORIAN ANALYSIS:')
        print('-' * 45)
        print('Most frequently mentioned historians:')
        for chronicler, count in chronicler_counts.most_common(8):
            print(f'  • {chronicler}: {count} mentions')
        
        search_results['analysis']['top_chroniclers'] = dict(chronicler_counts.most_common(5))
    
    # Analyze term patterns
    all_terms = []
    for finding in search_results['findings']:
        all_terms.extend(finding['found_terms'])
    
    if all_terms:
        term_frequency = Counter(all_terms)
        print('\n📈 TERM FREQUENCY ANALYSIS:')
        print('-' * 35)
        print('Most common relevant terms:')
        for term, count in term_frequency.most_common(12):
            print(f'  • {term}: {count} occurrences')
        
        search_results['analysis']['top_terms'] = dict(term_frequency.most_common(10))

else:
    print('❌ No search results collected')

# Detailed analysis of potential sources
if search_results['potential_sources']:
    print('\n' + '=' * 90)
    print('PHASE 3: DETAILED ANALYSIS OF POTENTIAL SOURCE DOCUMENTS')
    print('=' * 90)
    
    print(f'Analyzing {len(search_results["potential_sources"])} potential source documents:')
    
    for i, source in enumerate(search_results['potential_sources'], 1):
        print(f'\n📖 POTENTIAL SOURCE {i}:')
        print('-' * 30)
        print(f'Search Query: {source["query"]}')
        print(f'Relevance Score: {source["score"]}')
        print(f'Key Terms: {", ".join(source["terms"][:10])}')
        
        if source['chroniclers']:
            print(f'Chroniclers Mentioned: {", ".join(source["chroniclers"])}')
        
        if source['snippets']:
            print('Key Text Snippets:')
            for j, snippet in enumerate(source['snippets'][:3], 1):
                print(f'  {j}. {snippet[:180]}...')
        
        print(f'Saved HTML File: {source["file"]}')
        print()

# Final conclusions and recommendations
print('\n' + '=' * 90)
print('FINAL ANALYSIS AND RECOMMENDATIONS')
print('=' * 90)

print('🎯 SEARCH OBJECTIVE ASSESSMENT:')
print('Finding 17th century documents recording:')
print('  ✓ Spectral army apparitions in Ireland')
print('  ✓ Unusual supernatural events')
print('  ✓ Chronicles/historical accounts from the period')
print('  ✓ Potential M.R. James source material')
print()

# Calculate overall success metrics
success_metrics = {
    'total_searches': len(search_results['search_queries']),
    'successful_results': len([f for f in search_results['findings'] if f['relevance_score'] > 0]),
    'high_relevance_hits': len([f for f in search_results['findings'] if f['relevance_score'] >= 20]),
    'chroniclers_identified': len(set(search_results['historians_identified'])),

}

print('📊 SEARCH PERFORMANCE METRICS:')
for metric, value in success_metrics.items():
    print(f'  • {metric.replace("_", " ").title()}: {value}')

search_results['analysis']['success_metrics'] = success_metrics

if search_results['potential_sources']:
    print('\n✅ PROMISING LEADS IDENTIFIED:')
    print('Based on search results, investigate these areas further:')
    
    # Identify most promising research directions
    if search_results.get('analysis', {}).get('top_chroniclers'):
        top_chronicler = list(search_results['analysis']['top_chroniclers'].keys())[0]
        print(f'  1. Focus on works by {top_chronicler} - most frequently mentioned')

with open(results_file, 'w', encoding='utf-8') as f:
    json.dump(search_results, f, indent=2, ensure_ascii=False)

    f.write(f'Chroniclers/historians found: {len(set(search_results["historians_identified"]))}\n\n')
    
    if search_results.get('analysis', {}).get('top_chroniclers'):
        f.write('TOP CHRONICLERS/HISTORIANS MENTIONED:\n')
        for chronicler, count in search_results['analysis']['top_chroniclers'].items():
            f.write(f'  • {chronicler}: {count} mentions\n')
        f.write('\n')
    
    f.write('NEXT RESEARCH STEPS:\n')
    f.write('1. Review all saved HTML files for specific document titles\n')
    f.write('2. Follow up on high-relevance search results\n')
    f.write('3. Investigate identified chroniclers\' complete works\n')
    f.write('4. Search academic databases for M.R. James source studies\n')
    f.write('5. Examine Irish manuscript collections for supernatural accounts\n')

print(f'📋 SUMMARY REPORT SAVED TO: {summary_file}')

print('\n🔍 FILES CREATED FOR FURTHER ANALYSIS:')
print(f'  • Main results: {results_file}')
print(f'  • Summary report: {summary_file}')
print(f'  • HTML search results: {len(search_results["findings"])} files in workspace/')

print('\n=== 17TH CENTURY IRISH SUPERNATURAL CHRONICLES SEARCH COMPLETE ===')
