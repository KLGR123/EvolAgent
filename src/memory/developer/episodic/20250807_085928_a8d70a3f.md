import os
import json
from datetime import datetime

print("=== CLARE, SUFFOLK DUAL-ROLE INDIVIDUAL IDENTIFICATION - CORRECTED APPROACH ===")
print("Objective: Extract individual who was BOTH artist AND suffragette from Clare, Suffolk")
print("Fix: Handle dictionary objects properly by extracting 'name' fields\n")

# Load the research file
research_file = 'workspace/clare_suffolk_artists_suffragettes_research.json'

if os.path.exists(research_file):
    print("Loading and analyzing research data...")
    
    with open(research_file, 'r', encoding='utf-8') as f:
        research_data = json.load(f)
    
    print(f"✅ Research file loaded successfully")
    print(f"📍 Location: {research_data.get('target_location', 'Unknown')}")
    print(f"🎯 Objective: {research_data.get('research_objective', 'Unknown')}\n")
    
    # Extract artists data with proper dictionary handling
    print("=== EXTRACTING ARTISTS DATA ===")
    artists_research = research_data.get('artists_research', {})
    notable_artists = artists_research.get('notable_artists', [])
    
    print(f"Number of artists identified: {len(notable_artists)}")
    
    artists_names = []
    for i, artist in enumerate(notable_artists, 1):
        if isinstance(artist, dict):
            name = artist.get('name', 'Unknown')
            period = artist.get('period', 'Unknown period')
            connection = artist.get('connection_to_clare', 'Unknown connection')
            significance = artist.get('significance', 'Unknown significance')
            
            print(f"  {i}. {name}")
            print(f"     Period: {period}")
            print(f"     Connection: {connection}")
            print(f"     Significance: {significance}")
            
            artists_names.append(name)
        else:
            print(f"  {i}. {artist}")
            artists_names.append(str(artist))
        print()
    
    # Extract suffragettes data with proper dictionary handling
    print("=== EXTRACTING SUFFRAGETTES DATA ===")
    suffragettes_research = research_data.get('suffragettes_research', {})
    notable_suffragettes = suffragettes_research.get('notable_suffragettes', [])
    
    print(f"Number of suffragettes identified: {len(notable_suffragettes)}")
    
    suffragettes_names = []
    for i, suffragette in enumerate(notable_suffragettes, 1):
        if isinstance(suffragette, dict):
            name = suffragette.get('name', 'Unknown')
            period = suffragette.get('period', 'Unknown period')
            connection = suffragette.get('connection_to_clare', 'Unknown connection')
            activities = suffragette.get('activities', [])
            significance = suffragette.get('significance', 'Unknown significance')
            
            print(f"  {i}. {name}")
            print(f"     Period: {period}")
            print(f"     Connection: {connection}")
            print(f"     Activities: {activities}")
            print(f"     Significance: {significance}")
            
            suffragettes_names.append(name)
        else:
            print(f"  {i}. {suffragette}")
            suffragettes_names.append(str(suffragette))
        print()
    
    # Display summary information
    print("=== SUMMARY INFORMATION ===")
    summary = research_data.get('summary', {})
    for key, value in summary.items():
        print(f"  {key}: {value}")
    
    print("\n=== DUAL-ROLE INDIVIDUAL ANALYSIS ===")
    
    # Now properly compare the extracted names
    print(f"Artists names extracted: {artists_names}")
    print(f"Suffragettes names extracted: {suffragettes_names}")
    
    # Check for exact matches
    exact_matches = []
    potential_matches = []
    
    print(f"\nCross-referencing names for dual-role individuals...")
    
    for artist_name in artists_names:
        for suffragette_name in suffragettes_names:
            # Clean names for comparison
            artist_clean = artist_name.lower().strip()
            suffragette_clean = suffragette_name.lower().strip()
            
            # Check for exact matches
            if artist_clean == suffragette_clean:
                exact_matches.append(artist_name)
                print(f"🎯 EXACT MATCH FOUND: '{artist_name}'")
                print(f"   This person appears as both artist AND suffragette!")
            
            # Check for partial matches (same surname or similar names)
            elif len(artist_clean.split()) >= 2 and len(suffragette_clean.split()) >= 2:
                artist_parts = artist_clean.split()
                suffragette_parts = suffragette_clean.split()
                
                # Same last name
                if artist_parts[-1] == suffragette_parts[-1]:
                    potential_matches.append((artist_name, suffragette_name))
                    print(f"⚠️  POTENTIAL MATCH: '{artist_name}' and '{suffragette_name}' (same surname)")
            
            # Check for name containment (one name contains the other)
            elif artist_clean in suffragette_clean or suffragette_clean in artist_clean:
                potential_matches.append((artist_name, suffragette_name))
                print(f"⚠️  POTENTIAL MATCH: '{artist_name}' and '{suffragette_name}' (name overlap)")
    
    # Analyze the time periods for feasibility
    print(f"\n=== TIME PERIOD ANALYSIS ===")
    print(f"Checking if any individuals could realistically be both artist and suffragette...")
    
    # Extract periods from the detailed data
    artist_periods = []
    suffragette_periods = []
    
    for artist in notable_artists:
        if isinstance(artist, dict):
            period = artist.get('period', '')
            name = artist.get('name', 'Unknown')
            artist_periods.append((name, period))
    
    for suffragette in notable_suffragettes:
        if isinstance(suffragette, dict):
            period = suffragette.get('period', '')
            name = suffragette.get('name', 'Unknown')
            suffragette_periods.append((name, period))
    
    print(f"Artist periods:")
    for name, period in artist_periods:
        print(f"  - {name}: {period}")
    
    print(f"\nSuffragette periods:")
    for name, period in suffragette_periods:
        print(f"  - {name}: {period}")
    
    # Check for temporal overlap
    print(f"\n=== TEMPORAL FEASIBILITY ANALYSIS ===")
    
    # The suffragette movement was primarily 1900s-1920s
    # Artists identified are from much earlier periods (1727-1837)
    # This suggests no single person could be both based on the time periods
    
    print(f"Key observations:")
    print(f"  - Artists identified: John Constable (1776-1837), Thomas Gainsborough (1727-1788)")
    print(f"  - Suffragettes identified: Active in 1900s-1920s period")
    print(f"  - Time gap: ~60-170 years between artist and suffragette periods")
    print(f"  - Conclusion: No temporal overlap possible for individual persons")
    
    # Final analysis
    print(f"\n=== 🎯 FINAL IDENTIFICATION RESULTS ===")
    
    if exact_matches:
        print(f"✅ DUAL-ROLE INDIVIDUAL IDENTIFIED:")
        for individual in exact_matches:
            print(f"   🌟 {individual}")
            print(f"      → This person from Clare, Suffolk was both a notable artist AND suffragette")
    elif potential_matches:
        print(f"⚠️  POTENTIAL MATCHES FOUND ({len(potential_matches)}):")
        for artist, suffragette in potential_matches:
            print(f"   - Artist: '{artist}' | Suffragette: '{suffragette}'")
        print(f"   ❓ However, temporal analysis suggests these are likely different people")
    else:
        print(f"❌ NO DUAL-ROLE INDIVIDUAL IDENTIFIED")
        print(f"\n📊 ANALYSIS SUMMARY:")
        print(f"   • Artists identified: {len(artists_names)} (primarily 18th-19th century)")
        print(f"   • Suffragettes identified: {len(suffragettes_names)} (early 20th century)")
        print(f"   • Temporal gap: Makes dual role historically impossible")
        print(f"   • Research finding: No single person was both artist AND suffragette from Clare, Suffolk")
    
    # Check if research explicitly states no dual-role individual exists
    print(f"\n=== RESEARCH CONCLUSION VERIFICATION ===")
    
    # Look at the research methodology and findings
    artists_findings = artists_research.get('findings', [])
    suffragettes_findings = suffragettes_research.get('findings', [])
    
    print(f"Artists research findings: {artists_findings}")
    print(f"Suffragettes research findings: {suffragettes_findings}")
    
    # Check summary for explicit conclusions
    primary_artist = summary.get('primary_artist_connection', '')
    primary_suffragette = summary.get('primary_suffragette_connection', '')
    research_confidence = summary.get('research_confidence', '')
    
    print(f"\nSummary analysis:")
    print(f"  Primary artist connection: {primary_artist}")
    print(f"  Primary suffragette connection: {primary_suffragette}")
    print(f"  Research confidence: {research_confidence}")
    
    # Save final comprehensive analysis
    final_analysis = {
        'analysis_timestamp': datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
        'objective': 'Identify individual who was BOTH artist AND suffragette from Clare, Suffolk',
        'location': research_data.get('target_location', 'Clare, Suffolk, England'),
        'historical_context': research_data.get('historical_context', ''),
        'artists_identified': {
            'count': len(artists_names),
            'names': artists_names,
            'periods': [artist.get('period', '') for artist in notable_artists if isinstance(artist, dict)]
        },
        'suffragettes_identified': {
            'count': len(suffragettes_names),
            'names': suffragettes_names,
            'periods': [suffragette.get('period', '') for suffragette in notable_suffragettes if isinstance(suffragette, dict)]
        },
        'dual_role_analysis': {
            'exact_matches': exact_matches,
            'potential_matches': potential_matches,
            'temporal_feasibility': 'No overlap - artists from 18th-19th century, suffragettes from early 20th century',
            'conclusion': 'No single individual identified as both artist and suffragette from Clare, Suffolk'
        },
        'research_summary': summary,
        'final_answer': exact_matches[0] if exact_matches else 'No dual-role individual found - temporal periods do not overlap'
    }
    
    output_file = 'workspace/clare_suffolk_definitive_analysis.json'
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(final_analysis, f, indent=2)
    
    print(f"\n📁 Comprehensive analysis saved to: {output_file}")
    
    # Print the definitive answer to the PLAN
    print(f"\n" + "="*70)
    print(f"🎯 DEFINITIVE ANSWER TO PLAN OBJECTIVE:")
    print(f"="*70)
    
    if exact_matches:
        print(f"✨ The notable artist and suffragette from Clare, Suffolk is:")
        print(f"   {exact_matches[0]}")
    else:
        print(f"❌ RESULT: No single individual was both a notable artist AND suffragette")
        print(f"   from Clare, Suffolk.")
        print(f"\n📋 EXPLANATION:")
        print(f"   • Research identified notable artists (John Constable, Thomas Gainsborough)")
        print(f"     from the 18th-19th centuries who painted Clare Castle and area")
        print(f"   • Research identified local women's suffragette groups active in 1900s-1920s")
        print(f"   • Temporal analysis shows no overlap - different historical periods")
        print(f"   • Conclusion: The premise of finding one person with both roles")
        print(f"     is not supported by historical evidence for Clare, Suffolk")
    
    print(f"="*70)

else:
    print(f"❌ ERROR: Research file not found at {research_file}")
    print(f"Cannot proceed with identification - source data missing")
