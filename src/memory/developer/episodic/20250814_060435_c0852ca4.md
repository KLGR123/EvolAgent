# Advanced matching functions
def normalize_name_advanced(name):
    """Advanced name normalization"""
    # Remove all non-alphabetic characters and convert to lowercase
    normalized = re.sub(r'[^a-zA-Z]', '', name.lower())
    return normalized

def phonetic_similarity(name1, name2):
    """Check phonetic similarity using various methods"""
    n1 = normalize_name_advanced(name1)
    n2 = normalize_name_advanced(name2)
    
    # Double metaphone-like simplification for Chinese transliterations
    def simplify_chinese_phonetics(name):
        # Common Chinese romanization variations
        replacements = {
            'ch': 'c', 'zh': 'z', 'sh': 's', 'th': 't',
            'hs': 's', 'ts': 'z', 'tz': 'z', 'tz': 'z',
            'iang': 'ang', 'iong': 'ong', 'uang': 'ang'
        }
        result = name
        for old, new in replacements.items():
            result = result.replace(old, new)
        return result
    
    s1 = simplify_chinese_phonetics(n1)
    s2 = simplify_chinese_phonetics(n2)
    
    if s1 == s2:
        return 'phonetic_exact', 0.9
    
    # Check if one is contained in the other after phonetic simplification
    if s1 in s2 or s2 in s1:
        return 'phonetic_contains', 0.7
    
    return 'no_phonetic', 0.0

def advanced_similarity_check(contributor, transliteration):
    """Comprehensive similarity checking"""
    matches = []
    
    # Normalize both names
    c_norm = normalize_name_advanced(contributor)
    t_norm = normalize_name_advanced(transliteration)
    
    # 1. Exact match
    if c_norm == t_norm:
        matches.append(('exact', 1.0))
    
    # 2. Contains match
    if c_norm in t_norm or t_norm in c_norm:
        overlap = min(len(c_norm), len(t_norm)) / max(len(c_norm), len(t_norm))
        matches.append(('contains', 0.8 * overlap))
    
    # 3. Phonetic similarity
    phon_type, phon_score = phonetic_similarity(contributor, transliteration)
    if phon_score > 0:
        matches.append((phon_type, phon_score))
    
    # 4. Split name analysis
    c_parts = re.split(r'[\s\-_]', contributor.lower())
    t_parts = re.split(r'[\s\-_]', transliteration.lower())
    
    part_matches = 0
    total_parts = max(len(c_parts), len(t_parts))
    
    for c_part in c_parts:
        for t_part in t_parts:
            c_part_norm = normalize_name_advanced(c_part)
            t_part_norm = normalize_name_advanced(t_part)
            
            if len(c_part_norm) > 2 and len(t_part_norm) > 2:
                if c_part_norm == t_part_norm:
                    part_matches += 1
                elif c_part_norm in t_part_norm or t_part_norm in c_part_norm:
                    part_matches += 0.7
    
    if part_matches > 0:
        part_score = (part_matches / total_parts) * 0.6
        matches.append(('part_match', part_score))
    
    # 5. Levenshtein-like distance for short sequences
    if len(c_norm) >= 4 and len(t_norm) >= 4:
        common_chars = set(c_norm) & set(t_norm)
        if len(common_chars) >= 3:
            char_score = len(common_chars) / max(len(set(c_norm)), len(set(t_norm)))
            if char_score > 0.5:
                matches.append(('char_similarity', char_score * 0.5))
    
    # Return best match
    if matches:
        best_match = max(matches, key=lambda x: x[1])
        return best_match[0], best_match[1]
    
    return 'no_match', 0.0
