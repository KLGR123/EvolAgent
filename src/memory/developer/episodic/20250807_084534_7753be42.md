### Development Step 4: Identify Suffolk’s 17th-Century Ash Tree Spider Curse and Irish Spectral Army Behind ‘The Ash Tree’

**Description**: Research M.R. James's short story 'The Ash Tree' to identify the Suffolk location that inspired it, focusing on finding the real 17th century arachnid event involving an ash tree-linked spider curse that served as the story's basis. Search for information about the historical incident, the specific Suffolk town or village where it occurred, and any documented sources from that period that recorded both this spider-related event and a spectral army apparition in Ireland.

**Use Cases**:
- Literary research on gothic fiction: automated extraction and analysis of “The Ash Tree” text to identify real Suffolk locations for an academic paper on M.R. James’s use of regional topography.
- Local history society project: parsing the story for 17th-century spider curse references and matching manor names via regex to parish records in Bury St Edmunds for a museum exhibit.
- Cultural tourism development: building an interactive heritage trail app that maps extracted Suffolk village and hall names from “The Ash Tree” to actual sites and archives.
- Digital humanities initiative: large-scale cleaning and categorization of supernatural motifs (spiders, curses, spectral armies) across multiple ghost stories, exporting keyword contexts in JSON for comparative studies.
- Genealogical research tool: cross-referencing spectral army and witchcraft mentions in “The Ash Tree” with Irish military muster rolls and trial transcripts to trace ancestral links.
- Educational publishing platform: auto-generating annotated e-book editions with footnotes on church, manor and 17th-century historical references extracted from the story for literature curricula.
- E-book production workflow: integrating the analysis script to automatically tag and catalog place names, historical period terms, and supernatural elements as metadata in digital publications.

```
import os
import requests
from bs4 import BeautifulSoup
import json
import re
import time

print('=== DIRECT APPROACH: LOCATING AND ANALYZING "THE ASH TREE" ===\n')
print('Objective: Successfully extract the full story text and identify Suffolk location clues')
print('Strategy: Use multiple targeted approaches with robust error handling\n')

# Ensure workspace exists
workspace = 'workspace'
os.makedirs(workspace, exist_ok=True)

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.5',
    'Connection': 'keep-alive'
}

print('=== STEP 1: DIRECT PROJECT GUTENBERG PLAIN TEXT ACCESS ===\n')

# Try the plain text version first as it's easier to parse
gutenberg_txt_url = 'https://www.gutenberg.org/files/8486/8486-0.txt'
print(f'Accessing plain text version: {gutenberg_txt_url}')

story_text = None
story_source = None

try:
    response = requests.get(gutenberg_txt_url, headers=headers, timeout=30)
    print(f'Response status: {response.status_code}')
    
    if response.status_code == 200:
        full_text = response.text
        print(f'Full document length: {len(full_text):,} characters')
        
        # Save the full Project Gutenberg text for reference
        full_text_file = os.path.join(workspace, 'gutenberg_full_text.txt')
        with open(full_text_file, 'w', encoding='utf-8') as f:
            f.write(full_text)
        print(f'✓ Full Gutenberg text saved to: {full_text_file}')
        
        # Look for "THE ASH-TREE" in the text
        print('\nSearching for story boundaries...')
        
        # Find the story start
        story_markers = ['THE ASH-TREE', 'The Ash-tree', 'THE ASH TREE']
        story_start_pos = None
        used_marker = None
        
        for marker in story_markers:
            pos = full_text.find(marker)
            if pos != -1:
                story_start_pos = pos
                used_marker = marker
                print(f'✓ Found story marker "{marker}" at position {pos}')
                break
        
        if story_start_pos is not None:
            # Find the next story or end marker
            next_story_markers = [
                'THE SCHOOL STORY',
                'A VIEW FROM A HILL',
                'NUMBER 13',
                'COUNT MAGNUS',
                'OH, WHISTLE',
                '*** END OF THE PROJECT GUTENBERG EBOOK',
                'End of the Project Gutenberg EBook'
            ]
            
            story_end_pos = len(full_text)
            end_marker_used = None
            
            for end_marker in next_story_markers:
                pos = full_text.find(end_marker, story_start_pos + 100)
                if pos != -1 and pos < story_end_pos:
                    story_end_pos = pos
                    end_marker_used = end_marker
            
            print(f'Story boundaries: {story_start_pos} to {story_end_pos}')
            if end_marker_used:
                print(f'End marker used: {end_marker_used}')
            
            # Extract the story
            raw_story = full_text[story_start_pos:story_end_pos]
            
            # Clean up the text
            lines = raw_story.split('\n')
            cleaned_lines = []
            
            skip_patterns = ['***', 'Produced by', 'Updated:', 'Language:', 'Character set']
            
            for line in lines:
                line = line.strip()
                if line and not any(pattern in line for pattern in skip_patterns):
                    cleaned_lines.append(line)
            
            # Join paragraphs properly
            story_text = '\n\n'.join(cleaned_lines)
            
            print(f'✓ Extracted story: {len(story_text):,} characters')
            
            # Save the extracted story
            story_file = os.path.join(workspace, 'the_ash_tree_story.txt')
            with open(story_file, 'w', encoding='utf-8') as f:
                f.write('"The Ash Tree" by M.R. James\n')
                f.write(f'Source: Project Gutenberg Plain Text\n')
                f.write(f'Start marker: {used_marker}\n')
                f.write(f'End marker: {end_marker_used or "End of document"}\n')
                f.write(f'Extracted: {time.strftime("%Y-%m-%d %H:%M:%S")}\n')
                f.write('=' * 80 + '\n\n')
                f.write(story_text)
            
            print(f'✓ Story saved to: {story_file}')
            story_source = 'Project Gutenberg Plain Text'
            
            # Show story preview
            print('\n=== STORY PREVIEW ===\n')
            preview_length = 600
            preview = story_text[:preview_length] + '...' if len(story_text) > preview_length else story_text
            print(preview)
            
        else:
            print('❌ Could not locate story start marker')
            
except Exception as e:
    print(f'❌ Error accessing Project Gutenberg: {str(e)}')

print('\n=== STEP 2: COMPREHENSIVE LOCATION AND HISTORICAL ANALYSIS ===\n')

if story_text:
    print('✅ Story text available - proceeding with detailed analysis')
    print(f'Analyzing {len(story_text):,} characters of story content\n')
    
    # Create comprehensive search categories
    search_categories = {
        'suffolk_locations': {
            'keywords': ['suffolk', 'bury st edmunds', 'bury', 'ipswich', 'felixstowe', 'lowestoft', 
                        'newmarket', 'sudbury', 'haverhill', 'aldeburgh', 'southwold', 'woodbridge',
                        'stowmarket', 'beccles', 'brandon', 'mildenhall', 'clare', 'lavenham'],
            'description': 'Direct Suffolk place references'
        },
        'neighboring_areas': {
            'keywords': ['norfolk', 'essex', 'cambridge', 'cambridgeshire', 'hertfordshire', 
                        'norwich', 'colchester', 'chelmsford', 'hertford'],
            'description': 'Neighboring county references'
        },
        'property_types': {
            'keywords': ['hall', 'manor', 'house', 'court', 'grange', 'priory', 'abbey', 'rectory',
                        'vicarage', 'castle', 'tower', 'lodge', 'mansion', 'estate'],
            'description': 'Types of properties/buildings'
        },
        'ecclesiastical': {
            'keywords': ['church', 'cathedral', 'chapel', 'parish', 'rector', 'vicar', 'curate',
                        'bishop', 'archdeacon', 'churchyard', 'cemetery', 'grave', 'tomb'],
            'description': 'Religious/church references'
        },
        'historical_period': {
            'keywords': ['17th century', '1600s', 'sixteen', 'charles', 'cromwell', 'civil war',
                        'restoration', 'commonwealth', 'stuart', 'puritan', 'cavalier', 'roundhead',
                        '1640', '1650', '1660', '1670', '1680', '1690'],
            'description': '17th century historical references'
        },
        'supernatural_elements': {
            'keywords': ['spider', 'arachnid', 'web', 'curse', 'witch', 'witchcraft', 'trial',
                        'hanging', 'execution', 'gallows', 'condemned', 'accused', 'sorcery'],
            'description': 'Supernatural/witchcraft elements'
        },
        'tree_ash_references': {
            'keywords': ['ash', 'tree', 'oak', 'elm', 'branch', 'trunk', 'root', 'leaf', 'bark',
                        'grove', 'wood', 'forest'],
            'description': 'Tree and ash-specific references'
        },
        'ireland_connection': {
            'keywords': ['ireland', 'irish', 'dublin', 'cork', 'belfast', 'spectral', 'army',
                        'soldiers', 'battle', 'war', 'regiment', 'troops'],
            'description': 'Ireland/spectral army connections'
        }
    }
    
    story_lower = story_text.lower()
    analysis_results = {}
    all_findings = []
    
    # Analyze each category
    for category_name, category_data in search_categories.items():
        keywords = category_data['keywords']
        description = category_data['description']
        category_findings = {}
        
        print(f'Analyzing {description.lower()}...')
        
        for keyword in keywords:
            if keyword in story_lower:
                # Find all occurrences
                positions = []
                start = 0
                while True:
                    pos = story_lower.find(keyword, start)
                    if pos == -1:
                        break
                    positions.append(pos)
                    start = pos + 1
                
                # Extract contexts
                contexts = []
                for pos in positions:
                    context_start = max(0, pos - 200)
                    context_end = min(len(story_text), pos + 200)
                    context = story_text[context_start:context_end].replace('\n', ' ').strip()
                    contexts.append({
                        'position': pos,
                        'context': context
                    })
                
                category_findings[keyword] = {
                    'count': len(positions),
                    'contexts': contexts
                }
                
                # Add to all findings for summary
                for ctx in contexts:
                    all_findings.append({
                        'category': category_name,
                        'keyword': keyword,
                        'position': ctx['position'],
                        'context': ctx['context']
                    })
        
        if category_findings:
            analysis_results[category_name] = category_findings
            print(f'  ✓ Found {len(category_findings)} terms in this category')
        else:
            print(f'  - No matches in this category')
    
    print(f'\n=== DETAILED FINDINGS BY CATEGORY ===\n')
    
    for category_name, findings in analysis_results.items():
        category_desc = search_categories[category_name]['description']
        print(f'{category_desc.upper()}:')
        
        for keyword, data in findings.items():
            print(f'  • "{keyword}" - {data["count"]} occurrence(s)')
            
            # Show first context for each keyword
            if data['contexts']:
                context = data['contexts'][0]['context']
                print(f'    Context: ...{context}...')
            print()
    
    print('=== SEARCHING FOR SPECIFIC PLACE NAMES WITH REGEX ===\n')
    
    # Enhanced regex patterns for place names
    place_patterns = [
        (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Hall|Manor|House|Court|Grange|Priory|Abbey)', 'Estate/Manor names'),
        (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Church|Cathedral|Chapel)', 'Religious buildings'),
        (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Village|Parish|Town)', 'Settlements'),
        (r'the\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(estate|property|lands)', 'Property references'),
        (r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+in\s+(Suffolk|Norfolk|Essex|Cambridgeshire)', 'County-specific places'),
        (r'Sir\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)', 'Named individuals (potential historical figures)')
    ]
    
    place_matches = []
    
    for pattern, description in place_patterns:
        matches = list(re.finditer(pattern, story_text, re.IGNORECASE))
        if matches:
            print(f'{description}: {len(matches)} match(es)')
            
            for match in matches:
                if len(match.groups()) >= 2:
                    place_name = match.group(1).strip()
                    place_type = match.group(2).strip()
                    full_match = match.group(0)
                else:
                    place_name = match.group(1).strip()
                    place_type = 'Individual'
                    full_match = match.group(0)
                
                position = match.start()
                
                # Extract context
                context_start = max(0, position - 150)
                context_end = min(len(story_text), position + 150)
                context = story_text[context_start:context_end].replace('\n', ' ').strip()
                
                place_info = {
                    'place_name': place_name,
                    'place_type': place_type,
                    'full_match': full_match,
                    'pattern_description': description,
                    'context': context,
                    'position': position
                }
                
                place_matches.append(place_info)
                
                print(f'  → "{place_name}" ({place_type})')
                print(f'    Full match: "{full_match}"')
                print(f'    Context: ...{context}...')
                print()
    
    # Save comprehensive analysis
    comprehensive_analysis = {
        'extraction_info': {
            'story_source': story_source,
            'story_length': len(story_text),
            'analysis_timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
            'total_categories_analyzed': len(search_categories),
            'categories_with_matches': len(analysis_results)
        },
        'keyword_analysis': analysis_results,
        'place_name_matches': place_matches,
        'all_findings_summary': {
            'total_keyword_matches': len(all_findings),
            'total_place_matches': len(place_matches),
            'categories_found': list(analysis_results.keys())
        },
        'search_categories': search_categories
    }
    
    analysis_file = os.path.join(workspace, 'ash_tree_suffolk_analysis.json')
    with open(analysis_file, 'w', encoding='utf-8') as f:
        json.dump(comprehensive_analysis, f, indent=2, ensure_ascii=False)
    
    print(f'✓ Comprehensive analysis saved to: {analysis_file}')
    
    # Summary of most important findings
    print('\n=== KEY FINDINGS SUMMARY ===\n')
    
    if 'suffolk_locations' in analysis_results:
        suffolk_refs = analysis_results['suffolk_locations']
        print(f'🎯 DIRECT SUFFOLK REFERENCES: {len(suffolk_refs)} found')
        for term, data in suffolk_refs.items():
            print(f'  - "{term}": {data["count"]} occurrence(s)')
        print()
    
    if 'historical_period' in analysis_results:
        historical_refs = analysis_results['historical_period']
        print(f'📅 17TH CENTURY REFERENCES: {len(historical_refs)} found')
        for term, data in historical_refs.items():
            print(f'  - "{term}": {data["count"]} occurrence(s)')
        print()
    
    if 'supernatural_elements' in analysis_results:
        supernatural_refs = analysis_results['supernatural_elements']
        print(f'🕷️ SUPERNATURAL ELEMENTS: {len(supernatural_refs)} found')
        for term, data in supernatural_refs.items():
            print(f'  - "{term}": {data["count"]} occurrence(s)')
        print()
    
    if place_matches:
        print(f'🏛️ SPECIFIC PLACE NAMES: {len(place_matches)} identified')
        for place in place_matches[:5]:  # Show first 5
            print(f'  - {place["place_name"]} ({place["place_type"]})')
        if len(place_matches) > 5:
            print(f'  ... and {len(place_matches) - 5} more')
        print()
    
    if 'ireland_connection' in analysis_results:
        ireland_refs = analysis_results['ireland_connection']
        print(f'🇮🇪 IRELAND CONNECTIONS: {len(ireland_refs)} found')
        for term, data in ireland_refs.items():
            print(f'  - "{term}": {data["count"]} occurrence(s)')
        print()
    
else:
    print('❌ No story text available for analysis')
    print('Attempting alternative research approach...')

print('\n=== FINAL STATUS REPORT ===\n')

if story_text:
    print('✅ SUCCESS: "The Ash Tree" story successfully extracted and analyzed')
    print(f'📖 Story length: {len(story_text):,} characters')
    if 'analysis_results' in locals():
        print(f'🔍 Categories with matches: {len(analysis_results)}')
        print(f'📍 Total keyword occurrences: {len(all_findings)}')
        print(f'🏛️ Specific place names identified: {len(place_matches)}')
        
        # Highlight the most promising leads for Suffolk location identification
        print('\n🎯 MOST PROMISING LEADS FOR SUFFOLK LOCATION:')
        
        # Check for direct Suffolk mentions
        if 'suffolk_locations' in analysis_results:
            print('  ✓ Direct Suffolk location references found in story')
        
        # Check for specific manor/hall names
        manor_places = [p for p in place_matches if p['place_type'].lower() in ['hall', 'manor', 'house', 'court']]
        if manor_places:
            print(f'  ✓ {len(manor_places)} manor/hall names identified - potential real locations')
        
        # Check for historical period alignment
        if 'historical_period' in analysis_results:
            print('  ✓ 17th century historical references found - period alignment confirmed')
        
        # Check for supernatural elements
        if 'supernatural_elements' in analysis_results:
            print('  ✓ Spider/witch/curse elements found - supernatural theme confirmed')
else:
    print('❌ FAILED: Could not extract story text')
    print('📋 Recommendation: Try manual download and analysis')

print(f'\n📁 Workspace files created:')
for file in sorted(os.listdir(workspace)):
    filepath = os.path.join(workspace, file)
    size = os.path.getsize(filepath)
    print(f'  - {file} ({size:,} bytes)')
```
