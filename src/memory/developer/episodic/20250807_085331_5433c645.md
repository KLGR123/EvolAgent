import requests
import json
import time
from bs4 import BeautifulSoup
from urllib.parse import quote_plus

print('=== COMPREHENSIVE SEARCH: 17TH CENTURY IRISH CHRONICLES DOCUMENTING SPECTRAL ARMIES ===')

            print(f'  • {key}: {previous_data[key]}')

else:

search_queries = [
    # Direct chronicle searches

    '"Geoffrey Keating" "Foras Feasa ar Éirinn" supernatural events Ireland',
    '"Annals of Ulster" 17th century ghostly apparitions Ireland',

    # General supernatural chronicle searches
    'Irish chronicles supernatural events 17th century historical records',

    'Celtic supernatural folklore 17th century chronicles Ireland'
]

print(f'Executing {len(search_queries)} targeted searches for Irish chronicles:\n')

# Headers for web requests
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
    'Accept-Language': 'en-US,en;q=0.9',
    'Connection': 'keep-alive'
}

# Initialize comprehensive results storage
chronicle_search_results = {
    'search_timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
    'objective': 'Find 17th century Irish chronicles documenting spectral army apparitions and supernatural events',
    'method': 'Comprehensive web search using multiple engines',
    'queries_executed': search_queries,
    'search_results': [],
    'chronicle_candidates': [],
    'supernatural_events': [],
    'analysis_summary': {}
}

def analyze_search_content(html_content, query, search_engine):
    """Extract and analyze search results from HTML content"""
    soup = BeautifulSoup(html_content, 'html.parser')
    
    # Remove script and style elements
    for script in soup(["script", "style"]):
        script.decompose()
    
    # Get text content
    text_content = soup.get_text().lower()
    
    # Define key terms for analysis
    chronicle_terms = ['chronicle', 'chronicles', 'annals', 'historical record', 'manuscript', 'historical account']
    time_terms = ['17th century', 'seventeenth century', '1600', '1650', '1680', '1690']
    irish_terms = ['irish', 'ireland', 'celtic', 'gaelic', 'dublin']
    supernatural_terms = ['spectral army', 'phantom army', 'ghostly host', 'supernatural army', 'apparition', 'phantom', 'spectral', 'ghostly', 'ghost', 'spirit']
    
    # Count relevant terms
    chronicle_count = sum(text_content.count(term) for term in chronicle_terms)
    time_count = sum(text_content.count(term) for term in time_terms)
    irish_count = sum(text_content.count(term) for term in irish_terms)
    supernatural_count = sum(text_content.count(term) for term in supernatural_terms)
    
    relevance_score = chronicle_count * 3 + time_count * 2 + irish_count * 2 + supernatural_count * 2
    
    # Extract potential result snippets
    result_snippets = []
    
    # Look for result containers (generic approach)
    potential_results = soup.find_all(['div', 'article', 'section'], class_=lambda x: x and any(term in str(x).lower() for term in ['result', 'search', 'content']))
    
    if not potential_results:
        # Fallback: look for any text blocks that might contain results
        potential_results = soup.find_all(['p', 'div'], string=lambda text: text and len(text.strip()) > 50)
    
    for element in potential_results[:10]:
        if element and element.get_text():
            snippet_text = element.get_text().strip()
            if 50 <= len(snippet_text) <= 500:
                snippet_lower = snippet_text.lower()
                
                # Check if snippet contains relevant terms
                snippet_relevance = 0
                matched_terms = []
                
                for term_list, weight in [(chronicle_terms, 3), (time_terms, 2), (irish_terms, 2), (supernatural_terms, 2)]:
                    for term in term_list:
                        if term in snippet_lower:
                            snippet_relevance += weight
                            matched_terms.append(term)
                
                if snippet_relevance >= 4:  # Only include highly relevant snippets
                    result_snippets.append({
                        'text': snippet_text,
                        'relevance_score': snippet_relevance,
                        'matched_terms': list(set(matched_terms))
                    })
    
    return {
        'query': query,
        'search_engine': search_engine,
        'relevance_score': relevance_score,
        'term_counts': {
            'chronicle': chronicle_count,
            'time': time_count,
            'irish': irish_count,
            'supernatural': supernatural_count
        },
        'result_snippets': result_snippets[:5]  # Top 5 most relevant snippets
    }

print('🔍 EXECUTING COMPREHENSIVE CHRONICLE SEARCHES:')
print('=' * 60)

# Execute searches using multiple approaches
for i, query in enumerate(search_queries, 1):
    print(f'\nSearch {i}/{len(search_queries)}: {query}')
    print('-' * 70)

        ('Google', f'https://www.google.com/search?q={quote_plus(query)}'),
        ('Bing', f'https://www.bing.com/search?q={quote_plus(query)}'),
        ('DuckDuckGo', f'https://html.duckduckgo.com/html/?q={quote_plus(query)}')
    ]
    
    for engine, url in search_attempts:
        try:
            print(f'  Trying {engine}...')
            response = requests.get(url, headers=headers, timeout=20)
            
            if response.status_code == 200:
                print(f'    ✅ {engine}: Status {response.status_code}')
                
                # Save raw HTML
                filename = f'search_{i}_{engine.lower()}_{query[:30].replace(" ", "_").replace('"', "")}.html'
                filepath = os.path.join('workspace', filename)
                
                with open(filepath, 'w', encoding='utf-8') as f:
                    f.write(response.text)
                
                print(f'    Saved: {filename}')
                
                # Analyze content
                analysis = analyze_search_content(response.text, query, engine)
                analysis['html_file'] = filepath
                
                print(f'    Relevance score: {analysis["relevance_score"]}')
                print(f'    Terms found: Chronicle({analysis["term_counts"]["chronicle"]}) | '
                      f'17th Century({analysis["term_counts"]["time"]}) | '
                      f'Irish({analysis["term_counts"]["irish"]}) | '
                      f'Supernatural({analysis["term_counts"]["supernatural"]})')
                
                if analysis['result_snippets']:
                    print(f'    High-relevance snippets: {len(analysis["result_snippets"])}')
                    for j, snippet in enumerate(analysis['result_snippets'][:2], 1):
                        print(f'      {j}. {snippet["text"][:100]}...')
                        print(f'         Terms: {" | ".join(snippet["matched_terms"][:4])}')
                
                chronicle_search_results['search_results'].append(analysis)
                
                # Identify potential chronicle candidates
                if analysis['relevance_score'] >= 10:
                    print('    🎯 HIGH RELEVANCE - Potential chronicle source identified!')
                    chronicle_search_results['chronicle_candidates'].append(analysis)
                
                # Extract supernatural events if found
                for snippet in analysis['result_snippets']:
                    if any(term in snippet['text'].lower() for term in ['spectral army', 'phantom army', 'ghostly host', 'supernatural army']):
                        chronicle_search_results['supernatural_events'].append({
                            'description': snippet['text'],
                            'source_query': query,
                            'source_engine': engine,
                            'matched_terms': snippet['matched_terms']
                        })
                
                break  # Success, no need to try other engines for this query
                
            else:
                print(f'    ❌ {engine}: Status {response.status_code}')
                
        except Exception as e:
            print(f'    ❌ {engine}: Error - {str(e)}')
    
    time.sleep(3)  # Rate limiting

print('\n' + '=' * 80)
