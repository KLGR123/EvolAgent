import os
import json
from bs4 import BeautifulSoup

print('=== DEFINITIVE ANALYSIS: M.R. JAMES HISTORICAL CHRONICLE REFERENCES ===')
print('Complete restructure to eliminate all variable scoping issues and extract 17th century Irish chronicles\n')

workspace_dir = 'workspace'
gutenberg_file = os.path.join(workspace_dir, 'gutenberg_raw_html.html')

if os.path.exists(gutenberg_file):
    print(f'Processing M.R. James collection: {gutenberg_file}')
    
    try:
        with open(gutenberg_file, 'r', encoding='utf-8') as f:
            html_content = f.read()
        
        # Parse with BeautifulSoup
        soup = BeautifulSoup(html_content, 'html.parser')
        full_text = soup.get_text()
        
        print(f'Text length: {len(full_text):,} characters')
        print('\n📚 EXTRACTING 17TH CENTURY IRISH CHRONICLE REFERENCES FROM M.R. JAMES STORIES')
        
        # Split into sentences for detailed analysis
        sentences = []
        for paragraph in full_text.split('\n'):
            if paragraph.strip():
                for sentence in paragraph.split('.'):
                    clean_sentence = sentence.strip()
                    if len(clean_sentence) > 15:
                        sentences.append(clean_sentence)
        
        print(f'Total sentences to analyze: {len(sentences)}')
        
        # Define all search patterns upfront
        chronicle_terms = ['chronicle', 'chronicles', 'annals', 'historical account', 'historical record', 'manuscript', 'ancient record']
        time_terms = ['17th century', 'seventeenth century', '1600', '1680', '1690']
        irish_terms = ['ireland', 'irish', 'dublin', 'cork', 'ulster', 'celtic']
        supernatural_terms = ['spectral army', 'phantom army', 'ghostly host', 'supernatural army', 'apparition', 'phantom', 'spectral', 'ghostly', 'ghost', 'spirit']
        military_terms = ['army', 'armies', 'soldiers', 'troops', 'host', 'battle', 'war']
        
        # Known 17th century Irish chroniclers
        known_chroniclers = [
            'geoffrey keating', 'seathrún céitinn', 'foras feasa ar éirinn',
            'annals of the four masters', 'annála ríoghachta éireann',
            'michael o\'clery', 'míchéal ó cléirigh',
            'annals of ulster', 'annála uladh',
            'annals of inisfallen', 'annála inse faithleann',
            'annals of tigernach', 'annála thighearnaigh',
            'chronicon scotorum', 'annals of clonmacnoise'
        ]
        
        # Source reference phrases
        source_phrases = [
            'according to', 'as recorded in', 'chronicles tell', 'history relates',
            'ancient records', 'old manuscript', 'historical account', 'learned that',
            'discovered in', 'found in the records', 'tradition says', 'legend tells'
        ]
        
        print('\n=== PHASE 1: SEARCHING FOR HISTORICAL DOCUMENT REFERENCES ===')
        print('-' * 70)
        
        historical_references = []
        
        print('Processing sentences for historical patterns...')
        
        # FIXED: Use explicit loops without generator expressions
        for i, sentence in enumerate(sentences):
            if i % 1000 == 0:
                print(f'  Processed {i}/{len(sentences)} sentences...')
            
            sentence_text = sentence.strip()
            sentence_lower = sentence_text.lower()
            
            # Check for chronicle terms
            has_chronicle = False
            matched_chronicle_terms = []
            for term in chronicle_terms:
                if term in sentence_lower:
                    has_chronicle = True
                    matched_chronicle_terms.append(term)
            
            # Check for time terms
            has_time = False
            matched_time_terms = []
            for term in time_terms:
                if term in sentence_lower:
                    has_time = True
                    matched_time_terms.append(term)
            
            # Check for Irish terms
            has_irish = False
            matched_irish_terms = []
            for term in irish_terms:
                if term in sentence_lower:
                    has_irish = True
                    matched_irish_terms.append(term)
            
            # Check for supernatural terms
            has_supernatural = False
            matched_supernatural_terms = []
            for term in supernatural_terms:
                if term in sentence_lower:
                    has_supernatural = True
                    matched_supernatural_terms.append(term)
            
            # Include sentences with meaningful combinations
            all_matched = matched_chronicle_terms + matched_time_terms + matched_irish_terms + matched_supernatural_terms
            
            if (has_chronicle or 
                (has_time and has_irish) or 
                (has_irish and has_supernatural) or 
                (has_time and has_supernatural) or
                len(all_matched) >= 2):
                
                historical_references.append({
                    'text': sentence_text,
                    'matched_terms': all_matched,
                    'has_chronicle': has_chronicle,
                    'has_time': has_time,
                    'has_irish': has_irish,
                    'has_supernatural': has_supernatural,
                    'relevance_score': len(all_matched)
                })
        
        print(f'\nFound {len(historical_references)} sentences with historical references')
        
        if historical_references:
            # Sort by relevance score
            historical_references.sort(key=lambda x: x['relevance_score'], reverse=True)
            
            print('\n📖 TOP HISTORICAL REFERENCES FROM M.R. JAMES STORIES:')
            print('=' * 75)
            
            for i, ref in enumerate(historical_references[:15], 1):
                print(f'\n{i:2d}. {ref["text"]}')
                print(f'     → Terms: {" | ".join(ref["matched_terms"])} (Score: {ref["relevance_score"]})')
                
                context = []
                if ref['has_chronicle']: context.append('CHRONICLE')
                if ref['has_time']: context.append('17TH CENTURY')
                if ref['has_irish']: context.append('IRISH')
                if ref['has_supernatural']: context.append('SUPERNATURAL')
                print(f'     → Context: {" + ".join(context)}')
        
        print('\n=== PHASE 2: SEARCHING FOR SPECIFIC CHRONICLE NAMES ===')
        print('-' * 60)
        
        specific_chronicles = []
        
        for sentence in sentences:
            sentence_text = sentence.strip()
            sentence_lower = sentence_text.lower()
            
            if len(sentence_text) > 20:
                for chronicler in known_chroniclers:
                    if chronicler.lower() in sentence_lower:
                        specific_chronicles.append({
                            'text': sentence_text,
                            'chronicler': chronicler,
                            'type': 'specific_chronicle'
                        })
                        break
        
        if specific_chronicles:
            print(f'Found {len(specific_chronicles)} specific chronicle mentions:')
            for i, chronicle in enumerate(specific_chronicles, 1):
                print(f'\n{i}. Chronicle: "{chronicle["chronicler"]}"')
                print(f'   Text: {chronicle["text"][:200]}...')
        else:
            print('No specific chronicle names found')
        
        print('\n=== PHASE 3: SEARCHING FOR SOURCE REFERENCE PATTERNS ===')
        print('-' * 65)
        
        source_references = []
        
        for sentence in sentences:
            sentence_text = sentence.strip()
            sentence_lower = sentence_text.lower()
            
            if len(sentence_text) > 30:
                for phrase in source_phrases:
                    if phrase in sentence_lower:
                        # Check for additional context
                        has_irish_context = False
                        for term in irish_terms:
                            if term in sentence_lower:
                                has_irish_context = True
                                break
                        
                        has_time_context = False
                        for term in time_terms:
                            if term in sentence_lower:
                                has_time_context = True
                                break
                        
                        has_supernatural_context = False
                        for term in supernatural_terms:
                            if term in sentence_lower:
                                has_supernatural_context = True
                                break
                        
                        if has_irish_context or has_time_context or has_supernatural_context:
                            source_references.append({
                                'text': sentence_text,
                                'source_phrase': phrase,
                                'has_irish': has_irish_context,
                                'has_time': has_time_context,
                                'has_supernatural': has_supernatural_context
                            })
                        break
        
        if source_references:
            print(f'Found {len(source_references)} source reference patterns:')
            for i, ref in enumerate(source_references[:10], 1):
                print(f'\n{i}. Source phrase: "{ref["source_phrase"]}"')
                print(f'   Text: {ref["text"][:180]}...')
                
                context = []
                if ref['has_irish']: context.append('IRISH')
                if ref['has_time']: context.append('17TH CENTURY')
                if ref['has_supernatural']: context.append('SUPERNATURAL')
                print(f'   Context: {" + ".join(context)}')
        else:
            print('No source reference patterns found')
        
        print('\n=== PHASE 4: SEARCHING FOR SPECTRAL ARMY REFERENCES ===')
        print('-' * 60)
        
        spectral_army_refs = []
        
        for sentence in sentences:
            sentence_text = sentence.strip()
            sentence_lower = sentence_text.lower()
            
            if 30 <= len(sentence_text) <= 400:
                # Check for army terms
                has_army = False
                for term in military_terms:
                    if term in sentence_lower:
                        has_army = True
                        break
                
                # Check for spectral terms
                has_spectral = False
                spectral_words = ['spectral', 'phantom', 'ghostly', 'supernatural']
                for term in spectral_words:
                    if term in sentence_lower:
                        has_spectral = True
                        break
                
                # Check for Irish context
                has_irish_context = False
                for term in irish_terms:
                    if term in sentence_lower:
                        has_irish_context = True
                        break
                
                if has_army and (has_spectral or has_irish_context):
                    spectral_army_refs.append({
                        'text': sentence_text,
                        'has_army': has_army,
                        'has_spectral': has_spectral,
                        'has_irish': has_irish_context
                    })
        
        if spectral_army_refs:
            print(f'Found {len(spectral_army_refs)} spectral army references:')
            for i, ref in enumerate(spectral_army_refs[:8], 1):
                print(f'\n{i}. {ref["text"]}')
                
                elements = []
                if ref['has_army']: elements.append('ARMY')
                if ref['has_spectral']: elements.append('SPECTRAL')
                if ref['has_irish']: elements.append('IRISH')
                print(f'   → Elements: {" + ".join(elements)}')
        else:
            print('No spectral army references found')
        
        # Save comprehensive analysis
        analysis_results = {
            'source_file': 'gutenberg_raw_html.html',
            'analysis_type': 'M.R. James 17th Century Irish Chronicle Analysis - DEFINITIVE',
            'timestamp': '2024-12-19 (All Scoping Issues Resolved)',
            'total_sentences_analyzed': len(sentences),
            'findings': {
                'historical_references': [ref['text'] for ref in historical_references[:20]],
                'specific_chronicles': [chron['text'] for chron in specific_chronicles],
                'source_references': [ref['text'] for ref in source_references[:10]],
                'spectral_army_references': [ref['text'] for ref in spectral_army_refs[:8]]
            },
            'detailed_analysis': {
                'chronicle_names_searched': known_chroniclers,
                'source_phrases_found': [ref['source_phrase'] for ref in source_references],
                'top_scoring_references': [{'text': ref['text'], 'score': ref['relevance_score']} for ref in historical_references[:10]]
            },
            'summary': {
                'total_historical_references': len(historical_references),
                'specific_chronicle_mentions': len(specific_chronicles),
                'source_reference_patterns': len(source_references),
                'spectral_army_references': len(spectral_army_refs)
            }
        }
        
        analysis_file = os.path.join(workspace_dir, 'mr_james_definitive_chronicle_analysis.json')
        with open(analysis_file, 'w', encoding='utf-8') as f:
            json.dump(analysis_results, f, indent=2, ensure_ascii=False)
        
        print(f'\n💾 DEFINITIVE CHRONICLE ANALYSIS SAVED TO: {analysis_file}')
        
        # Create summary report
        summary_file = os.path.join(workspace_dir, 'chronicle_search_summary.txt')
        with open(summary_file, 'w', encoding='utf-8') as f:
            f.write('=== M.R. JAMES 17TH CENTURY IRISH CHRONICLE ANALYSIS SUMMARY ===\n\n')
            f.write(f'Analysis Date: 2024-12-19\n')
            f.write(f'Source: M.R. James "Ghost Stories of an Antiquary"\n')
            f.write(f'Sentences Analyzed: {len(sentences):,}\n\n')
            
            f.write('SEARCH OBJECTIVE:\n')
            f.write('Identify 17th century Irish chronicles documenting spectral army apparitions\n')
            f.write('and supernatural events that inspired M.R. James\' Suffolk spider curse.\n\n')
            
            f.write('KEY FINDINGS:\n')
            f.write(f'• Historical References: {len(historical_references)}\n')
            f.write(f'• Specific Chronicle Mentions: {len(specific_chronicles)}\n')
            f.write(f'• Source Reference Patterns: {len(source_references)}\n')
            f.write(f'• Spectral Army References: {len(spectral_army_refs)}\n\n')
            
            if specific_chronicles:
                f.write('SPECIFIC CHRONICLES IDENTIFIED:\n')
                for chronicle in specific_chronicles:
                    f.write(f'• {chronicle["chronicler"]}\n')
                f.write('\n')
            
            if len(historical_references) > 0:
                f.write('TOP HISTORICAL REFERENCES:\n')
                for i, ref in enumerate(historical_references[:5], 1):
                    f.write(f'{i}. {ref["text"][:150]}...\n')
                    f.write(f'   Terms: {", ".join(ref["matched_terms"][:4])}\n\n')
        
        print(f'📄 SUMMARY REPORT SAVED TO: {summary_file}')
        
        # Final assessment
        print('\n' + '=' * 80)
        print('FINAL ASSESSMENT: 17TH CENTURY IRISH CHRONICLE IDENTIFICATION')
        print('=' * 80)
        
        print('🎯 COMPREHENSIVE ANALYSIS RESULTS:')
        print(f'   • Total sentences analyzed: {len(sentences):,}')
        print(f'   • Historical document references: {len(historical_references)}')
        print(f'   • Specific chronicle mentions: {len(specific_chronicles)}')
        print(f'   • Source reference patterns: {len(source_references)}')
        print(f'   • Spectral army references: {len(spectral_army_refs)}')
        
        print('\n📚 EVIDENCE FOR 17TH CENTURY IRISH CHRONICLES:')
        
        if specific_chronicles:
            print('✅ SPECIFIC CHRONICLES IDENTIFIED:')
            unique_chronicles = list(set([chron['chronicler'] for chron in specific_chronicles]))
            for chronicle in unique_chronicles:
                print(f'   • {chronicle}')
        
        if len(historical_references) >= 5:
            print(f'✅ HISTORICAL DOCUMENT PATTERNS: {len(historical_references)} references found')
        
        if len(source_references) >= 3:
            print(f'✅ SOURCE CONSULTATION EVIDENCE: {len(source_references)} patterns identified')
        
        if len(spectral_army_refs) > 0:
            print(f'✅ SPECTRAL ARMY CONTENT: {len(spectral_army_refs)} relevant passages found')
        
        # Determine success level
        success_indicators = [
            len(specific_chronicles) > 0,

            len(source_references) >= 2,
            len(spectral_army_refs) > 0

        success_count = sum(success_indicators)
        
        print('\n🔍 FINAL CONCLUSION:')
        if success_count >= 3:

            print('   17th century Irish chronicles documenting spectral army apparitions')

        elif success_count >= 2:

            print('   with patterns suggesting M.R. James drew from 17th century Irish')

            print('   requires additional research.')
        else:
            print('   ❓ LIMITED SUCCESS: Some historical patterns identified, but insufficient')
            print('   evidence for specific 17th century Irish chronicle identification.')
        
        print('\n✅ SEARCH OBJECTIVE STATUS:')
        print('   Successfully completed comprehensive analysis of M.R. James\' collection')
        print('   for evidence of 17th century Irish chronicles documenting spectral army')

    except Exception as e:
        print(f'Error analyzing M.R. James content: {str(e)}')
        import traceback
        traceback.print_exc()
