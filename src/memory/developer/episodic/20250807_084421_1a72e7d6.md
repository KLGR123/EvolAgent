# Start with a comprehensive search on M.R. James and The Ash Tree
search_queries = [

research_data = {

    'suffolk_connections': [],
    'historical_events': []
}

print('Conducting initial web searches for background information...')

# Search for academic and scholarly sources first
academic_sources = [
    'https://en.wikipedia.org/wiki/M._R._James',
    'https://en.wikipedia.org/wiki/Ghost_Stories_of_an_Antiquary',
    'https://www.britannica.com/biography/M-R-James'
]

for i, url in enumerate(academic_sources, 1):
    print(f'Accessing source {i}: {url}')
    try:
        response = requests.get(url, headers=headers, timeout=30)
        if response.status_code == 200:
            soup = BeautifulSoup(response.content, 'html.parser')
            
            # Remove scripts and styles
            for element in soup(['script', 'style', 'nav', 'header', 'footer']):
                element.decompose()
            
            # Extract text content
            page_text = soup.get_text()
            lines = (line.strip() for line in page_text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split('  '))
            clean_text = ' '.join(chunk for chunk in chunks if chunk)
            
            print(f'✓ Successfully accessed {url}')
            print(f'  Content length: {len(clean_text):,} characters')
            
            # Check for key terms
            key_terms = ['ash tree', 'suffolk', 'spider', 'curse', '17th century', 'arachnid', 'spectral army']
            found_terms = [term for term in key_terms if term.lower() in clean_text.lower()]
            
            if found_terms:
                print(f'  Found key terms: {found_terms}')
                
                # Extract relevant passages
                text_lower = clean_text.lower()
                for term in found_terms:
                    if term.lower() in text_lower:
                        # Find context around the term
                        term_index = text_lower.find(term.lower())
                        if term_index != -1:
                            context_start = max(0, term_index - 200)
                            context_end = min(len(clean_text), term_index + 200)
                            context = clean_text[context_start:context_end].replace('\n', ' ').strip()
                            
                            research_data['key_findings'].append({
                                'source': url,
                                'term': term,
                                'context': context
                            })
            else:
                print(f'  No key terms found')
            
            # Save the full content for later analysis
            filename = f'source_{i}_{url.split("/")[-1]}.txt'
            filepath = os.path.join(workspace, filename)
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(f'Source: {url}\n')
                f.write(f'Accessed: {time.strftime("%Y-%m-%d %H:%M:%S")}\n')
                f.write('=' * 80 + '\n\n')
                f.write(clean_text)
            
            print(f'  Saved to: {filepath}')
            
        else:
            print(f'❌ Failed to access {url} - Status: {response.status_code}')
    
    except Exception as e:
        print(f'❌ Error accessing {url}: {str(e)}')
    
    print()
    time.sleep(2)  # Be respectful to servers

# Try to get the actual text of "The Ash Tree" from Project Gutenberg
print('Attempting to access the full text of "The Ash Tree" from Project Gutenberg...')

gutenberg_urls = [
