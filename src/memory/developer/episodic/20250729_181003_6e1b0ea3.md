# Get SerpAPI key from environment variables

all_search_results = {}

    "q": query1,

    "engine": "google",

    "safe": "off",
    "num": 15,
    "type": "search"
}

response1 = requests.get("https://serpapi.com/search.json", params=params1)

if response1.status_code == 200:
    results1 = response1.json()
