import os
from bs4 import BeautifulSoup
import re
import json

print("=== MANUAL INSPECTION OF COMIC DATABASE HTML FILES ===")
print("Focusing on files that showed 'Little Monsters' mentions")
print("ComicVine: 15 mentions, MyComicShop: 12 mentions")
print("=" * 70)

# First, let's see what HTML files we have in workspace
workspace_dir = 'workspace'
if os.path.exists(workspace_dir):
    html_files = [f for f in os.listdir(workspace_dir) if f.endswith('.html')]
    print(f"\nHTML files available for inspection: {len(html_files)}")
    for file in html_files:
        print(f"  - {file}")
else:
    print("No workspace directory found")
    exit()

# Focus on the comic database files that showed promising results
target_files = [
    'comicvine_search.html',
    'mycomicshop_search.html',
    'league_of_comic_geeks_search.html'
]

artist_findings = []

for target_file in target_files:
    filepath = os.path.join(workspace_dir, target_file)
    if os.path.exists(filepath):
        print(f"\n{'='*50}")
        print(f"INSPECTING: {target_file}")
        print(f"{'='*50}")
        
        try:
            with open(filepath, 'r', encoding='utf-8') as f:
                html_content = f.read()
            
            print(f"âœ“ File loaded successfully ({len(html_content)} characters)")
            
            # Parse with BeautifulSoup for better text extraction
            soup = BeautifulSoup(html_content, 'html.parser')
            
            # Remove script and style elements for cleaner text
            for script in soup(["script", "style"]):
                script.decompose()
            
            # Get clean text
            text_content = soup.get_text()
            
            # Count 'Little Monsters' mentions
            little_monsters_count = text_content.lower().count('little monsters')
            print(f"'Little Monsters' mentions found: {little_monsters_count}")
            
            if little_monsters_count > 0:
                print(f"\nExtracting context around 'Little Monsters' mentions...")
                
                # Split into lines and find lines containing 'little monsters'
                lines = text_content.split('\n')
                relevant_lines = []
                
                for i, line in enumerate(lines):
                    line_clean = ' '.join(line.strip().split())
                    if 'little monsters' in line_clean.lower() and len(line_clean) > 10:
                        # Include context lines (previous and next lines)
                        context_lines = []
                        for j in range(max(0, i-2), min(len(lines), i+3)):
                            context_line = ' '.join(lines[j].strip().split())
                            if len(context_line) > 5:
                                context_lines.append(context_line)
                        
                        relevant_lines.append({
                            'main_line': line_clean,
                            'context': context_lines,
                            'line_number': i
                        })
                
                print(f"Found {len(relevant_lines)} relevant line groups")
                
                # Show the most promising lines
                for idx, line_group in enumerate(relevant_lines[:5], 1):
                    print(f"\n--- Context Group {idx} ---")
                    print(f"Main line: {line_group['main_line'][:150]}...")
                    print(f"Context ({len(line_group['context'])} lines):")
                    for ctx_line in line_group['context'][:3]:
                        print(f"  {ctx_line[:100]}...")
                
                # Look for artist/creator patterns in the relevant content
                all_relevant_text = ' '.join([lg['main_line'] for lg in relevant_lines])
                
                # Enhanced patterns to find artist information
                artist_patterns = [
                    r'(?i)(?:artist|art by|illustrated by|artwork by|drawn by)[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)',
                    r'(?i)(?:writer|written by|story by|script by)[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)',
                    r'(?i)(?:creator|created by|by)[:\s]+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)',
                    r'(?i)little monsters.*?(?:by|artist|writer)[:\s]*([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)',
                    r'([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*).*?(?:artist|illustrator).*?little monsters',
                    r'little monsters.*?([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*).*?(?:artist|writer|creator)'
                ]
                
                found_creators = []
                for pattern in artist_patterns:
                    matches = re.findall(pattern, all_relevant_text)
                    if matches:
                        print(f"\nPattern '{pattern[:50]}...' found {len(matches)} matches:")
                        for match in matches[:3]:
                            print(f"  - {match}")
                            if len(match.strip()) > 2 and len(match.strip()) < 50:
                                found_creators.append(match.strip())
                
                # Also search in HTML structure for comic-specific elements
                print(f"\nSearching HTML structure for comic-specific elements...")
                
                # Look for common comic database HTML patterns
                comic_elements = soup.find_all(['div', 'span', 'p', 'td'], 
                                             string=re.compile(r'little monsters', re.IGNORECASE))
                
                if comic_elements:
                    print(f"Found {len(comic_elements)} HTML elements mentioning 'Little Monsters'")
                    
                    for idx, element in enumerate(comic_elements[:3], 1):
                        print(f"\nHTML Element {idx}:")
                        print(f"Tag: {element.name}")
                        print(f"Text: {element.get_text()[:100]}...")
                        
                        # Check parent and sibling elements for creator info
                        parent = element.parent
                        if parent:
                            parent_text = parent.get_text()
                            creator_matches = re.findall(r'(?i)(?:artist|writer|creator)[:\s]*([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)', parent_text)
                            if creator_matches:
                                print(f"  Creator info in parent: {creator_matches}")
                                found_creators.extend(creator_matches)
                
                if found_creators:
                    # Remove duplicates and filter reasonable names
                    unique_creators = list(set(found_creators))
                    filtered_creators = []
                    
                    for creator in unique_creators:
                        # Filter out common false positives
                        if (len(creator.split()) >= 2 and 
                            len(creator) < 30 and 
                            not any(word in creator.lower() for word in ['little', 'monsters', 'comic', 'book', 'series'])):
                            filtered_creators.append(creator)
                    
                    if filtered_creators:
                        print(f"\n*** POTENTIAL CREATORS FOUND IN {target_file.upper()} ***")
                        for creator in filtered_creators:
                            print(f"  - {creator}")
                            artist_findings.append({
                                'name': creator,
                                'source': target_file,
                                'confidence': 'Medium'
                            })
                    else:
                        print(f"No valid creator names extracted after filtering")
                else:
                    print(f"No creator patterns found in relevant text")
            else:
                print(f"No 'Little Monsters' mentions found in this file")
                
        except Exception as e:
            print(f"âœ— Error processing {target_file}: {e}")
    else:
        print(f"\n{target_file} not found in workspace")

print(f"\n{'='*70}")
print("MANUAL INSPECTION SUMMARY")
print(f"{'='*70}")

if artist_findings:
    print(f"\n*** ARTISTS/CREATORS IDENTIFIED ***")
    from collections import Counter
    
    # Count frequency of names across sources
    all_names = [finding['name'] for finding in artist_findings]
    name_frequency = Counter(all_names)
    
    print(f"Total potential creators found: {len(artist_findings)}")
    print(f"Unique names: {len(name_frequency)}")
    
    print(f"\nCreator frequency analysis:")
    for name, count in name_frequency.most_common():
        sources = [f['source'] for f in artist_findings if f['name'] == name]
        print(f"  {name}: {count} mention(s) in {', '.join(set(sources))}")
    
    # Save findings
    final_results = {
        'comic_title': 'Little Monsters',
        'search_method': 'Manual HTML inspection',
        'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
        'artist_findings': artist_findings,
        'name_frequency': dict(name_frequency),
        'top_candidate': name_frequency.most_common(1)[0] if name_frequency else None
    }
    
    results_file = 'workspace/little_monsters_manual_inspection_results.json'
    with open(results_file, 'w', encoding='utf-8') as f:
        json.dump(final_results, f, indent=2, ensure_ascii=False)
    
    print(f"\nâœ“ Manual inspection results saved to: {results_file}")
    
    if name_frequency:
        top_candidate = name_frequency.most_common(1)[0]
        print(f"\n*** MOST LIKELY ARTIST: {top_candidate[0]} ***")
        print(f"*** CONFIDENCE: {top_candidate[1]} mention(s) across sources ***")
else:
    print(f"\n*** NO DEFINITIVE ARTIST INFORMATION FOUND ***")
    print(f"Possible reasons:")
    print(f"1. The HTML files may not contain the specific 'Little Monsters' comic we're looking for")
    print(f"2. The comic database search results may be for different comics with similar names")
    print(f"3. Creator information may be in a format not captured by our patterns")
    print(f"4. The specific 'Little Monsters' comic may be very obscure or indie")
    
    print(f"\nRecommendations:")
    print(f"1. Try searching for specific 'Little Monsters' comic series by publisher")
    print(f"2. Look for 'Little Monsters' with specific genre qualifiers (horror, children's, etc.)")
    print(f"3. Check if there are multiple comics with this title")
    print(f"4. Search for ISBN or specific publication details")

print(f"\nManual inspection completed. Check saved results for detailed findings.")

import time
