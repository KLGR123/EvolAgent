        return ""
    # Remove extra whitespace
    cleaned = re.sub(r'\s+', ' ', text)
    # Remove citations/references [1], [2], etc.
    cleaned = re.sub(r'\[\d+\]', '', cleaned)
    # Remove leading/trailing whitespace
    cleaned = cleaned.strip()
    return cleaned
